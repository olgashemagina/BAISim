279bool TLFDetectedItem::LoadXML(TiXmlElement* parent)
280{
281 if (parent == NULL)
282 return false;
283 if (strcmp(parent->Value(), GetName()) != 0)
284 return false;
285
286 this->m_strDetectorName = parent->Attribute("DetectorName");
287 double v;
288 parent->QueryDoubleAttribute("Raiting", &v);
289 this->m_Raiting = v;
290 m_type = parent->Attribute("Type");
291 parent->Attribute("Angle", &this->m_angle);
292 parent->Attribute("Racurs", &this->m_racurs);
293 parent->Attribute("Bw", &this->m_bw);
294 parent->Attribute("Bh", &this->m_bh);
295 if (parent->Attribute("Comment") != NULL)
296 {
297 m_strComment = parent->Attribute("Comment");
298 }
299
300 awpRect r;
301 int iv;
302 parent->Attribute("left", &iv);
303 r.left = iv;
304 parent->Attribute("top", &iv);
305 r.top = iv;
306 parent->Attribute("right", &iv);
307 r.right = iv;
308 parent->Attribute("bottom", &iv);
309 r.bottom = iv;
310
311
312 m_rect.SetRect(r);
313
314 TiXmlElement* child = parent->FirstChildElement();
315 if (child)
316 {
317 if (m_zone == NULL)
318 m_zone = new TLFZone();
319 m_zone->LoadXML(child);
320 }
321
322 return true;
323}

348void TLFDetectedItem::Rotate(awpPoint c, int angle)
349{
350 awpRect awpr = GetBounds().GetRect();
351
352 double fi = angle*AWP_PI / 180.;
353
354 // rotate rect
355 awpPoint p;
356 int w, h;
357 p.X = awpr.left;
358 p.Y = awpr.top;
359 w = awpr.right - awpr.left;
360 h = awpr.bottom - awpr.top;
361 p.X = int(c.X + (p.X - c.X)*cos(fi) + (p.Y - c.Y)*sin(fi));
362 p.Y = int(c.Y - (p.X - c.X)*sin(fi) + (p.Y - c.Y)*cos(fi));
363 awpr.left = p.X;
364 awpr.right = p.X + w;
365 awpr.top = p.Y;
366 awpr.bottom = p.Y + h;
367 m_rect.SetRect(awpr);
368}

403void TLFDetectedItem::Update(ILFDetectEngine* engine, const TLFRect& rect)
404{
405 if (engine == NULL || m_predictor == NULL)
406 return;
407
408 ILFScanner* s = engine->GetScanner(0);
409
410 auto square = rect.Square();
411
412 if (square == 0 && m_predictor->GetPredicted() == NULL)
413 this->m_state = 3;
414 else if (square == 0)
415 {
416 TLFRect* predicted = m_predictor->GetPredicted();
417 if (predicted != NULL)
418 {
419 m_rect.SetRect(predicted->GetRect());
420 this->m_state = 2;
421 }
422 else
423 this->m_state = 3;
424 }
425 else
426 {
427 TLFRect* predicted = m_predictor->GetPredicted();
428 if (predicted != NULL)
429 {
430 m_rect.SetRect(predicted->GetRect());
431 m_state = 2;
432 }
433 else
434 {
435 this->m_predictor->Learn(engine, rect);
436 m_rect.SetRect(rect.GetRect());
437 m_state = 1;
438 }
439 }

488void TLFDetectedItem::SetZone(TLFZone* zone, int w, int h)
489{
490 if (m_zone != NULL)
491 delete m_zone;
492 m_zone = new TLFZone(*zone);
493 // update m_pRect with new settings

497 awpRect _r;

499 _r.top = floor(_rect2D->GetLeftTop().Y*h / 100. + 0.5);
500
501 _r.right = floor(_rect2D->GetRightBottom().X*w / 100. + 0.5);
502 _r.bottom = floor(_rect2D->GetRightBottom().Y*h / 100. + 0.5);
503 m_rect.SetRect(_r);
504 delete _rect2D;
505 }
