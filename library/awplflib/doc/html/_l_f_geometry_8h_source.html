442class TLFAlignedTransform : public TLFObject
443{
444protected:
445 double m_scale_x;
446 double m_scale_y;
447
448 int m_dx;
449 int m_dy;
450
451public:
452
453 TLFAlignedTransform()
454 {
455 m_scale_x = 1.0;
456 m_scale_y = 1.0;
457
458 m_dx = 0;
459 m_dy = 0;
460 }
461
462 TLFAlignedTransform(double scale)
463 : m_scale_x(scale)
464 , m_scale_y(scale)
465 , m_dx(0), m_dy(0){
466 }
467
468 explicit TLFAlignedTransform(double scale_x, double scale_y, int dx, int dy)
469 : m_scale_x(scale_x)
470 , m_scale_y(scale_y)
471 , m_dx(dx)
472 , m_dy(dy) {
473 }
474
475 TLFRect Apply(const TLFRect& rect) const {
476
477 return TLFRect( int(rect.Left() * m_scale_x + m_dx + 0.5),
478 int(rect.Top() * m_scale_y + m_dy + 0.5),
479 int(rect.Width() * m_scale_x + 0.5),
480 int(rect.Height() * m_scale_y + 0.5) );
481 }
482
483 bool operator == (const TLFAlignedTransform& other) const {
484 return m_scale_x == other.m_scale_x && m_scale_y == other.m_scale_y && m_dx == other.m_dx && m_dy == other.m_dy;
485 }
486
487 bool operator != (const TLFAlignedTransform& other) const {
488 return !(*this == other);
489 }
490
491
492 virtual const char* GetName()
493 {
494 return "TLFAlignedTransform";
495 }
496
497public:
498 static TLFAlignedTransform GetTransform(TLFImage* image, double base_width, double base_height, int dx = 0, int dy = 0) {
499 double scale = std::min<double>(image->width() / base_width, image->height() / base_height);
500 return TLFAlignedTransform(scale, scale, dx, dy);
501 }
