866 bool TLFZones::IsPointInContour(int x, int y, TLF2DContour* list, double dx, double dy)
867 {
868  if (list == NULL)
869  return false;
870  if (list->GetNumPoints() == 0)
871  return false;
872  awp2DPoint point0;
873  awp2DPoint point1;
874  int j;
875  double x1, k,b,t;
876  awp2DContour* c = list->GetContour();
877  int count = 0;
878  for (AWPDWORD i = 0; i < c->NumPoints; i++)
879  {
880  j = (i + 1) % c->NumPoints;
881  point0 = c->Points[i];
882  point1 = c->Points[j];
883 
884  point0.X *= dx;
885  point0.Y *= dy;
886  point1.X *= dx;
887  point1.Y *= dy;
888 
889  if ((int)point0.Y == (int)point1.Y)
890  continue;
891  else if (point0.Y > y && point1.Y > y)
892  continue;
893  else if (point0.Y < y && point1.Y < y)
894  continue;
895  else if ((int)AWP_MIN(point0.X, point1.X) > x)
896  continue;
897  else if ((int)AWP_MIN(point0.Y, point1.Y) == (int)y)
898  continue;
899  else if ((int)AWP_MAX(point0.Y, point1.Y) == (int)y)
900  {
901  x1 = point0.Y > point1.Y ? point0.X : point1.X;
902  if (x > x1)
903  count++;
904  }
905  else if (point0.X == point1.X)
906  {
907  count++;
908  }
909  else
910  {
911  k, b;
912  k = (point1.Y - point0.Y) / (point1.X - point0.X);
913  b = point1.Y - k*point1.X;
914  t;
915  t = (y - b) / k;
916  if (t < x)
917  count++;
918  }
919  }
920 
921  return count & 1;
922 }

924 int TLFZones::IsPointInZone(int x, int y, double dx, double dy)
925 {
926  for (int i = 0; i < this->GetCount(); i++)
927  {
928  TLFZone* zone = GetZone(i);
929  if (zone == NULL)
930  continue;
931 
932  if (zone->IsRect())
933  {
934  TLF2DRect* rect = zone->GetRect();
935  if (x > dx*rect->GetLeftTop().X && x < dx*rect->GetRightBottom().X &&
936  y > dy*rect->GetLeftTop().Y && y < dy*rect->GetRightBottom().Y)
937  return zone->GetClassID();
938  }
939  if (zone->IsCircle())
940  {
941  TLF2DLineSegment* s = zone->GetLineSegmnet();
942  awp2DPoint sp = s->GetStart();
943  awp2DPoint ep = s->GetFinish();
944  sp.X *= dx;
945  sp.Y *= dy;
946  ep.X *= dx;
947  ep.Y *= dy;
948  double r1 = sqrt((sp.X - ep.X)*(sp.X - ep.X) + (sp.Y - ep.Y)*(sp.Y - ep.Y));
949  double r2 = sqrt((sp.X - x)*(sp.X - x) + (sp.Y - y)*(sp.Y-y));
950  if (r2 < r1) {
951  return zone->GetClassID();
952  }
953  }
954  if (IsPointInContour(x, y, zone->GetContour(), dx, dy))
955  return zone->GetClassID();
956  }
957  return -1;
