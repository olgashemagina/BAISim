226TLFLabColor* TLFLabColorDatabase::Center()
227{
228 if (this->GetCount() == 0)
229 return NULL;
230
231 AWPDOUBLE l, a, b;
232 l = 0;
233 b = 0;
234 a = 0;
235 int cc = 0;
236 for (int i = 0; i < GetCount(); i++)
237 {
238 TLFLabColor* c = dynamic_cast<TLFLabColor*>(Get(i));
239 if (c != NULL)
240 {
241 AWPDOUBLE ll, aa, bb;
242 c->GetLab(ll,aa,bb);
243 l += ll;
244 a += aa;
245 b += bb;
246 cc++;
247 }
248 }
249 if (cc == 0)
250 return NULL;
251
252 return new TLFLabColor(l/cc, a /cc, b / cc);
253}
254TLFLabColor* TLFLabColorDatabase::Nearlest(TLFLabColor& color, double& dist, int& ind)
255{
256 dist = 100;
257 ind = -1;
258 if (this->GetCount() == 0)
259 return NULL;
260 int index = 0;
261 for (int i = 0; i < GetCount(); i++)
262 {
263 TLFLabColor* c = dynamic_cast<TLFLabColor*>(Get(i));
264 if (c != NULL)
265 {
266 double v = color.CIE76(*c);
267 if (v < dist)
268 {
269 dist = v;
270 index = i;
271 }
272 }
273 }
274 ind = index;
275 return GetColor(index);
