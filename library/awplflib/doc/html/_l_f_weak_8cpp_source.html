77static void convert_16int_512_byte (unsigned int* a, AWPBYTE* b)
78{
79 int c = 0;
80 for (int i = 0; i < 16; i++)
81 {
82 for (int j = 0; j < 32; j++)
83 {
84 unsigned int m = a[i] & bit_mask[j];
85 b[c] = m > 0 ? 1 : 0;
86 c++;
87 }
88 }
89}

92{
93 memset(a, 0, 16*sizeof(int));
94 int c = 0;
95 for (int i = 0; i < 16; i++)
96 {
97 for (int j = 0; j < 32; j++)
98 {
99 if (b[c] > 0)
100 a[i] |= bit_mask[j];
101
102 c++;
103 }
104 }

492void TIEFSWeak::Train()
493{
494 if (m_Class0Counter == 0 || m_Class1Counter == 0)
495 return;
496
497 m_AvgClass0 /= m_Class0Counter;
498 m_AvgClass1 /= m_Class1Counter;
499
500 m_DspClass0 /= m_Class0Counter;
501 m_DspClass1 /= m_Class1Counter;
502
503 m_DspClass0 = m_DspClass0 - m_AvgClass0*m_AvgClass0;
504 m_DspClass1 = m_DspClass1 - m_AvgClass1*m_AvgClass1;
505
506 m_Class0Counter = 0;
507 m_Class1Counter = 0;
508
509 CalcTheta();
510
511 m_DspClass0 = 1.0 / m_DspClass0;
512 m_DspClass1 = 1.0 / m_DspClass1;
513}

515void TIEFSWeak::CalcTheta()
516{
517 if (sqrt(m_DspClass0) / sqrt(m_DspClass1) <= 0)
518 m_Theta = 0;
519 else
520 m_Theta = log(sqrt(m_DspClass0) / sqrt(m_DspClass1));
