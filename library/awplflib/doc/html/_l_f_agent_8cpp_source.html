93#endif

95 auto features = pool_.GetObject();
96
97 // Init features builder
98 size_t batch_begin = batch_size_ * batch;
99 size_t batch_end = batch_begin + batch_size_;
100 batch_end = batch_end < fragments_count ? batch_end : fragments_count;
101
102 // Setting up bounds of batch and alloc memory if needed
103 features->SetBounds(batch_begin, batch_end);
104
105 // Process batch by detector
106 workers_[current_thread]->Detect(img, *features.get());
107
108 // Apply corrections and calc corrections vector
109 correctors_.Apply(*features.get(), features->GetMutableCorrections());
110
111 // No new correctors without supervisor
112 if (supervisor_ && trainer_) {
113 // Process features and check if we can train new corrector(s)
114 trainer_->CollectSamples(scanner, *features.get(), gt_detections);
115
116 }
117
118 // Prepare for NMS
119 for (size_t n = features->frags_begin(); n < features->frags_end(); ++n) {
120 if (features->GetCorrectedResult(n)) {
121#ifdef _OMP_
122#pragma omp critical
123#endif
124 {
125 prior_detections_.emplace_back( scanner->GetFragmentRect(n),
126 features->GetScore(n));
127 }
128 }
129 }
130 }
131
132 if (supervisor_ && trainer_) {
133 trainer_->Train();
134 }
135
136 TItemAttributes attr = { detector_->GetType(),
137 scanner->GetBaseWidth(),
138 scanner->GetBaseHeight(), 0, 0,
139 detector_->GetName() };
140
141 return NonMaximumSuppression(prior_detections_, nms_threshold_, attr);

245static std::vector<TLFDetectedItem> NonMaximumSuppression(
246 std::vector<std::pair<TLFRect, float>>& rects,
247 float overlap_threshold, const TItemAttributes& attrs ) {
248
249 std::vector<TLFDetectedItem> items;
250
251 // Sort by score
252 std::sort(rects.begin(), rects.end(), [](const auto& a, const auto& b) {
253 return a.second > b.second;
254 });
255
256 for (int i = 0; i < rects.size(); ++i) {
257 const auto& [base_rect, base_score] = rects[i];
258
259 if (base_score == 0)
260 continue;
261
262 // Averaging position
263
264 float x = base_rect.Center().X, y = base_rect.Center().Y;
265 float w = base_rect.Width(), h = base_rect.Height();
266
267 int similar_count = 1;
268
269 for (size_t j = i + 1; j < rects.size(); ++j) {
270 auto& [rect, score] = rects[j];
271
272 if (score > 0 && rect.RectOverlap(base_rect) > overlap_threshold) {
273 x += rect.Center().X;
274 y += rect.Center().Y;
275 w += rect.Width();
276 h += rect.Height();
277
278 similar_count++;
279
280
281 // Skip from using in futher tests
282 score = 0;
283 }
284
285 }
286 x /= float(similar_count);
287 y /= float(similar_count);
288 w /= float(similar_count);
289 h /= float(similar_count);
290
291 awpRect r{ x - w / 2.f, y - h / 2.f, x + w / 2.f, y + h / 2.f };
292
293 UUID id;
294 // memset(id, 0, sizeof(UUID));
295 LF_NULL_UUID_CREATE(id);
296 // add object to result list
297
298 {
299 items.emplace_back(std::move(r), base_score, std::string(attrs.type),
300 attrs.angle, attrs.racurs, attrs.base_width, attrs.base_height, std::string(attrs.name), id);
301 }
302
303 }
304
305 std::cout << "Detected objects: " << rects.size() << " NMS: " << items.size() << std::endl;
306
307 return items;
308}
