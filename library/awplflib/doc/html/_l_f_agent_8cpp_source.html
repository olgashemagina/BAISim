259
260

262 std::sort(rects.begin(), rects.end(), [](const auto& a, const auto& b) {
263 return a.score > b.score;
264 });
265
266 // Count of corrected items;
267 int corrected = 0;
268
269 for (int i = 0; i < rects.size(); ++i) {
270 const auto& [base_rect, base_score, is_base_corrected] = rects[i];
271
272 if (base_score < 0)
273 continue;
274
275 // Averaging position
276
277 float x = base_rect.Center().X, y = base_rect.Center().Y;
278 float w = base_rect.Width(), h = base_rect.Height();
279
280 int similar_count = 1;
281 bool is_corrected = is_base_corrected;
282
283 for (size_t j = i + 1; j < rects.size(); ++j) {
284 auto& [rect, score, is_prior_corrected] = rects[j];
285
286 if (score >= 0 && rect.RectOverlap(base_rect) > overlap_threshold) {
287 x += rect.Center().X;
288 y += rect.Center().Y;
289 w += rect.Width();
290 h += rect.Height();
291
292 similar_count++;
293 // Item is corrected if all priors are corrected;
294 is_corrected = is_corrected && is_prior_corrected;
295
296 // Skip from using in futher tests
297 score = -1.f;
298 }
299
300 }
301 x /= float(similar_count);
302 y /= float(similar_count);
303 w /= float(similar_count);
304 h /= float(similar_count);
305
306 awpRect r{ x - w / 2.f, y - h / 2.f, x + w / 2.f, y + h / 2.f };
307
308 UUID id;
309 // memset(id, 0, sizeof(UUID));
310 LF_NULL_UUID_CREATE(id);
311 // add object to result list
312
313 {
314 TLFDetectedItem item(std::move(r), base_score, std::string(attrs.type),
315 attrs.angle, attrs.racurs, attrs.base_width, attrs.base_height, std::string(attrs.name), id);
316 items.push_back({ std::move(item), is_corrected });
317 }
318
319 corrected += is_corrected ? 1 : 0;
320
321 }
322
323 std::cout << "Detected objects: " << rects.size() << " NMS: " << items.size() << " corrs: " << corrected << std::endl;

372std::pair<int, int> CalcStat(const agent::TDetections& gt, const std::vector<TLFAgent::item_t>& dets, float overlap) {
373 //Calc overlap matrix
374 int FP = dets.size();
375 int FN = gt.size();
376
377 unsigned char gt_mask[128] = { 0 };
378
379 for (size_t d = 0; d < dets.size(); ++d) {
380 unsigned char det_found = 0;
381
382 for (size_t r = 0; r < gt.size(); ++r) {
383 if (dets[d].detected.GetBounds().RectOverlap(gt[r]) > overlap) {
384 det_found = 1;
385 gt_mask[r] = 1;
386 }
387 }
388
389 if (det_found)
390 FP--;
391 }
392
393 for (size_t r = 0; r < gt.size(); ++r) {
394 if (gt_mask[r])
395 FN--;
396 }
397
398 return { FP, FN };
399}
