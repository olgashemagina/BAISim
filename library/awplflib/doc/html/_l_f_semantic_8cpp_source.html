176 bool TLFSemanticDictinaryItem::LoadXML(TiXmlElement* parent)
177 {
178  if (parent == NULL)
179  return false;
180  if (strcmp(parent->Value(), GetName()) != 0)
181  return false;
182  this->m_strLabel = parent->Attribute("noun");
183  const char* ch = parent->Attribute("color");
184  if (ch != NULL)
185  {
186  parent->QueryIntAttribute("color", &m_color);
187  }
188  else
189  awpRGBtoWeb(0,255,0, &m_color);
190  const char* id = parent->Attribute("id");
191  if (id != NULL)
192  m_id = id;
193  else
194  {
195  UUID id;
196  LF_UUID_CREATE(id)
197  m_id = LFGUIDToString(&id);
198  }
199  if (!this->m_scanner->LoadXML(parent->FirstChildElement()))
200  return false;
201  const char* zt = parent->Attribute("zoneType");
202  if (zt != NULL)
203  {
204  int value = 0;
205  parent->QueryIntAttribute("zoneType", &value);
206  m_zoneType = (TEZoneTypes)value;
207  }
208  return true;

313  //int w, h;
