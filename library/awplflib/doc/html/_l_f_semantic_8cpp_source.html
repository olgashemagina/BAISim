179TiXmlElement* TLFSemanticDictinaryItem::SaveXML()
180{
181 TiXmlElement* dscr = new TiXmlElement(GetName());
182 dscr->SetAttribute("noun", m_strLabel.c_str());
183 dscr->SetAttribute("color", m_color);
184 dscr->SetAttribute("id", m_id.c_str());
185 dscr->SetAttribute("zoneType", m_zoneType);
186 if (m_scanner != NULL)

193bool TLFSemanticDictinaryItem::LoadXML(TiXmlElement* parent)
194{
195 if (parent == NULL)
196 return false;
197 if (strcmp(parent->Value(), GetName()) != 0)
198 return false;
199 this->m_strLabel = parent->Attribute("noun");
200 const char* ch = parent->Attribute("color");
201 if (ch != NULL)
202 {
203 parent->QueryIntAttribute("color", &m_color);
204 }
205 else
206 awpRGBtoWeb(0,255,0, &m_color);
207 const char* id = parent->Attribute("id");
208 if (id != NULL)
209 m_id = id;
210 else
211 {
212 UUID id;
213 LF_UUID_CREATE(id)
214 m_id = LFGUIDToString(&id);
215 m_needRepairItem = true;
216 }
217 if (!this->m_scanner->LoadXML(parent->FirstChildElement()))
218 return false;
219 const char* zt = parent->Attribute("zoneType");
220 if (zt != NULL)
221 {
222 int value = 0;
223 parent->QueryIntAttribute("zoneType", &value);
224 m_zoneType = (TEZoneTypes)value;
225 }
226 else
227 {
228 m_zoneType = ZTRect;
229 m_needRepairItem = true;
230 }
231 return true;
232}

318bool TLFSemanticDictinary::LoadXML(const char* lpFileName)
319{
320 this->Clear();
321
322 TiXmlDocument doc(lpFileName);
323 if (!doc.LoadFile())
324 return false;
325 TiXmlHandle hDoc(&doc);
326 TiXmlElement* pElem;// = NULL;
327
328 pElem = hDoc.FirstChildElement().Element();
329 if (pElem == NULL)
330 return false;
331 const char* lpValue = pElem->Value();
332 const char* lpName = GetName();
333 if (strcmp(lpValue, lpName) != 0)
334 return false;
335
336 //int w, h;
337 this->m_Description = pElem->Attribute("Description");
338
339
340 for (TiXmlNode* child = pElem->FirstChild(); child; child = child->NextSibling())
341 {
342 TLFSemanticDictinaryItem* di = new TLFSemanticDictinaryItem();
343 if (strcmp(child->Value(), di->GetName()) != 0)
344 {
345 Clear();
346 return false;
347 }
348 if (!di->LoadXML(child->ToElement()))
349 {
350 Clear();
351 return false;
352 }
353 this->Add(di);
354 }
355
356 return true;
