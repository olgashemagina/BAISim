371 int TSCObjectDetector::DetectInRect(awpRect roi)
372 {
373 
374  int res = 0;
375  if (this->m_scanner == NULL)
376  return res;
377  if (this->m_Image.GetImage() == NULL)
378  return res;
379  for (int i = 0; i < m_scanner->GetFragmentsCount(); i++)
380  {
381  awpRect fr = m_scanner->GetFragmentRect(i);
382  awpPoint c;
383  c.X = (fr.left + fr.right) / 2;
384  c.Y = (fr.top + fr.bottom) / 2;
385  if (c.X >= roi.left && c.X <= roi.right &&
386  c.Y >= roi.top && c.Y <= roi.bottom)
387  {
388  if (!m_scanner->GetFragment(i)->HasObject)
389  {
390 
391  bool has_object = true;
392  double scale = (double)(m_scanner->GetFragmentRect(i).right - m_scanner->GetFragmentRect(i).left)/(double)this->m_baseWidth;
393  m_scanner->GetFragment(i)->raiting = 0;
394  for (int j = 0; j < m_Strongs.GetCount(); j++)
395  {
396  TCSStrong* strong = (TCSStrong*)m_Strongs.Get(j);
397  strong->Setup(m_scanner->GetFragment(i)->Rect, m_baseWidth);
398  double err = 0;
399  if (strong->Classify(&m_Image, err) == 0)

401  has_object = false;

406

680 int TLFFGBGDetector::ClassifyRect(awpRect Fragmnet, double* err, int* vect)
681 {
682  if (m_Image.GetImage() == NULL || m_Image.GetIntegralImage() == NULL)
683  return 0;
684 
685 
686  int w = Fragmnet.right - Fragmnet.left;
687  int h = Fragmnet.bottom - Fragmnet.top;
688  double s = w*h;
689  double value = m_Image.CalcLnSum(Fragmnet.left, Fragmnet.top, w, h);
690  value /= s;
691  double v1 = err[0] / (double)this->m_counter;
