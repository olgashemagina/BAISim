69/* 128 bit GUID to human-readable string */
70static char * guid_to_str(GUID* id, char * out) {
71 int i;
72 char * ret = out;
73 out += sprintf(out, "%.8lX-%.4hX-%.4hX-", id->Data1, id->Data2, id->Data3);
74 for (i = 0; i < sizeof(id->Data4); ++i) {
75 out += sprintf(out, "%.2hhX", id->Data4[i]);
76 if (i == 1) *(out++) = '-';
77 }
78 return ret;
79}

82TiXmlElement* ILFObjectDetector::TDescription::SaveXML() const
83{
84 TiXmlElement* sample_desc = new TiXmlElement("LFSampleDescriptor");
85 sample_desc->SetAttribute("Result", result);
86 sample_desc->SetDoubleAttribute("Score", score);
87
88 for (const auto& strong : descs) {
89 TiXmlElement* stage_desc = new TiXmlElement("LFStageDescriptor");
90 stage_desc->SetDoubleAttribute("Score", strong.score);
91 stage_desc->SetAttribute("Result", strong.result);
92
93 for (const auto& weak : strong.weaks) {
94
95 TiXmlElement* weak_desc = new TiXmlElement("LFWeakDescriptor");
96 weak_desc->SetAttribute("Result", weak.result);
97
98 TiXmlElement* features_desc = new TiXmlElement("LFFeatureDescriptor");
99 const auto& feature = weak.feature;
100 features_desc->SetAttribute("Size", feature.features.size());
101 features_desc->SetDoubleAttribute("Value", feature.value);
102
103 if (feature.parent) {
104 auto rect = feature.parent->GetRect();
105 features_desc->SetAttribute("RectLeft", rect.Left());
106 features_desc->SetAttribute("RectTop", rect.Top());
107 features_desc->SetAttribute("RectRight", rect.Right());
108 features_desc->SetAttribute("RectBottom", rect.Bottom());
109 }
110
111
112 if (!weak.feature.features.empty()) {
113 auto data = std::accumulate(
114 std::next(weak.feature.features.begin()),
115 weak.feature.features.end(),
116 std::to_string(weak.feature.features.front()),
117 [](const auto& a, const auto& b) {
118 return a + " " + std::to_string(b);
119 }
120 );
121
122 TiXmlText* features_data = new TiXmlText(data);
123 features_desc->LinkEndChild(features_data);
124 }
125
126 weak_desc->LinkEndChild(features_desc);
127 stage_desc->LinkEndChild(weak_desc);
128 }
129 sample_desc->LinkEndChild(stage_desc);
130
131 }
132 return sample_desc;
133}

327bool TSCObjectDetector::AddStrong(ILFStrong* strong)
328{
329 if (strong == NULL)
330 return false;
331 TCSStrong* strong0 = dynamic_cast<TCSStrong*>(strong);
332 if (strong0 == NULL)
333 false;
334 int n = 1;
335 for (int k = 0; k < n; k++)
336 {
337 TCSStrong* st = new TCSStrong();
338 for (int i = 0; i < strong0->GetCount(); i++)
339 {
340 ILFWeak* iweak = strong0->GetWeak(i);
341 TCSWeak* weak = dynamic_cast<TCSWeak*>(iweak);
342 TCSWeak* w = new TCSWeak(*weak);
343 st->AddWeak(w);
344 }
345 st->SetThreshold(strong->GetThreshold());
346 this->m_Strongs.Add(st);
347 }
348 return true;
349}

351// classification
352int TSCObjectDetector::ClassifyRect(awpRect fragment)
353{
354 double score = FLT_MAX;
355 double scale_x = (fragment.right - fragment.left) / double(m_baseWidth);
356 double scale_y = (fragment.bottom - fragment.top) / double(m_baseHeight);
357 double scale = std::min<double>(scale_x, scale_y);
358 bool has_object = true;
359
360 TDescription description;
361
362 if (desc_callback_) {
363 description.descs.reserve(m_Strongs.GetCount());
364 }
365
366 TLFAlignedTransform transform = TLFAlignedTransform(scale, scale, fragment.left, fragment.top);
367 for (int j = 0; j < m_Strongs.GetCount(); j++)
368 {
369
370 TCSStrong* strong = (TCSStrong*)m_Strongs.Get(j);
371
372 auto desc = strong->Classify(&m_Image, transform);
373
374 if (desc.result != 0) {
375 score = std::min<double>(desc.score, score);
376 } else {
377 has_object = false;
378 score = desc.score;
379 }
380
381 if (desc_callback_) {
382 //Gather
383 description.descs.emplace_back(std::move(desc));
384 }
385 else if (!has_object) {
386 //Stop searching object;
387 break;
388 }
389
390 }
391
392 description.score = score;
393 description.result = has_object;
394
395 if (desc_callback_) {
396 desc_callback_(0, TLFBounds{0, 0, fragment}, description);
397 }
398
399 return has_object;

403int TSCObjectDetector::Detect()
404{
405
406 if (this->m_scanner == NULL)
407 return -1;
408 if (this->m_Image.GetImage() == NULL)
409 return -1;
410
411 m_objects.Clear();
412
413#ifdef _OMP_
414#pragma omp parallel for num_threads(omp_get_max_threads())
415#endif
416 for (int i = 0; i < m_scanner->GetFragmentsCount(); i++)
417 {
418 TDescription description;
419
420 if (desc_callback_) {
421 description.descs.reserve(m_Strongs.GetCount());
422 }
423
424 bool has_object = true;
425 awpRect rect = m_scanner->GetFragmentRect(i);
426
427 double scale_x = (rect.right - rect.left) / double(this->m_baseWidth);
428 double scale_y = (rect.bottom - rect.top) / double(this->m_baseHeight);
429 double scale = std::min<double>(scale_x, scale_y);
430
431 TLFAlignedTransform transform(scale, scale, rect.left, rect.top);
432 double score = FLT_MAX;
433
434 for (int j = 0; j < m_Strongs.GetCount(); j++)
435 {
436 ILFStrong* strong = static_cast<ILFStrong*>(m_Strongs.Get(j));
437
438 auto desc = strong->Classify(&m_Image, transform);
439
440 //Check If object detected;
441 if (desc.result != 0) {
442 score = std::min<double>(desc.score, score);
443 }
444 else {
445 has_object = false;
446 score = desc.score;
447 }
448
449 if (desc_callback_) {
450 //Gather
451 description.descs.emplace_back(std::move(desc));
452 }
453 else if (!has_object) {
454 //Stop searching object;
455 break;
456 }
457 }
458
459 description.score = score;
460 description.result = has_object;
461
462
463 if (desc_callback_) {
464 desc_callback_(i, *m_scanner->GetFragment(i), description);
465 }
466
467 if (has_object)

469 //object detected
470 UUID id;
471 LF_NULL_UUID_CREATE(id);
472 // .
473 TLFDetectedItem* de = new TLFDetectedItem(rect, score, this->m_Type, this->m_Angle,

504 for (int i = 0; i < m_scanner->GetFragmentsCount(); i++)
505 {
506 awpRect fr = m_scanner->GetFragmentRect(i);
507 awpPoint c;
508 c.X = (fr.left + fr.right) / 2;
509 c.Y = (fr.top + fr.bottom) / 2;
510 if (c.X >= roi.left && c.X <= roi.right &&
511 c.Y >= roi.top && c.Y <= roi.bottom)
512 {
513 TDescription description;
514

532 ILFStrong* strong = static_cast<ILFStrong*>(m_Strongs.Get(j));
