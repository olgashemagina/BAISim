8int TBaselineCorrector::Process(const float* features, size_t size) const {
9 // Process corrections and push [0, 1] to corrections.
10 // 0 - No Corrections Needed
11 // 1 - Correct result
12
13 if (size < pca_proj_.rows())
14 return -1;
15
16 std::vector<float> projected(pca_proj_.cols(), 0);
17
18
19 // proj = (data - centre) * pca_proj;
20 for (size_t p = 0; p < pca_proj_.cols(); ++p) {
21 float proj_value = 0;
22 for (size_t f = 0; f < pca_proj_.rows(); ++f) {
23 const float* proj_row = pca_proj_.GetRow(f);
24 proj_value += (features[f] - center_[f]) * proj_row[p];
25 }
26 projected[p] = proj_value;
27 }
28
29 // Distance to clusters;
30 size_t best_centroid = 0;
31 float best_dist = FLT_MAX;
32 for (size_t c = 0; c < centroids_.size(); ++c) {
33 const auto& centroid = centroids_[c];
34 float dist = 0;
35 for (size_t p = 0; p < pca_proj_.cols(); ++p) {
36 dist += std::powf(centroid[p] - projected[p], 2);
37 }
38
39 if (best_dist > dist) {
40 best_centroid = c;
41 best_dist = dist;
42 }
43 }
44
45 // Project threshold;
46 float thres_proj = 0;
47 for (size_t p = 0; p < pca_proj_.cols(); ++p) {
48 thres_proj += fisher_.GetRow(p)[best_centroid] * projected[p];
49 }
50
51 return (thres_[best_centroid] >= thres_proj) ? 1 : 0;
52
