200  FreeImages();

215  throw 0;

217  catch(...)
218  {
219  res = false;
220  }
221  if (pDump != NULL)

467 void TLFImage::Show( HDC dc, int x, int y, int w, int h, int from_x, int from_y )
468 {
469  if (m_pImage == NULL) return;
470  if( m_pImage->sSizeX > 0 )
471  {
472  unsigned char buffer[sizeof(BITMAPINFOHEADER) + 1024];
473  BITMAPINFO* bmi = (BITMAPINFO*)buffer;
474  int bmp_w = m_pImage->sSizeX;
475  int bmp_h = m_pImage->sSizeY;
476 
477 
478  BITMAPINFOHEADER* bmih = &(bmi->bmiHeader);
479 
480  memset( bmih, 0, sizeof(*bmih));
481  bmih->biSize = sizeof(BITMAPINFOHEADER);
482  bmih->biWidth = bmp_w;
483  bmih->biHeight = -abs(bmp_h);
484  bmih->biPlanes = 1;
485  bmih->biCompression = BI_RGB;
486 
487  if (m_pImage->dwType == AWP_BYTE && m_pImage->bChannels == 3)
488  bmih->biBitCount = 24;
489  else
490  {
491  bmih->biBitCount = 8;
492  RGBQUAD* palette = bmi->bmiColors;
493  int i;
494  for( i = 0; i < 256; i++ )
495  {
496  palette[i].rgbBlue = palette[i].rgbGreen = palette[i].rgbRed = (BYTE)i;
497  palette[i].rgbReserved = 0;
498  }
499 
500  }
501  int sw = MAX( MIN( bmp_w - from_x, w ), 0 );
502  int sh = MAX( MIN( bmp_h - from_y, h ), 0 );
503 
504  SetDIBitsToDevice(
505  dc, // handle to DC
506  x, // x-coord of destination upper-left corner
507  y, // y-coord of destination upper-left corner
508  sw, // source rectangle width
509  sh, // source rectangle height
510  from_x, // x-coord of source lower-left corner
511  from_y, // y-coord of source lower-left corner
512  from_y, // first scan line in array
513  sh, // number of scan lines
514  m_pImage->pPixels , // array of DIB bits
515  (BITMAPINFO*)bmi, // bitmap information
516  DIB_RGB_COLORS ); // RGB or palette indexes
517  }
518 }
