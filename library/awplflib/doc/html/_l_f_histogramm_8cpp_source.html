86void TLFHistogramm::AddElememt(double v)
87{
88 if (v < this->m_min)
89 {
90 m_data[0]++;
91 return;
92 }
93 if (v > this->m_max)
94 {
95 m_data[this->m_num_bins - 1]++;
96 return;
97 }
98 int index = 0;
99 index = (int)floor((this->m_num_bins - 1)*(v - m_min) / (m_max - m_min));
100 m_data[index]++;
101}

128double TLFHistogramm::GetSumToBinForward(int bin)
129{
130 if (bin < 0 || bin > this->m_num_bins)
131 return -1;
132 double sum = 0;
133 double sum1 = 0;
134 for (int i = 0; i < this->m_num_bins; i++)
135 sum += this->m_data[i];
136 if (sum == 0)
137 return -1;
138 for (int i = 0; i < bin; i++)
139 sum1 += this->m_data[i];
140 return 100 * sum1 / sum;
141}
142double TLFHistogramm::GetSumToBinBackward(int bin)
143{
144 if (bin < 0 || bin > this->m_num_bins)
145 return -1;
146 double sum = 0;
147 double sum1 = 0;
148 for (int i = 0; i < this->m_num_bins; i++)
149 sum += this->m_data[i];
150 if (sum == 0)
151 return -1;
152 for (int i = this->m_num_bins - 1; i >= bin; i--)
153 sum1 += this->m_data[i];
154 return 100 * sum1 / sum;

237bool TLFHistogramm::LoadXML(TiXmlElement* parent)
238{
239 if (parent == NULL)
240 return false;
241 if (strcmp(parent->Value(), GetName()) != 0)
242 return false;
243 parent->QueryDoubleAttribute("min", &this->m_min);
244 parent->QueryDoubleAttribute("max", &this->m_max);
245 parent->QueryIntAttribute("num", &this->m_num_bins);
246 if (m_num_bins < 0)
247 return false;
248 if (m_data != NULL)
249 free(m_data);
250 if (m_num_bins == 0)
251 return true;
252 m_data = (double*)malloc(m_num_bins*sizeof(double));
253 int c = 0;
254 TLFDouble d(0.);
255 for(TiXmlNode* child = parent->FirstChild(); child; child = child->NextSibling() )
256 {
257 TiXmlElement* e = child->ToElement();
258 if(!d.LoadXML(e))
259 return false;
260 m_data[c] = d.Get();
261 c++;
262 }
263
264 return true;
265}
