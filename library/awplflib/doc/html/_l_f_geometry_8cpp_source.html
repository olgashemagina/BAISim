1417 grad = 90;
1418 else
1419 grad = 45;
1420 }

1424 grad = 0;

1426 grad = 90;
1427 else
1428 grad = 135;

1441 g[1] = b[(y0 - 1)*img->sSizeX + x0 + 1];
