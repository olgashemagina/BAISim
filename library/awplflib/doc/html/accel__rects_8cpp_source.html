747 void DecodeMeasures(const positions_t& stages_positions,
748 const int* table,
749 const uint16_t* measures,
750 int table_size,
751 int min_stages,
752 triggered_t& triggered,
753 features_t* features = nullptr) {
754
755 triggered.resize(table_size, -1);
756 int stages_plus_2 = int(stages_positions.size()) + 1;
757
758
759
760 if (features) {
761 features->resize(stages_positions.back() * table_size, 0);
762 float* measures_output = features->data();
763 size_t measures_size = features->size();
764
765 for (int t = 0; t < table_size; ++t, table += stages_plus_2) {
766 // First number is number of stage triggered or -1;
767
768 const TStageInfo* info = (const TStageInfo*)table;
769 float score = info->score;
770 int triggered_stage = info->triggered;
771 triggered[t] = triggered_stage;
772
773 const int* stage_offset = (const int*)(info + 1);
774 int num_stages = triggered_stage < 0 ? stages_positions.size() - 1 : std::max<int>(triggered_stage + 1, min_stages);
775
776 for (int s = 0; s < num_stages; ++s, stage_offset ++) {
777 int stage_size = stages_positions[s + 1] - stages_positions[s];
778
779 const uint16_t* stage_measures = measures + *stage_offset;
780
781 for (int i = 0; i < stage_size; i++) {
782 measures_output[i] = float(stage_measures[i] / MEASURE_COMPRESS_FACTOR);
783 }
784 }
785 measures_output += stages_positions.back();
786 measures_size -= stages_positions.back();
787 }
788 }
789 else {
790 for (int t = 0; t < table_size; ++t, table += stages_plus_2) {
791
792 const TStageInfo* info = (const TStageInfo*)table;
793 float score = info->score;
794 int triggered_stage = info->triggered;
795 triggered[t] = triggered_stage;
796 }
797
798 }
