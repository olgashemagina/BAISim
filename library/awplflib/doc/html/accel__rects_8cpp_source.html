220 template<typename T>
221 class Pool {
222 public:
223 Pool(size_t limits = 0) : limits_(limits) {}
224
225 template<typename ...A>
226 T* Create(A &&... args) {
227 std::unique_lock<std::mutex> locker(mtx_);
228
229 while (limits_ > 0 && objects_.size() >= limits_ && pool_.size() == 0) {
230 cv_.wait(locker);
231 }
232
233 if (pool_.empty()) {
234 objects_.emplace_back(std::make_unique<T>(args...));
235 pool_.push_back(objects_.back().get());
236 }
237
238 tasks_count_.fetch_add(1, std::memory_order_relaxed);
239
240 T* obj = pool_.back();
241 pool_.pop_back();
242 return obj;
243 }
244
245 void Release(T* obj) {
246 std::lock_guard<std::mutex> locker(mtx_);
247 pool_.push_back(obj);
248 tasks_count_.fetch_sub(1, std::memory_order_relaxed);
249 cv_.notify_one();
250 }
251
252 int tasks_count() const { return tasks_count_.load(std::memory_order_acquire); }
253
254 void WaitTask() {
255 std::unique_lock<std::mutex> locker(mtx_);
256
257 if (objects_.size() > pool_.size()) {
258 cv_.wait(locker);
259 }
260 }
261
262 void WaitAllTasks() {
263 std::unique_lock<std::mutex> locker(mtx_);
264
265 while (objects_.size() > pool_.size()) {
266 cv_.wait(locker);
267 }
268 }
269
270 private:
271 std::vector<std::unique_ptr<T>> objects_;
272
273 size_t limits_ = 0;
274
275 std::vector<T*> pool_;
276 std::mutex mtx_;
277 std::condition_variable cv_;
278
279 std::atomic_int32_t tasks_count_ = 0;

641 bool Enqueue(const std::shared_ptr<Core>& core, const integral_ptr& integral, const data_ptr_var& data, const Kernels& kernels, const transforms_t& transforms, callback_t callback) {
642 cl_int err = 0;
643
644 cb_ = std::move(callback);
645
646 integral_data_ = integral;
647 core_ = core;
648
649 if (std::holds_alternative<features_ptr>(data)) {
650 features_data_ = std::get<features_ptr>(data);
651 } else if (std::holds_alternative<detector_ptr>(data)) {
652 detector_data_ = std::get<detector_ptr>(data);
653 features_data_ = features_ptr(detector_data_, &detector_data_->features_data());
654 }
655
656
657 int stages_count = detector_data_ ? detector_data_->stages().count() : 0;
658 int feats_count = features_data_->features().count() ;
659
660 dims_ = { int(transforms.size()), stages_count, feats_count };
661
662 //trans_ = std::move(transforms);
663 trans_ = transforms;
664 feats_.resize(feats_count * trans_.size(), 0);
665 dets_.resize(stages_count * trans_.size(), 0);
666
667 if (!ResizeBufferIfNeeded(core_->context(), trans_mem_, CL_MEM_READ_ONLY, sizeof(cl_float4) * trans_.size()))
668 return false;
669
670 if (!ResizeBufferIfNeeded(core_->context(), feats_mem_, CL_MEM_READ_WRITE, sizeof(cl_ushort) * feats_.size()))
671 return false;
672
673
674 cl::Event feats_evt, dets_evt, read_feats_evt, read_dets_evt, evt_write;
675
676
677 CL_CHECK_BOOL(core->queue().enqueueWriteBuffer(trans_mem_, CL_FALSE, 0, sizeof(cl_float4) * trans_.size(), trans_.data(), 0, &evt_write));
678
679 waiting_evts_.push_back(evt_write);
680
681 cl::Kernel feats_kernel = kernels.feats_kernel();
682 CL_CHECK_BOOL(feats_kernel.setArg(0, integral_data_->integral()));
683 CL_CHECK_BOOL(feats_kernel.setArg<int>(1, feats_count));
684 CL_CHECK_BOOL(feats_kernel.setArg(2, features_data_->rects()));
685 CL_CHECK_BOOL(feats_kernel.setArg(3, trans_mem_));
686 CL_CHECK_BOOL(feats_kernel.setArg(4, feats_mem_));
687
688 CL_CHECK_BOOL(core->queue().enqueueNDRangeKernel(feats_kernel, cl::NullRange, cl::NDRange(feats_count, trans_.size()), cl::NullRange, &waiting_evts_, &feats_evt));
689
690 waiting_evts_.clear();
691 waiting_evts_.push_back(feats_evt);
692
693 CL_CHECK_BOOL(core->queue().enqueueReadBuffer(feats_mem_, CL_FALSE, 0, sizeof(cl_ushort) * feats_.size(), feats_.data(), &waiting_evts_, &evt_complete_));
694
695 //Also run detector stages
696 if (detector_data_) {
697 if (!ResizeBufferIfNeeded(core_->context(), dets_mem_, CL_MEM_WRITE_ONLY, sizeof(cl_uchar) * dets_.size()))
698 return false;
699
700 cl::Kernel dets_kernel = kernels.dets_kernel();
701
702 CL_CHECK_BOOL(dets_kernel.setArg<int>(0, feats_count));
703 CL_CHECK_BOOL(dets_kernel.setArg<int>(1, stages_count));
704 CL_CHECK_BOOL(dets_kernel.setArg(2, feats_mem_));
705 CL_CHECK_BOOL(dets_kernel.setArg(3, detector_data_->stages_data().positions()));
706 CL_CHECK_BOOL(dets_kernel.setArg(4, detector_data_->stages_data().responses()));
707 CL_CHECK_BOOL(dets_kernel.setArg(5, detector_data_->stages_data().weights()));
708 CL_CHECK_BOOL(dets_kernel.setArg(6, detector_data_->stages_data().thresholds()));
709 CL_CHECK_BOOL(dets_kernel.setArg(7, dets_mem_));
710
711
712 waiting_evts_.clear();
713 waiting_evts_.push_back(feats_evt);
714
715 CL_CHECK_BOOL(core->queue().enqueueNDRangeKernel(dets_kernel, cl::NullRange, cl::NDRange(stages_count, trans_.size()), cl::NullRange, &waiting_evts_, &dets_evt));
716
717 waiting_evts_.clear();
718 waiting_evts_.push_back(dets_evt);
719
720 CL_CHECK_BOOL(core->queue().enqueueReadBuffer(dets_mem_, CL_FALSE, 0, sizeof(cl_uchar) * dets_.size(), dets_.data(), &waiting_evts_, &read_dets_evt));
721
722 //Adding marker of finishing reading
723 waiting_evts_.clear();
724 read_feats_evt = evt_complete_;
725 waiting_evts_.push_back(read_dets_evt);
726 waiting_evts_.push_back(read_feats_evt);
727 CL_CHECK_BOOL(core->queue().enqueueMarkerWithWaitList(&waiting_evts_, &evt_complete_));
728 }
729
730 CL_CHECK_BOOL(evt_complete_.setCallback(CL_COMPLETE, &Task::EvtCallback, this));
731
732
733 return true;
734 }
