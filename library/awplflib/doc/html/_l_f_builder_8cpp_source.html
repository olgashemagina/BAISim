29 return true;
30 if (strExt == ".jpeg")

32 if (strExt == ".png")

177 // начальная инициализация детектора
178 if (!InitDetector())

183

185 // главный цикл процесса
186 while (BuildBkground())
187 {
188 //загрузка

194 //обновление

196 break;
197 m_AdaBoost.InitFeatures();
198 cascade++;
199 };
200 return true;

568 r.top = 0;

570 if (d->ClassifyRect(r) == 0)
571 {
572 //NumFailed++;
573 m_AdaBoost.DbgMsg(names[i] + " Failed.\n");
574 }
575
576 }
