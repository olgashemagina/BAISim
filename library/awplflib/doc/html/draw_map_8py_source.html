730def interpol(X, y, nodes, r):
731 # interpol calculates value of def y defined in data points X in each
732 # node from nodes.
733 #
734 # Inputs:
735 # X is n-by-d data matrix with one data point in each row.
736 # y is n-by-1 vector with values of def. y(i) contains def
737 # value for point X(i, :).
738 # nodes is m-by-d matrix of nodes to calculate def values.
739 # r is smoothing parameter for def calculation.
740 #
741 # memSize is constant for quick
742
743 # Calculate variances for all attributes
744 smooth = -1 / (r * np.mean(np.var(X, axis=0), axis=0))
