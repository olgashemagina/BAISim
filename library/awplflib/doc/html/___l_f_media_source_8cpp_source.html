170static bool _GetImageNames(const char* path, TLFStrings& names)
171{
172 names.clear();
173 TLFStrings allFiles;
174 if (!LFGetDirFiles(path, allFiles))
175 return false;
176 for (unsigned int i = 0; i < allFiles.size(); i++)
177 {
178 if (LFIsImageFile(allFiles[i].c_str()))
179 names.push_back(allFiles[i]);
180 }
181 return true;
182}

258bool TLFImagesSource::DecodeCurrentFrame()
259{
260 const char* lpFileName = m_fileNames[m_currentFrame].c_str();
261 if (!m_image.LoadFromFile(lpFileName))
262 return false;
263 if (m_OnNewFrame != NULL)
264 m_OnNewFrame(m_currentFrame, &m_image);
265 return true;
