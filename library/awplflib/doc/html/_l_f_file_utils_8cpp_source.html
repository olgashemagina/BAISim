263static char * guid_to_str(UUID* id, char * out) {
264 unsigned int i;
265 char * ret = out;
266 out += sprintf(out, "%.8lX-%.4hX-%.4hX-", id->Data1, id->Data2, id->Data3);
267 for (i = 0; i < sizeof(id->Data4); ++i) {
268 out += sprintf(out, "%.2hhX", id->Data4[i]);
269 if (i == 1) *(out++) = '-';
270 }
271 return ret;
272}
