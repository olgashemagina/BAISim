39 all = np.concatenate((CRLS, WRLS), axis=0)

53 # Центрирование данных
54 CRLS = CRLS - self.centre

58 match setPC:
59 case 1:
60 both = np.concatenate((CRLS, WRLS), axis=0)

74
75 elif space == "reduced":
76 # PCA для уменьшения размерности
77 pca = PCA(n_components=numPC, whiten=whitening)
78 match setPC:
79 case 1:

81 pca.fit(both)

83 pca.fit(CRLS)

88 self.project = pca.components_.T
89 else:
90 raise ValueError("Invalid value for space. Valid options are: 'original' or 'reduced'.")
91
92 # Кластеризация
93 kmeans = KMeans(n_clusters=numClust, random_state=0, n_init="auto").fit(WRLSR)

97 for i in range(numClust):
98 if not (labels == i).sum():
99 labels = [x-1 if x > i else x for x in labels]
100 self.centroids = np.delete(self.centroids, i, axis=0)

102 self.numClust = self.centroids.shape[0]

105 covCRLSR = np.cov(CRLSR.T)

107 self.FD = np.zeros((numPC, self.numClust))
108 for k in range(self.numClust):

110 if tmp.shape[0] > 1:
111 cov1 = np.cov(tmp.T)
112 else:
113 cov1 = np.zeros((WRLSR.shape[1], WRLSR.shape[1]))
114 self.FD[:, k] = np.linalg.solve(covCRLSR + cov1, meanCRLSR - np.mean(tmp, axis=0))
115
116 self.FD /= np.sqrt(np.sum(self.FD ** 2, axis=0))

122 )

125 data = (self.WRLS - self.centre) @ self.project,
126 centroids = self.centroids,
127 FD = self.FD

129

135 "Distribution of training set for cluster %03d" % k, "far")

166 def splitToClusters(data, centroids, FD):
167 #% Inputs:
168 #% data is set of data points, one point in row.
169 #% centroids matrix of centroids for clusters with one centroid per row
170 #% FD is matrix of Fisher's discriminant directions. One direction per
171 #% column.
172 #% Outputs:
173 #% res is cell array. Each cell contains projections of datapoints of one
174 #% cluster onto corresponding FD of this cluster.
175
176 # % Create array for output
177 #print(centroids.shape, data.shape)
178 nClust = FD.shape[1]
179 res = []
180 # % Calculate distances to centroids
181 dist = scipy.spatial.distance.cdist(data, centroids)
182 #print(dist.shape); raise
183 # dist = np.sum(data ** 2, axis=1) + np.sum(centroids.conj().T ** 2, axis=0)
184 # dist -= 2 * data @ centroids.conj().T
185 # % Find the minimal distance
186 lab = np.argmin(dist, axis=1)
187 # % Identify all points with the closest centroids and calculate
188 # % projections on FD
189 for k in range(nClust):
190 proj = data[lab == k] @ FD[:, k]
191 res.append(proj)
192 return res
