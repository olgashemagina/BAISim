399bool TLFTileScanner::Scan(int w, int h)
400{
401 fragments_.clear();
402 double overlap = (double)((TLFParameter*)(this->m_Params.Get(0)))->GetValue();
403 overlap = 1 - overlap / 100.;
404
405 int num_x = (int)floor((double)w / (overlap*this->m_BaseWidth));
406 int num_y = (int)floor((double)h / (overlap*this->m_BaseHeight));
407
408 size_t fragments = num_x * num_y;
409
410 fragments_.reserve(fragments);
411
412 for (int i = 0; i < num_y; i++)
413 {
414 for (int j = 0; j < num_x; j++)
415 {
416 if (overlap*j*this->m_BaseWidth + m_BaseWidth > w || overlap*i*this->m_BaseHeight + m_BaseHeight > h)
417 continue;
418
419 awpRect rect;
420
421 rect.left = AWPSHORT(overlap*j*this->m_BaseWidth + 0.5);
422 rect.top = AWPSHORT(overlap*i*this->m_BaseHeight + 0.5);
423
424 if (overlap*j*this->m_BaseWidth + m_BaseWidth == w)
425 rect.right = rect.left + this->m_BaseWidth - 1;
426 else
427 rect.right = rect.left + this->m_BaseWidth;
428
429 if (overlap * i * this->m_BaseHeight + m_BaseHeight == h)
430 rect.bottom = rect.top + this->m_BaseHeight - 1;
431 else
432 rect.bottom = rect.top + this->m_BaseHeight;
433
434 if (fragments_.empty())
435 {
436 m_minX = rect.left;
437 m_maxX = rect.left + m_BaseWidth;
438 m_minY = rect.top;
439 m_maxY = rect.top + m_BaseHeight;
440 }
441 else
442 {
443 if ((int)m_minX > rect.left)
444 m_minX = rect.left;
445 if (m_maxX < rect.left + m_BaseWidth)
446 m_maxX = rect.left + m_BaseWidth;
447 if ((int)m_minY > rect.top)
448 m_minY = rect.top;
449 if (m_maxY < rect.top + m_BaseHeight)
450 m_maxY = rect.top + m_BaseHeight;
451 }
452
453 fragments_.push_back({ 0, fragments_.size(), rect});
454
455 }
456 }
457
458 m_numX = num_x;
459 m_numY = num_y;
460 return true;
461}
