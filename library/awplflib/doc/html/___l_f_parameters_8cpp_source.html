112ILFParameter* _TLFParameters::Parameter(const char* value)
113{
114 unsigned int index = 0;
115 TLFString strName = value;
116 for (int i = 0; i < m_list.GetCount(); i++)
117 {
118 ILFParameter* current = (ILFParameter*)m_list.Get(i);
119 TLFString strCurrent = current->name();
120 if (strName == strCurrent)
121 return current;
122 }
123 return NULL;
124}
