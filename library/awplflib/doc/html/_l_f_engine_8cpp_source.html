477bool TLFDetectEngine::FindObjects()
478{
479 if (this->m_SourceImage.GetImage() == NULL)
480 return false;
481 if (this->m_SourceImage.GetImage()->dwType != AWP_BYTE)
482 return false;
483
484 m_tmpList.Clear();
485 if (this->m_needTrack)
486 this->m_result.Predict(this);
487 else
488 m_result.Clear();
489 // detecting
490 int cc = 0;
491
492//#ifdef _OMP_
493 //omp_set_num_threads(m_detectors.GetCount());
494 //#pragma omp parallel for reduction(+: cc) num_threads(m_detectors.GetCount())
495//#endif
496 for (int k = 0; k < m_detectors.GetCount(); k++)
497 {
498 ++cc;
499 ILFObjectDetector* d = (ILFObjectDetector*)m_detectors.Get(k);
500 d->Detect();
501 }
502 // receiving information from the detectors
503 for (int k = 0; k < m_detectors.GetCount(); k++)
504 {
505 ILFObjectDetector* d = (ILFObjectDetector*)m_detectors.Get(k);
506 if (d->GetNumItems() > 0)
507 {
508 for (int i = 0; i < d->GetNumItems(); i++)
509 {
510 TLFDetectedItem* item = d->GetItem(i);
511 TLFDetectedItem* di = new TLFDetectedItem(item);
512 di->Resize(m_resizeCoef);
513 m_tmpList.Add(di);
514 }
515 }
516 }
517 if (this->m_needTrack)
518 {
519 m_tmp_result.Clear();
520 this->OverlapsFilter(&m_tmp_result);
521 this->m_result.Update(this, &m_tmp_result);
522 }
523 else
524 {
525 if (!this->m_needCluster)
526 {
527 m_result.Clear();
528 for (int i = 0; i < m_tmpList.GetCount();i++)
529 {
530 TLFDetectedItem* di = (TLFDetectedItem*)m_tmpList.Get(i);
531 TLFDetectedItem* dd = new TLFDetectedItem(di);
532 m_result.AddDetectedItem(dd);
533 }
534 m_tmpList.Clear();
535 }
536 else
537 OverlapsFilter(&this->m_result);
538 }
539 return true;
540}

542void TLFDetectEngine::OverlapsFilter(TLFSemanticImageDescriptor* descriptor)
543{
544 if (descriptor == NULL)
545 return;
546
547 TLFObjectList objectDescr;
548 TLFObjectList objects;
549
550 while (m_tmpList.GetCount() > 0 )
551 {
552 objectDescr.Clear();
553 TLFDetectedItem* de = new TLFDetectedItem(( TLFDetectedItem* )m_tmpList.First());
554 m_tmpList.Delete(0);
555 objectDescr.Add(de);
556 // overlap analysis
557 for (int i = m_tmpList.GetCount() - 1; i >= 0; i--)
558 {
559 TLFDetectedItem* di = (TLFDetectedItem*)m_tmpList.Get(i);
560 if (di->GetBounds().IsCover(de->GetBounds()) || de->GetBounds().IsCover(di->GetBounds()) || de->GetBounds().RectOverlap(di->GetBounds()) > 0.1)// <- this is overlap threshold
561 {
562 //
563 TLFDetectedItem* dd = new TLFDetectedItem(di);
564 objectDescr.Add(dd);
565 m_tmpList.Delete(i);
566 }
567 }
568 // objectDescr analysis
569 awpPoint p; p.X = 0; p.Y = 0;
570 int xsum = 0, ysum = 0, wsum = 0, hsum = 0;
571 AWPSHORT w = 0;
572 AWPSHORT h = 0;
573 double max_raiting = 0;
574
575 TLFDetectedItem* firstObject = (TLFDetectedItem*)objectDescr.Get(0);
576 max_raiting = firstObject->GetRaiting();
577 for (int i = 0; i < objectDescr.GetCount(); i++)
578 {
579 TLFDetectedItem* d = (TLFDetectedItem*)objectDescr.Get(i);
580 xsum += d->GetBounds().Center().X;
581 ysum += d->GetBounds().Center().Y;
582 wsum += d->GetBounds().Width();
583 hsum += d->GetBounds().Height();
584 if (max_raiting < d->GetRaiting())
585 max_raiting = d->GetRaiting();
586 }
587 p.X = xsum/objectDescr.GetCount();
588 p.Y = ysum/objectDescr.GetCount();
589 w = wsum/objectDescr.GetCount();
590 h = hsum/objectDescr.GetCount();
591 // Object attributes
592 awpRect r;
593 double raiting;
594 int bw;
595 int bh;
596 int angle;
597 std::string type;
598 r.left = p.X - w/2;
599 r.right = p.X + w/2;
600 r.top = p.Y - h/2;
601 r.bottom = p.Y + h / 2;
602 angle = firstObject->GetAngle();
603 type = firstObject->GetType();
604 raiting = max_raiting;
605 bw = firstObject->GetBW();
606 bh = firstObject->GetBH();
607 UUID id;
608 // memset(id, 0, sizeof(UUID));
609 LF_NULL_UUID_CREATE(id);
610 // add object to result list
611 TLFDetectedItem* dd = new TLFDetectedItem(r, raiting, type, firstObject->GetAngle(), firstObject->GetRacurs(), bw, bh, firstObject->GetDetectorName(), id);
612 descriptor->AddDetectedItem(dd);
613 delete dd;
614 }

678 }

682 m_relax = true;

744void TLFFGEngine::OverlapsFilter(TLFSemanticImageDescriptor* descriptor)
745{
746 awpImage* img = m_foreground.GetImage();
747#ifdef _DEBUG
748 awpSaveImage("foreground.awp", img);
749#endif
750 awpStrokeObj* obj = NULL;
751 int num = 0;
752 awpGetStrokes(img, &num, &obj, 128, NULL);
753 if (num == 0)
754 return;
755 for (int i =0; i < num; i++)
756 {
757 awpRect rect;
758 awpCalcObjRect(&obj[i], &rect);
759 TLFRect r(rect);
760 r.Scale(m_resizeCoef);
761 rect = r.GetRect();
762 bool found = false;
763
764 if (descriptor != &this->m_result)
765 {
766 for (int i = 0; i < this->GetItemsCount(); i++)
767 {
768 TLFDetectedItem* di = this->GetItem(i);
769
770 if (di->GetBounds().RectOverlap(r)>0 || di->GetBounds().Distance(r) < 5)
771 {
772 found = true;
773 break;
774 }
775 }
776 }
777
778 if (found)
779 continue;
780
781 if (SquareFilter(rect) && ARFilter(rect))
782 {
783 UUID id;
784 LF_NULL_UUID_CREATE(id);
785 TLFDetectedItem* dd = new TLFDetectedItem(rect, 0, "Foreground", 0, 0, 24, 24, "TLFFGBGDetector", id);
786 descriptor->AddDetectedItem(dd);
787 delete dd;
788 }
789 }
790 awpFreeStrokes(num, &obj);
791}
