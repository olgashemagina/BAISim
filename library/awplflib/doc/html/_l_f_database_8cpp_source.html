50static bool _IsImageFile(std::string& strFileName)
51{
52 std::string strExt = LFGetFileExt(strFileName);
53 std::transform(strExt.begin(), strExt.end(), strExt.begin(), ::tolower);
54
55 if (strExt == ".awp")
56 return true;
57 if (strExt == ".jpg")
58 return true;
59 if (strExt == ".jpeg")
60 return true;
61 if (strExt == ".png")
62 return true;
63 if (strExt == ".bmp")
64 return true;
65 return false;
66}

68static bool _HasDescription(std::string& strFileName)
69{
70 if (!_IsImageFile(strFileName))
71 return false;
72
73 if (LFFileExists(LFChangeFileExt(strFileName, ".ieye")))
74 return true;
75 if (LFFileExists(LFChangeFileExt(strFileName, ".xml")))
76 return true;
77 if (LFFileExists(LFChangeFileExt(strFileName, ".face")))
78 return true;
79
80 return false;
81}

143bool TLFDBLabeledImages::LoadDatabase(const char* path)
144{
145 Clear();
146 printf("open database %s\n", path);
147 std::string strDictinary = path;
148 strDictinary += c_separator;
149 strDictinary += c_lpDictFileName;
150 if (!m_dictinary.LoadXML(strDictinary.c_str()))
151 {
152 printf("error: cannot load dictinary \n");
153 return false;
154 }
155
156 bool repair = false;
157 for (int i = 0; i < m_dictinary.GetCount(); i++)
158 {
159 TLFSemanticDictinaryItem* sdi = (TLFSemanticDictinaryItem*)m_dictinary.Get(i);
160 if (sdi->GetNeedRepairItem())
161 repair = true;
162 }
163
164 std::string strPath = path;
165 strPath += c_separator;
166 TLFStrings names;
167 if(!LFGetDirFiles(strPath.c_str(), names))
168 return false;
169 if (names.size() == 0)
170 return false;
171 for (unsigned int i = 0; i < names.size(); i++)
172 {
173 if (_IsImageFile(names[i]))
174 {
175 TLFDBSemanticDescriptor* d = new TLFDBSemanticDescriptor(names[i].c_str());
176 if (repair)
177 {
178 // repair descriptor
179 for (int j = 0; j < d->GetCount(); j++)
180 {
181 TLFDetectedItem* di = d->GetDetectedItem(j);
182 TLFString di_type = di->GetType();
183 for (int k = 0; k < m_dictinary.GetCount(); k++)
184 {
185 TLFSemanticDictinaryItem* sdi = (TLFSemanticDictinaryItem*)m_dictinary.Get(k);
186 TLFString str_sdi_type = sdi->GetItemLabel();
187 if (di_type != "970C3A11-DD73-459B-A741-FB23AA23C883")
188 {
189 // repair here
190 di->SetType(sdi->GetId().c_str());
191
192 }
193 }
194 }
195
196
197 std::string strXmlName = LFChangeFileExt(names[i], ".xml");
198 if (!d->SaveXML(strXmlName.c_str()))
199 {
200 return false;
201 }
202 }
203 this->m_dataFiles.Add(d);
204 }
205 }
206 if (repair)
207 {
208 for (int i = 0; i < m_dictinary.GetCount(); i++)
209 {
210 TLFSemanticDictinaryItem* sdi = (TLFSemanticDictinaryItem*)m_dictinary.Get(i);
211 sdi->SetNeedRepairItem(false);
212 }
213 m_dictinary.SaveXML(strDictinary.c_str());
214 }
215 m_strPath = strPath;
216 return true;
217}

254void TLFDBLabeledImages::GetFarHST(TLFDetectEngine& engine, TLFHistogramm& hst, int stage, bool all, double overlap)
255{
256 ILFObjectDetector* detector = engine.GetDetector();
257 if (detector == NULL)
258 return;
259 if (stage < 0 || stage >= detector->GetStagesCount())
260 return;
261
262 ILFScanner* scanner = detector->GetScanner();
263 if (scanner == NULL)
264 return;
265 TLFObjectList* strongs = detector->GetStrongs();
266 if (strongs == NULL)
267 return;
268 ILFStrong* classifier = (ILFStrong*)strongs->Get(stage);
269 if (classifier == NULL)
270 return;
271
272 hst.Setup(32, 0, classifier->GetThreshold() * 2);
273 for (int i = 0; i < m_dataFiles.GetCount(); i++)
274 {
275 TLFDBSemanticDescriptor* d = (TLFDBSemanticDescriptor*)m_dataFiles.Get(i);
276 std::string strImageName = d->GetImageFile();
277 TLFImage img;
278 if (!img.LoadFromFile(strImageName.c_str()))
279 continue;
280 engine.SetSourceImage(&img, false);
281 TLFImage* img1 = detector->GetImage();
282 if (i % 100 == 0)
283 printf(">");
284 for (int j = 0; j < scanner->GetFragmentsCount(); j++)
285 {
286 double err = 0;
287 double scale;// = 1;
288 awpRect rect = scanner->GetFragmentRect(j);
289 TLFRect lf_rect(rect);
290 if (d->Overlap(lf_rect) < overlap)
291 {
292 scale = (double)(rect.right - rect.left) / (double)scanner->GetBaseWidth();
293 if (!all)
294 {
295 auto res = classifier->Classify(img1, TLFAlignedTransform(scale, scale, rect.left, rect.top));
296 err = res.score;
297 }
298 else
299 {
300 for (int k = 0; k <= stage; k++)
301 {
302 ILFStrong* s = (ILFStrong*)strongs->Get(k);
303
304
305 auto res = s->Classify(img1, TLFAlignedTransform(scale, scale, rect.left, rect.top));
306 err = res.score;
307 if (res.result == 0)
308 {
309 if (k!=stage)
310 err = 0;
311 break;
312 }
313 }
314 }
315 hst.AddElememt(err);
316 }
317 }
318 }

386void TLFDBLabeledImages::GetFarFrrHST(TLFDetectEngine& engine, TLFHistogramm& far_hst, TLFHistogramm& frr_hst, int stage, bool all, double overlap)
387{
388 ILFObjectDetector* detector = engine.GetDetector();

394 ILFScanner* scanner = detector->GetScanner();

398 if (strongs == NULL)
399 return;
400 ILFStrong* classifier = (ILFStrong*)strongs->Get(stage);
401 if (classifier == NULL)
402 return;
403
404 far_hst.Setup(32, 0, classifier->GetThreshold() * 2);
405 frr_hst.Setup(32, 0, classifier->GetThreshold() * 2);
406 for (int i = 0; i < m_dataFiles.GetCount(); i++)
407 {
408 TLFDBSemanticDescriptor* d = (TLFDBSemanticDescriptor*)m_dataFiles.Get(i);
409 std::string strImageName = d->GetImageFile();

414 TLFImage* img1 = detector->GetImage();
415 if (i % 100 == 0)
416 printf(">");
417 for (int j = 0; j < scanner->GetFragmentsCount(); j++)
418 {
419 double err = 0;
420 double scale;// = 1;
421 awpRect rect = scanner->GetFragmentRect(j);
422 TLFRect lf_rect(rect);
423 double overlap_det = d->Overlap(lf_rect);
424 scale = (double)(rect.right - rect.left) / (double)scanner->GetBaseWidth();
425
426 if (!all)
427 {
428 auto res = classifier->Classify(img1, TLFAlignedTransform(scale, scale, rect.left, rect.top));
429 err = res.score;
430 }
431 else
432 {
433 for (int k = 0; k <= stage; k++)
434 {
435 ILFStrong* s = (ILFStrong*)strongs->Get(k);
436
437 auto res = s->Classify(img1, TLFAlignedTransform(scale, scale, rect.left, rect.top));
438 err = res.score;
439 if (res.result == 0)

443 break;
444 }
445 }
446 }

461void TLFDBLabeledImages::GetFarFrr(TLFDetectEngine& engine, double& Far, double& Frr, double& FFar, double& t, double& precision, double& recall, double overlap)
462{
463 Far = 0;
464 Frr = 0;
465 precision = 0;
466 recall = 0;
467 if (GetItemsCount() == 0)
468 return;
469 int num_fragments = 0;
470 int num_founded = 0;
471 int TP = 0;
472 int FP = 0;
473 int FN = 0;
474
475 unsigned long ptime = 0;
476 unsigned long ctime;// = 0;
477 for (int i = 0; i < m_dataFiles.GetCount(); i++)
478 {
479 TLFDBSemanticDescriptor* d = (TLFDBSemanticDescriptor*)m_dataFiles.Get(i);
480 std::string strImageName = d->GetImageFile();
481 printf("processing: %s\n", strImageName.c_str());
482 TLFImage img;
483 img.LoadFromFile(strImageName.c_str());
484 ctime = LFGetTickCount();
485 engine.SetSourceImage(&img, true);
486 ptime += (LFGetTickCount() - ctime);
487 ILFScanner* sc = engine.GetScanner();
488 num_fragments += sc->GetFragmentsCount();
489 TLFSemanticImageDescriptor* d1 = engine.GetSemantic();
490 num_founded += d1->GetCount();
491 double b = d1->Compare(d, overlap);
492 if (b > 0)
493 {
494 Far += b;
495 FP += b;
496 }
497 double a = d->Compare(d1, overlap);
498 if (a > 0)
499 {
500 Frr += a;// d->Compare(d1, overlap);
501 FN += a;
502 }
503
504 TP += (d->GetCount() + d1->GetCount() -a -b)/2;
505 // printf("failed.\n");
506// if (i % 100 == 0)
507// printf(">");
508
509 if (m_progress != NULL)
510 {
511 std::string s = LFGetFileName(strImageName);
512 s += LFGetFileExt(strImageName);
513 m_progress(s.c_str(), 100*i/ m_dataFiles.GetCount());
514 }
515
516 }
517 printf("\n");
518 t = ptime / this->GetImagesCount();
519 FFar = Far / num_fragments;
520 Far /= GetItemsCount();
521 //Far /= num_founded;
522 Frr /= GetItemsCount();
523 precision = double(TP)/(TP + FP);
524 recall = double(TP)/(FN + TP);
525}

532 for (int i = 0; i < m_dataFiles.GetCount(); i++)
