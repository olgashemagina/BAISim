134void TLFList::Clear(bool del)
135{
136 if (m_headNode == NULL)
137 return;
138 TLFListNode* newnode = m_headNode->NextNode();
139 while(newnode != NULL)
140 {
141 TLFListNode* node = newnode;
142 newnode = newnode->NextNode();
143 if (del && node->GetElement() != NULL)
144 delete node->GetElement();
145
146 delete node;
147 }
148 delete m_headNode;
149 m_headNode = NULL;

152void TLFList::PushBack(TLFObject* object)
153{
154 TLFListNode* node = new TLFListNode(object);
155 if (m_headNode == NULL)
156 {
157 m_headNode = node;
158 }
159 else
160 {
161 TLFListNode* tmp_node = m_headNode;
162 while (tmp_node->NextNode() != NULL)
163 tmp_node = tmp_node->NextNode();
164 tmp_node->SetNextNode(node);
165 }
166}
167void TLFList::PopBack()
168{
169 if (m_headNode == NULL)
170 return;
171 /*
172 находим последний узел и удаляем его.
173 у предпоследнего узла устанавливаем
174 следующий узел в NULL
175 */
176 TLFListNode* node = m_headNode; //предпоследний узел
177 TLFListNode* node1 = m_headNode->NextNode(); //последний узел
178 if (node1 == NULL)
179 {
180 /*в списке всего один элемент - удалим его*/
181 delete node;
182 m_headNode = NULL;
183 return;
184 }
185 else
186 {
187 /*в списке больше чем один элемент, надо найти последний
188 и предпоследний*/
189 while (true)
190 {
191 if (node1->NextNode() == NULL)
192 {
193 node->SetNextNode(NULL);
194 delete node1;
195 return;
196 }
197
198 node = node1; // устанавливаем предпоследний элемент
199 node1 = node1->NextNode(); // устанавливаем последний элемент
200 }
201 }
202}
203TLFObject* TLFList::Pop(TLFObject* object)
204{
205 if (m_headNode == NULL)
206 return NULL;
207 if (First() == object)
208 {
209 // удаляем первый элемент.
210 TLFListNode* node = m_headNode->NextNode();
211 delete m_headNode;
212 m_headNode = node;
213 if (node != NULL)
214 return node->GetElement();
215 else
216 return NULL;
217 }
218 else if (Last() == object)
219 {
220 PopBack();
221 return NULL;
222 }
223 else
224 {
225 TLFListNode* node = m_headNode;
226 TLFListNode* node1 = m_headNode->NextNode();
227 if (node1 == NULL)
228 return NULL;
229 while (node1->NextNode() != NULL)
230 {
231 if (node1->GetElement() == object)
232 {
233 node->SetNextNode(node1->NextNode());
234 delete node1;
235 if (node->NextNode() != NULL)
236 return node->NextNode()->GetElement();
237 else
238 return NULL;
239 }
240 node = node1;
241 node1 = node1->NextNode();
242 }
243 }
244 return NULL;
245}

453void QuickSort(TLFObject** pList, int l, int r, TLFListSortCompare Compare)
454{
455 int i, j;
456 void* P;
457 void* T;
458
459 do
460 {
461 i = l;
462 j = r;
463 P = pList[(l+r)>>1];
464 do
465 {
466 while (Compare(pList[i], P) < 0)
467 i++;
468 while (Compare(pList[j], P) >0 )
469 j--;
470 if (i <= j )
471 {
472 T = pList[i];
473 pList[i] = pList[j];
474 pList[j] = (TLFObject*)T;
475 i++;
476 j--;
477 }
478 }while(i < j);
479 if (l < j)
480 QuickSort(pList, l,j, Compare);
481 l = i;
482 }while(i <= r);
483}

824double TLFSemanticImageDescriptor::Compare(TLFSemanticImageDescriptor* Descriptor, double overlap)
825{
826 int count = 0;
827 if (Descriptor == NULL)
828 return this->GetCount();
829 if (Descriptor->GetCount() == 0)
830 return this->GetCount();
831
832 for (int i = 0; i < GetCount(); i++)
833 {
834 TLFDetectedItem* item1 = this->GetDetectedItem(i);
835
836 if (Descriptor->Overlap(item1->GetBounds()) > overlap)
837 count++;
838 }
839 return GetCount() - count;
840}

947void TLFSemanticImageDescriptor::Update(ILFDetectEngine* engine, TLFSemanticImageDescriptor* sid)
948{
949 if (engine == NULL || sid == NULL)
950 return;
951 // сюда приходит sid без предикторов. это временное состояние машины
952 // здесь к элементу описания добавляется предиктор и он записывается в этот sid
953 double d_factor = engine->GetResize() ? 1. / engine->GetResizeCoef() : 1;
954 double u_factor = engine->GetResize() ? engine->GetResizeCoef() : 1;
955 for (int i = 0; i < this->GetCount(); i++)
956 {
957 TLFDetectedItem* di1 = this->GetDetectedItem(i);
958
959 if (di1)
960 {
961 di1->Resize(d_factor);
962
963 bool found = false;
964 for (int j = sid->GetCount()-1; j >=0; j--)
965 {
966 TLFDetectedItem* di2 = sid->GetDetectedItem(j);
967 if (di2)
968 {
969 di2->Resize(d_factor);
970
971 if (di1->GetBounds().RectOverlap(di2->GetBounds()) > 0)
972 {
973 found = true;
974 di1->Update(engine, di2->GetBounds());
975 sid->Delete(j);
976 }
977 else
978 di2->Resize(u_factor);
979 }
980 }
981
982 if (!found)
983 // Empty rect
984 di1->Update(engine, TLFRect());
985
986 di1->Resize(u_factor);
987 }
988 }
989
990 // clear intersections
991 for (int i = 0; i < GetCount(); i++)
992 {
993 TLFDetectedItem* d0 = this->GetDetectedItem(i);
994 double s0 = d0->GetBounds().Square ();
995 for (int j = i+1; j < GetCount(); j++ )
996 {
997 TLFDetectedItem* d1 = this->GetDetectedItem(j);
998
999 if (d0->GetBounds().RectOverlap(d1->GetBounds()) > 0.1)
1000 {
1001 double s1 = d1->GetBounds().Square();
1002 if (s0 < s1)
1003 d0->SetState(3);
1004 else
1005 d1->SetState(3);
1006 }
1007 }
1008
1009 }
1010
1011 // remove all elements whose Health ==0
1012 for (int i = this->GetCount() - 1; i >= 0; i--)
1013 {
1014 TLFDetectedItem* item = this->GetDetectedItem(i);
1015 if (item->GetHealth() == 0 && item->GetState() == 3)
1016 Delete(i);
1017 else if (item->GetState() == 3)
1018 {
1019 item->Delete();
1020 }
1021 }
1022 // adding new elements
1023 for (int i = 0; i < sid->GetCount(); i++)
1024 {
1025 TLFDetectedItem* di = sid->GetDetectedItem(i);
1026 awpRect pRect = di->GetBounds().GetRect();
1027 ILFPredictor* predictor = LFCreatePredictor(engine->GetPredictorName(), engine);
1028 UUID id;
1029 LF_UUID_CREATE(id);
1030 TLFDetectedItem* item = new TLFDetectedItem(pRect, di->GetRaiting(), di->GetType(), di->GetAngle(), di->GetRacurs(), di->GetBW(), di->GetBH(), di->GetDetectorName(), id, predictor);
1031 item->Resize(d_factor);
1032
1033 item->Update(engine, item->GetBounds());
1034 item->Resize(u_factor);
1035 Add(item);
1036 }
1037}
