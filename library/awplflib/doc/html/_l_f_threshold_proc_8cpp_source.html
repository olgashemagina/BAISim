63 {
64 sx = m_ImageRoi.Left();
65 sy = m_ImageRoi.Top();
66 ex = sx + m_ImageRoi.Width();
67 ey = sy + m_ImageRoi.Height();

81 if (i >= sy && i <= ey && j >= sx && j <= ex)

83 if (invert != 0)

85 if (pix_dst[i*src->sSizeX + j] > thr)
