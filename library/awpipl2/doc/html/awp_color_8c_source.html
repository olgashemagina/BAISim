61AWPRESULT awpWebtoRGB(AWPINT web, AWPBYTE* r, AWPBYTE* g, AWPBYTE* b)
62{
63 AWPINT value;
64 if (r == NULL || g == NULL || b == NULL)
65 return AWP_BADARG;
66 value = web & 0xFF0000;
67 *r = value >> 16;
68 value = web & 0x00FF00;
69 *g = value >> 8;
70 *b = web & 0x0000FF;
71 return AWP_OK;
72}

74// Convert HSV image to RGB image
75AWPRESULT awpHSVtoRGBImage(const awpImage* Image, awpImage** ppRGB)
76{
77 AWPINT i = 0;
78 AWPINT c = 0;
79 awpColor* pc = NULL;
80 awpColor* pcRGB = NULL;
81 AWPRESULT res = AWP_OK;
82 AWPDOUBLE h, s, v;
83 AWPBYTE r, g, b;
84
85 res = awpCheckImage(Image);
86 if (res != AWP_OK)
87 return res;
88
89 if (Image->dwType != AWP_BYTE && Image->bChannels != 3)
90 return AWP_BADARG;
91
92 res = awpCreateImage(ppRGB, Image->sSizeX,
93 Image->sSizeY, 3, AWP_BYTE);
94 if (res != AWP_OK)
95 return res;
96
97 pc = (awpColor*)Image->pPixels;
98 pcRGB = (awpColor*)(*ppRGB)->pPixels;
99 c = Image->sSizeX*Image->sSizeY;
100 for (i = 0; i < c; i++)
101 {
102 h = pc[i].bRed / 255.0;
103 s = pc[i].bGreen / 255.0;
104 v = pc[i].bBlue / 255.0;
105 _awp_hsv_to_rgb(h, s, v, &r, &g, &b);
106 pcRGB[i].bRed = b;
107 pcRGB[i].bGreen = g;
108 pcRGB[i].bBlue = r;
109 }
110
111 return AWP_OK;
112}

203// Converts RGB image to HSL image
204AWPRESULT awpRGBtoHSLImage(const awpImage* Image, awpImage** ppHSL)
205{
206
207 const AWPINT planes = 3;
208 AWPINT x, y, x1;
209 AWPINT b, g, r;
210 AWPINT h, s, v;
211 AWPINT vmin, diff;
212 AWPINT vr, vg;
213 AWPINT min_ch_val[6];
214 AWPINT max_ch_val[6];
215
216 AWPBYTE* src = (AWPBYTE*)Image->pPixels;
217 AWPINT src_pitch = Image->sSizeX;
218 AWPBYTE* dst = NULL;
219 awpImage* im = NULL;
220 AWPINT dst_pitch = 0;
221
222
223 AWPINT i = 0;
224 AWPRESULT res = AWP_OK;
225
226 res = awpCheckImage(Image);
227 if (res != AWP_OK)
228 return res;
229
230 if (Image->dwType != AWP_BYTE && Image->bChannels != 3)
231 return res;
232
233 res = awpCreateImage(&im, Image->sSizeX,
234 Image->sSizeY, 3, AWP_BYTE);
235 if (res != AWP_OK)
236 return res;
237
238 dst = (AWPBYTE*)im->pPixels;
239 dst_pitch = im->sSizeX;
240
241 for (i = 0; i < 6; i++)
242 {
243 min_ch_val[i] = 0;
244 max_ch_val[i] = 255;
245 }
246
247
248 for (y = 0; y < Image->sSizeY; y++, src += src_pitch, dst += dst_pitch)
249 for (x = 0, x1 = 0; x < Image->sSizeX * planes; x += planes, x1++)
250 {
251
252 b = src[x];
253 g = src[x + 1];
254 r = src[x + 2];
255
256 // convert rgb->hsv
257 {
258
259 v = mmax(r, g);
260 v = mmax(v, b);
261 vmin = mmin(r, g);
262 vmin = mmin(vmin, b);
263
264 diff = v - vmin;
265 vr = v == r ? -1 : 0;
266 vg = v == g ? -1 : 0;
267
268 s = diff*div_table[v] >> SHIFT;
269 h = (vr&(g - b)) + (~vr&((vg&(b - r + 2 * diff)) + ((~vg)&(r - g + 4 * diff))));
270 h = ((h*div_table[diff] * 15 + (1 << (SHIFT + 6))) >> (7 + SHIFT)) + (h < 0 ? 30 * 6 : 0);
271 }
272
273 if (s >= min_ch_val[1] && v >= min_ch_val[2] && v <= max_ch_val[2])
274 dst[x1] = (AWPBYTE)h;
275 else
276 dst[x1] = 0;
277 }
278 /*
279 pc = (awpColor*)Image->pPixels;
280 pcHSL = (awpColor*)(*ppHSL)->pPixels;
281 c = Image->sSizeX*Image->sSizeY;
282 for (i = 0; i < c; i++)
283 {
284 _awpRGBtoHSL(pc[i].bBlue ,pc[i].bGreen ,pc[i].bRed ,&h, &s, &l );
285 h *= 255.0; s *= 255.0; l *= 255.0;
286 pcHSL[i].bRed = (AWPBYTE)h;
287 pcHSL[i].bGreen = (AWPBYTE)s;
288 pcHSL[i].bBlue = (AWPBYTE)l;
289 }
290 */
291 *ppHSL = im;
292 return AWP_OK;
293}
