18 type* rr = (type*)dst->pPixels;\

60AWPRESULT awpMakeBinary(const awpImage* pImage, awpImage** ppBinImage, const AWPDOUBLE sThreshold, const AWPLONG lOptions,
61 AWPDOUBLE MinValue, AWPDOUBLE MaxValue, awpRect* pRect)
62{
63 AWPRESULT res;
64 awpImage* dst;
65 awpRect r;
66 /* check the input parameters for validity
67 */
68 if (pImage == NULL || ppBinImage == NULL)
69 _ERROR_EXIT_RES_(AWP_BADARG)
70
71 if(lOptions != AWP_BINARY && lOptions != AWP_INV_BINARY)
72 _ERROR_EXIT_RES_(AWP_BADARG)
73
74 if (pImage->bChannels != 1)
75 _ERROR_EXIT_RES_(AWP_BADARG)
76 if (pRect != NULL)
77 {
78 _CHECK_RESULT_(res = awpRectInImage(pImage, pRect))
79 }
80
81 /* Preparation of the resulting image
82 */
83 dst = *ppBinImage;
84 if (dst == NULL)
85 {
86 /* there is not resulting image. create it
87 */
88 _CHECK_RESULT_(res = awpCreateImage(&dst, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, pImage->dwType) )
89 }
90 else
91 {
92 /* check on the validity of the resulting image
93 */
94 _CHECK_SAME_SIZES(pImage, dst)
95 _CHECK_SAME_TYPE(pImage, dst)
96 _CHECK_NUM_CHANNELS(pImage, dst)
97 }
98 /* establish the spatial range of image processing
99 */
100 if (pRect == NULL)
101 {
102 r.left = 0;
103 r.top = 0;
104 r.right = pImage->sSizeX;
105 r.bottom = pImage->sSizeY;
106 }
107 else
108 {
109 r = *pRect;
110 }
111 /* carry out the transformation
112 */
113 switch (pImage->dwType)
114 {
115 case AWP_BYTE:
116 res = _awpMakeBinaryBYTE(pImage, dst, sThreshold, lOptions, r, MaxValue, MinValue);
117 break;
118 case AWP_SHORT:
119 res = _awpMakeBinarySHORT(pImage, dst, sThreshold, lOptions, r, MaxValue, MinValue);
120 break;
121 case AWP_FLOAT:
122 res = _awpMakeBinaryFLOAT(pImage, dst, sThreshold, lOptions, r, MaxValue, MinValue);
123 break;
124 case AWP_DOUBLE:
125 res = _awpMakeBinaryDOUBLE(pImage, dst, sThreshold, lOptions, r, MaxValue, MinValue);
126 break;
127 }
128
129
130 *ppBinImage = dst;
131CLEANUP:
132 return res;
133}

139 AWPINT iIndex;

161 _CHECK_RESULT_(( res = awpGetImagePixelType(pImage, &iImageType)))
162
163 switch(iImageType){
164 case AWP_BYTE:
165 if( sThreshold < 0 || sThreshold > 255)
166 {
167 res = AWP_BADARG;
168 _ERR_EXIT_
169 }
170 break;
171 case AWP_SHORT:
172 if( sThreshold < AWP_SHRT_MIN || sThreshold > AWP_SHRT_MAX )

176 }

182 _ERR_EXIT_
183 }
184 break;
185 }
186
187 _CHECK_RESULT_(( res = awpGetImageSize(pImage, &iImageSize)))
188 if (ppBinImage != NULL){
189 _CHECK_RESULT_(( res = awpCreateImage(&pTempImage, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, iImageType)))
190 }else{
191 pTempImage = (awpImage*)pImage;
192 }
193
194 switch(iImageType)
195 {
196 case AWP_BYTE: pSrcBuf = (AWPBYTE*) pImage->pPixels;
197 pTempBuf = (AWPBYTE*) pTempImage->pPixels; break;
198 case AWP_SHORT: pSrcBuf = (AWPSHORT*) pImage->pPixels;
199 pTempBuf = (AWPSHORT*) pTempImage->pPixels; break;
200 case AWP_FLOAT: pSrcBuf = (AWPFLOAT*) pImage->pPixels;
201 pTempBuf = (AWPFLOAT*) pTempImage->pPixels; break;
202 case AWP_DOUBLE: pSrcBuf = (AWPDOUBLE*) pImage->pPixels;
203 pTempBuf = (AWPDOUBLE*) pTempImage->pPixels; break;
204
205 default:

207 res = AWP_BADARG;
208 _ERR_EXIT_
209 }
210 }
211
212 if(lOptions == AWP_BINARY)
213 for(iIndex = 0; iIndex < pImage->sSizeX*pImage->sSizeY*pImage->bChannels; ++iIndex)
214 {
215 switch(iImageType)
216 {
217 case AWP_BYTE: ((AWPBYTE*) pTempBuf)[iIndex] = (((AWPBYTE*) pSrcBuf)[iIndex] > sThreshold)?255:0; break;
218 case AWP_SHORT: ((AWPSHORT*) pTempBuf)[iIndex] = (((AWPSHORT*) pSrcBuf)[iIndex] > sThreshold)?255:0; break;
219 case AWP_FLOAT: ((AWPFLOAT*) pTempBuf)[iIndex] = (((AWPFLOAT*) pSrcBuf)[iIndex] > sThreshold)?255.f:0; break;
220 case AWP_DOUBLE: ((AWPDOUBLE*)pTempBuf)[iIndex] = (((AWPDOUBLE*)pSrcBuf)[iIndex] > sThreshold)?255.:0; break;
221
222 default:
223 {
224 res = AWP_BADARG;
225 _ERR_EXIT_
226 }
227 }
228 }
229 else
230 for(iIndex = 0; iIndex < pImage->sSizeX*pImage->sSizeY*pImage->bChannels; ++iIndex)
231 {
232 switch(iImageType)
233 {
234 case AWP_BYTE: ((AWPBYTE*) pTempBuf)[iIndex] = (((AWPBYTE*) pSrcBuf)[iIndex] > sThreshold)?0:255; break;
235 case AWP_SHORT: ((AWPSHORT*) pTempBuf)[iIndex] = (((AWPSHORT*) pSrcBuf)[iIndex] > sThreshold)?0:255; break;
236 case AWP_FLOAT: ((AWPFLOAT*) pTempBuf)[iIndex] = (((AWPFLOAT*) pSrcBuf)[iIndex] > sThreshold)?0:255.f; break;
237 case AWP_DOUBLE: ((AWPDOUBLE*)pTempBuf)[iIndex] = (((AWPDOUBLE*)pSrcBuf)[iIndex] > sThreshold)?0:255.; break;
238
239 default:
240 {
241 res = AWP_BADARG;
242 _ERR_EXIT_
243 }
244 }
245 }
246 if (ppBinImage!= NULL){
247 _CHECK_RESULT_((res = awpCopyImage(pTempImage, ppBinImage)))
248 }
249
250
251CLEANUP:

253 _SAFE_RELEASE_(pTempImage)
254 }
