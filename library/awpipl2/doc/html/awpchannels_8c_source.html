11AWPRESULT awpGetChannel(const awpImage *pImage, awpImage **ppChannel, const AWPBYTE bOptions)
12{
13 AWPINT iImageType = 0;
14 AWPINT iBytesCount = 0;
15 AWPINT iIndex = 0;
16
17 AWPBYTE *pTempBuf = NULL;
18 const AWPBYTE *pSrcBuf = NULL;
19
20 awpImage *pTempImage = NULL;
21
22 AWPRESULT res;
23 res = AWP_OK;
24
25 _CHECK_RESULT_((res = awpCheckImage(pImage)))
26
27 if(bOptions >= pImage->bChannels)
28 {
29 res = AWP_BADARG;
30 _ERR_EXIT_
31 }
32
33 _CHECK_RESULT_((res = awpGetImagePixelType(pImage, &iImageType)))
34
35 switch(iImageType) {
36 case AWP_BYTE : iBytesCount = 1; break;
37 case AWP_SHORT: iBytesCount = sizeof(AWPSHORT)/sizeof(AWPBYTE); break;
38 case AWP_FLOAT: iBytesCount = sizeof(AWPFLOAT)/sizeof(AWPBYTE); break;
39 case AWP_DOUBLE: iBytesCount = sizeof(AWPDOUBLE)/sizeof(AWPBYTE); break;
40
41 default:
42 {
43 res = AWP_BADARG;
44 _ERR_EXIT_
45 }
46 }
47
48 _CHECK_RESULT_((res = awpCreateImage(&pTempImage, pImage->sSizeX, pImage->sSizeY, 1, iImageType)))
49
50 pTempBuf = (AWPBYTE*)pTempImage->pPixels;
51 pSrcBuf = (AWPBYTE*)pImage->pPixels + bOptions*iBytesCount*sizeof(AWPBYTE) ;
52
53 for(iIndex = 0; iIndex < pImage->sSizeX*pImage->sSizeY-1; ++iIndex )
54 {
55 memcpy(pTempBuf, pSrcBuf, iBytesCount*sizeof(AWPBYTE));
56
57 pTempBuf += iBytesCount*sizeof(AWPBYTE);
58 pSrcBuf += pImage->bChannels*iBytesCount*sizeof(AWPBYTE);
59
60 }
61
62 _CHECK_RESULT_((res = awpCopyImage(pTempImage, ppChannel)))
63
64CLEANUP:
65 _SAFE_RELEASE_(pTempImage)
66 return res;
67}
