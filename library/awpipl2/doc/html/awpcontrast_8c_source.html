14AWPRESULT awpAutoLevels(awpImage* pImage)
15{
16 AWPRESULT res = AWP_OK;
17 AWPINT total = pImage->sSizeX*pImage->sSizeY;
18 AWPDOUBLE* min_values = NULL;
19 AWPDOUBLE* max_values = NULL;
20 AWPBYTE* pix = NULL;
21 AWPWORD x = 0;
22 AWPWORD y = 0;
23 AWPBYTE c = 0;
24 AWPBYTE* v = 0;
25 if (res = awpCheckImage(pImage) != AWP_OK)
26 _ERR_EXIT_
27
28 if (pImage->dwType != AWP_BYTE)
29 _ERROR_EXIT_RES_(AWP_NOTSUPPORT)
30
31 if (res = awpMinMax(pImage, &min_values, &max_values) != AWP_OK)
32 _ERR_EXIT_
33
34 pix = _AWP_BPIX_(pImage, AWPBYTE);
35 for (y = 0; y < pImage->sSizeY; y++)
36 {
37 for (x = 0; x < pImage->sSizeX; x++)
38 {
39 for (c = 0; c < pImage->bChannels; c++)
40 {
41 v = &pix[y*pImage->sSizeX*pImage->bChannels + x*pImage->bChannels + c];
42 if (max_values[c] - min_values[c] == 0)
43 *v = (AWPBYTE)max_values[c];
44 else
45 *v = (AWPBYTE)(255 * (*v - min_values[c]) / (max_values[c] - min_values[c]));
46 }
47 }
48 }
49CLEANUP:
50 _SAFE_FREE_(min_values);
51 _SAFE_FREE_(max_values);
52 return res;
53}

55AWPRESULT awpBilinearBlur( awpImage* pImage, AWPDOUBLE dfStrength )
56{
57 AWPRESULT res = AWP_OK;
58
59 AWPDOUBLE dfStrengthA = 1.0 / ( dfStrength + 1.0 );
60 AWPDOUBLE dfStrengthA1 = dfStrength * dfStrengthA;
61 AWPINT iX , iY;
62 //AWPINT iOffset00 , iOffset01 , iOffset10 , iOffset11;
63
64 AWPDOUBLE* pdfPixels = (AWPDOUBLE*)pImage->pPixels;
65
66 AWPDOUBLE* pdfPix00 = NULL;
67 AWPDOUBLE* pdfPix01 = NULL;
68 AWPDOUBLE* pdfPix10 = NULL;
69 AWPDOUBLE* pdfPix11 = NULL;
70
71
72 _CHECK_RESULT_( res = awpCheckImage(pImage) );
73 _CHECK_RESULT_( res = (pImage->dwType == AWP_DOUBLE) ? AWP_OK : AWP_BADARG );
74
75 pdfPix10 = pdfPixels;
76 pdfPix11 = pdfPixels + 1;
77 for ( iX = 1 ; iX<pImage->sSizeX ; ++iX, ++pdfPix11, ++pdfPix10 )//, --pdfPix00, --pdfPix01)
78 {
79 pdfPix11[0] = ( dfStrengthA * pdfPix11[0] + dfStrengthA1 * pdfPix10[0] );
80 }
81
82 --pdfPix11; --pdfPix10;
83 for ( iX = 1 ; iX<pImage->sSizeX ; ++iX, --pdfPix11, --pdfPix10 )//, --pdfPix00, --pdfPix01)
84 {
85 pdfPix10[0] = dfStrengthA * pdfPix10[0] + dfStrengthA1 * pdfPix11[0];
86 }
87
88 pdfPix01 = pdfPixels;
89 pdfPix11 = pdfPixels + pImage->sSizeX;
90 for ( iY = 1 ; iY<pImage->sSizeY ; ++iY, pdfPix01 += pImage->sSizeX, pdfPix11 += pImage->sSizeX )
91 {
92 pdfPix11[0] = ( dfStrengthA * pdfPix11[0] + dfStrengthA1 * pdfPix01[0] );
93 }
94
95 pdfPix01 -= pImage->sSizeX; pdfPix11 -= pImage->sSizeX;
96
97 for ( iY = 1 ; iY<pImage->sSizeY ; ++iY, pdfPix01 -= pImage->sSizeX, pdfPix11 -= pImage->sSizeX )
98 {
99 pdfPix01[0] = ( dfStrengthA * pdfPix01[0] + dfStrengthA1 * pdfPix11[0] );
100 }
101
102
103
104 pdfPix00 = pdfPixels;
105 pdfPix01 = pdfPixels + 1;
106 pdfPix10 = pdfPixels + pImage->sSizeX;
107 pdfPix11 = pdfPixels + pImage->sSizeX + 1;
108
109 for ( iY = 1 ; iY<pImage->sSizeY ; ++iY, ++pdfPix00, ++pdfPix01, ++pdfPix10, ++pdfPix11 )
110 for ( iX = 1 ; iX<pImage->sSizeX ; ++iX, ++pdfPix00, ++pdfPix01, ++pdfPix10, ++pdfPix11 )
111 {
112 pdfPix11[0] = dfStrengthA * ( dfStrengthA * pdfPix11[0] + dfStrength * ( pdfPix01[0] + pdfPix10[0] - dfStrengthA1 * pdfPix00[0] ) );
113 }
114
115
116
117
118 pdfPix00 = pdfPixels + pImage->sSizeX * pImage->sSizeY - 1;
119 pdfPix01 = pdfPix00 - 1;
120 pdfPix10 = pdfPix00 - pImage->sSizeX;
121 pdfPix11 = pdfPix10 - 1;
122
123 for ( iY = pImage->sSizeY-1; iY > 0; --iY, --pdfPix00, --pdfPix01, --pdfPix10, --pdfPix11 )
124 for ( iX = pImage->sSizeX-1 ; iX > 0; --iX, --pdfPix00, --pdfPix01, --pdfPix10, --pdfPix11 )
125 {
126 pdfPix11[0] = dfStrengthA * ( dfStrengthA * pdfPix11[0] + dfStrength * ( pdfPix01[0] + pdfPix10[0] - dfStrengthA1 * pdfPix00[0] ) );
127 }
128
129
130CLEANUP:
131 return res;
132}
