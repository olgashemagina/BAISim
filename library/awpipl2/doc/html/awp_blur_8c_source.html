11/*
12 awpCreateGaussKernel - creates Gauss kernel as with sigma. sigma < 50
13 high_precision = true - creates AWPDOUBLE kernel
14 else creates AWPFLOAT kernel.
15*/
16AWPRESULT awpCreateGaussKernel(awpImage** ppKernel, AWPDOUBLE sigma)
17{
18 AWPRESULT res = AWP_OK;
19 AWPSHORT x,y,hw,xx,yy;
20 AWPWORD w = 0;
21 void* p = NULL;
22 AWPDOUBLE* dp = NULL;
23 AWPDOUBLE falfa = 1;
24 AWPDOUBLE fbeta = 1;
25 if (sigma <= 0)
26 _ERROR_EXIT_RES_(AWP_BADARG)
27 w = (AWPWORD)(4*sigma);
28 if (w < 3) w = 3;
29 w = w % 2 == 0?w+1:w;
30 hw = w / 2;
31
32 _CHECK_RESULT_(res = awpCreateImage(ppKernel, w,w,1,AWP_DOUBLE))
33 p = (*ppKernel)->pPixels;
34 dp = (AWPDOUBLE*)p;
35 falfa = 1. /(2.*3.14159*sigma*sigma);
36 fbeta = -1./(2.*sigma*sigma);
37 for (y=0; y < w; y++)
38 {
39 yy = y - hw;
40 for (x = 0; x < w; x++)
41 {
42 xx = x - hw;
43 dp[y*w + x] = falfa*exp(fbeta*(xx*xx +yy*yy));
44 }
45 }
46
47CLEANUP:
48 return res;
49}
