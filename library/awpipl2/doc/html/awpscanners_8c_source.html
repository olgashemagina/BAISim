624float _awpItemsOverlap(awpDetectItem r1, awpDetectItem r2)
625{
626
627 AWPFLOAT colInt, rowInt, intersection, area1,area2;
628 AWPINT width, height,_width,_height;
629
630 width = r1.rect.right - r1.rect.left;
631 height = r1.rect.bottom - r1.rect.top;
632 _width = r2.rect.right - r2.rect.left;
633 _height = r2.rect.bottom - r2.rect.top;
634 if (r2.rect.left > r1.rect.left + width)
635 return 0;
636 if (r2.rect.top > r1.rect.top + height)
637 return 0;
638 if (r2.rect.left + _width < r1.rect.left)
639 return 0;
640 if (r1.rect.top + _height < r1.rect.top)
641 return 0;
642
643 colInt = AWP_MIN(r2.rect.left + _width, r1.rect.left + width) - AWP_MAX(r2.rect.left, r1.rect.left);
644 rowInt = AWP_MIN(r2.rect.top + _height, r1.rect.top + height) - AWP_MAX(r2.rect.top, r1.rect.top);
645
646 intersection = colInt*rowInt;
647 area1 = _width*_height;
648 area2 = width*height;
649
650 return intersection / (area1+area2-intersection);
651}

653static void _awpClusterItems(int numSrc, awpDetectItem* pSrc, int* numDst, awpDetectItem** ppDst)
654{
655 //
656 AWPINT i,j, NumClusters, idx, eCount, cidx;
657 AWPBOOL found;
658 AWPFLOAT overlap;
659 awpDetectItem r;
660
661 cawpDetectItem* crects = (cawpDetectItem*)malloc(numSrc*sizeof(cawpDetectItem));
662 for (i = 0; i < numSrc; i++)
663 {
664 crects[i].r = pSrc[i];
665 crects[i].c = -1;
666 }
667 //
668 NumClusters = 0;
669 found = FALSE;
670 idx = 0;
671 do
672 {
673 found = FALSE;
674 for (i = 0; i < numSrc; i++)
675 {
676 if (crects[i].c == -1)
677 {
678 found = TRUE;
679 NumClusters++;
680 crects[i].c = NumClusters;
681 idx = i;
682 break;
683 }
684 }
685 if (found)
686 {
687 for (i = 0; i < numSrc; i++)
688 {
689 if (i == idx)
690 continue;
691 if (crects[i].c != -1)
692 continue;
693 overlap = _awpItemsOverlap(crects[i].r , crects[idx].r);
694 if (overlap > 0.4)
695 {
696 crects[i].c = NumClusters;
697 }
698 }
699 }
700 }while(found);
701
702 //
703 eCount = 0;
704 cidx = 0;
705 (*numDst) = NumClusters;
706 *ppDst = (awpDetectItem*)malloc(NumClusters*sizeof(awpDetectItem));
707 for (i = 0; i < NumClusters; i++)
708 {
709 cidx = i+1;
710 r.rect.left = 0;
711 r.rect.right = 0;
712 r.rect.bottom = 0;
713 r.rect.top = 0;
714 eCount = 0;
715 for (j = 0; j < numSrc; j++)
716 {
717 if (crects[j].c == cidx)
718 {
719 r.rect.left += crects[j].r.rect.left;
720 r.rect.bottom += crects[j].r.rect.bottom;
721 r.rect.right += crects[j].r.rect.right;
722 r.rect.top += crects[j].r.rect.top;
723 eCount++;
724 }
725 }
726 r.rect.left /= eCount;
727 r.rect.right /= eCount;
728 r.rect.bottom /= eCount;
729 r.rect.top /= eCount;
730 (*ppDst)[i] = r;
731 }
732
733
734 //
735 free(crects);
736}
