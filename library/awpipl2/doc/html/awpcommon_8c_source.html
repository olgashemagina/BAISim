22AWPRESULT awpCheckImage(const awpImage* pImage)
23{
24 /*check pointer*/
25 if (pImage == NULL)
26 return AWP_BADMEMORY;
27 /*check magic number*/
28 if (pImage->nMagic != AWP_MAGIC2)
29 return AWP_NOTAWPIMAGE;
30 /*check channles*/
31 if (pImage->bChannels == 0)
32 return AWP_BADIMAGE_FORMAT;
33 /*check sizes*/
34 if (pImage->sSizeX == 0 ||pImage->sSizeY == 0)
35 return AWP_BADSIZES;
36 /*check format*/
37 if (pImage->dwType != AWP_BYTE &&
38 pImage->dwType != AWP_SHORT &&
39 pImage->dwType != AWP_FLOAT &&
40 pImage->dwType != AWP_DOUBLE )
41 return AWP_BADIMAGE_FORMAT;
42 /*check pixels*/
43 if (pImage->pPixels == NULL)
44 return AWP_BADMEMORY;
45
46 return AWP_OK;
47
48}

66AWPRESULT awpCreateImage(awpImage** ppImage,AWPWORD width,AWPWORD height,AWPBYTE channels, AWPDWORD dwType)
67{
68 awpImage* pImage;
69 AWPINT size;
70 if( ppImage == NULL )
71 return AWP_BADARG;
72 if (channels == 0 || width == 0 || height == 0)
73 return AWP_BADARG;
74 *ppImage = NULL;
75 /*todo: this code is subject to change
76 dwType may be not only just a set of channels....*/
77 switch(dwType)
78 {
79 case AWP_BYTE: size = channels*sizeof(AWPBYTE)*width*height;break;
80 case AWP_SHORT:size = channels*sizeof(AWPSHORT)*width*height;break;
81 case AWP_FLOAT:size = channels*sizeof(AWPFLOAT)*width*height;break;
82 case AWP_DOUBLE:size = channels*sizeof(AWPDOUBLE)*width*height;break;
83 default:
84 return AWP_BADARG;
85 }
86 /*alloc memory for header*/
87 pImage = NULL;
88 pImage = (awpImage*)malloc(sizeof(awpImage));
89 if (pImage == NULL)
90 return AWP_BADMEMORY;
91
92 pImage->nMagic = AWP_MAGIC2;
93 pImage->bChannels = channels;
94 pImage->sSizeX = width;
95 pImage->sSizeY = height;
96 pImage->dwType = dwType;
97 pImage->pPixels = NULL;
98
99 pImage->pPixels = malloc(size);
100 if (pImage->pPixels == NULL)
101 {
102 free(pImage);
103 return AWP_BADMEMORY;
104 }
105 /*fill memory*/
106 memset((AWPBYTE*)pImage->pPixels,0,size);
107 *ppImage = pImage;
108 return AWP_OK;
109}

111AWPRESULT awpCreateGrayImage(awpImage** ppImage, AWPWORD width, AWPWORD height, AWPBYTE channels, AWPBYTE* data)
112{
113 AWPRESULT res = AWP_OK;
114 awpImage* pImage = NULL;
115 AWPINT size = 0;
116 AWPINT i = 0;
117 AWPINT k = 0;
118 AWPINT c = 0;
119 AWPINT sum = 0;
120 AWPINT bpp = 0;
121 AWPBYTE* pix = NULL;
122 if (ppImage == NULL)
123 return AWP_BADARG;
124 if (channels == 0 || width == 0 || height == 0 || data == NULL)
125 return AWP_BADARG;
126 *ppImage = NULL;
127 res = awpCreateImage(ppImage, width, height, 1, AWP_BYTE);
128 if (res != AWP_OK)
129 return res;
130 pImage = *ppImage;
131 pix = (AWPBYTE*)pImage->pPixels;
132
133
134 bpp = channels > 3 ? 3 : channels;
135 for (i = 0; i < width*height*channels; i+= channels)
136 {
137 sum = 0;
138 for (k = 0; k < bpp; k++)
139 sum += data[i + k];
140 sum /= bpp;
141 pix[c] = (AWPBYTE)sum;
142 c++;
143 }
144
145 return res;
146}

237AWPRESULT awpGetImageSize(const awpImage* pImage, AWPINT* pSize)
238{
239 AWPRESULT res;

241 AWPINT itype;
242 /*init variables */
243 res = AWP_OK;
244 isize = 0;
245
246 _CHECK_RESULT_((res = awpCheckImage(pImage)))
247
248 isize = pImage->sSizeX*pImage->sSizeY*pImage->bChannels;
249
250 _CHECK_RESULT_((res = awpGetImagePixelType(pImage, &itype)))
251
252 switch(itype)
253 {
254 case AWP_BYTE: isize *= sizeof(AWPBYTE); break;
255 case AWP_SHORT:isize *= sizeof(AWPSHORT); break;
256 case AWP_FLOAT:isize *= sizeof(AWPFLOAT); break;
257 case AWP_DOUBLE:isize *= sizeof(AWPDOUBLE); break;
258 default:
259 {
260 res = AWP_BADARG;
261 _ERR_EXIT_
262 }
263 }
264
265 *pSize = isize;
266 res = AWP_OK;
267CLEANUP:
268 return res;

332AWPRESULT awpMinMax(const awpImage* pImage, AWPDOUBLE** min, AWPDOUBLE** max)
333{
334 AWPWORD i,j;
335 AWPBYTE c;
336 AWPINT k;
337 AWPRESULT res;
338 res=AWP_OK;
339
340 _CHECK_RESULT_(( res = awpCheckImage(pImage)))
341
342 if (min == NULL || max == NULL)
343 {
344 res = AWP_BADARG;
345 _ERR_EXIT_
346 }
347
348 /*alloc memory for arrays*/
349 *min = NULL;
350 *max = NULL;
351
352 *min = (AWPDOUBLE*)malloc(pImage->bChannels*sizeof(AWPDOUBLE));
353 if (*min == NULL)
354 {
355 res = AWP_BADMEMORY;
356 _ERR_EXIT_
357 }
358 *max = (AWPDOUBLE*)malloc(pImage->bChannels*sizeof(AWPDOUBLE));
359 if (*max == NULL)
360 {
361 res = AWP_BADMEMORY;
362 _SAFE_FREE_(*min)
363 _ERR_EXIT_
364 }
365 /*setup the minmax*/
366 for (c = 0; c < pImage->bChannels; c++)
367 {
368 switch(pImage->dwType)
369 {
370 case AWP_BYTE:
371 (*min)[c] = ((AWPBYTE*)pImage->pPixels)[c];
372 (*max)[c] = ((AWPBYTE*)pImage->pPixels)[c];
373 break;
374 case AWP_SHORT:
375 (*min)[c] = ((AWPSHORT*)pImage->pPixels)[c];
376 (*max)[c] = ((AWPSHORT*)pImage->pPixels)[c];
377 break;
378 case AWP_FLOAT:
379 (*min)[c] = ((AWPFLOAT*)pImage->pPixels)[c];
380 (*max)[c] = ((AWPFLOAT*)pImage->pPixels)[c];
381 break;
382 case AWP_DOUBLE:
383 (*min)[c] = ((AWPDOUBLE*)pImage->pPixels)[c];
384 (*max)[c] = ((AWPDOUBLE*)pImage->pPixels)[c];
385 break;
386 }
387 }
388 /*find the minmax*/
389 k = 0;
390 for(i = 0; i < pImage->sSizeY; i++)
391 {
392 for (j = 0; j < pImage->sSizeX; j++)
393 {
394 for (c = 0; c < pImage->bChannels; c++)
395 {
396 switch(pImage->dwType)
397 {
398 case AWP_BYTE:
399 if ((*min)[c] > ((AWPBYTE*)pImage->pPixels)[k])
400 (*min)[c] = ((AWPBYTE*)pImage->pPixels)[k];
401 if ((*max)[c] < ((AWPBYTE*)pImage->pPixels)[k])
402 (*max)[c] = ((AWPBYTE*)pImage->pPixels)[k];
403 break;
404 case AWP_SHORT:
405 if ((*min)[c] > ((AWPSHORT*)pImage->pPixels)[k])
406 (*min)[c] = ((AWPSHORT*)pImage->pPixels)[k];
407 if ((*max)[c] < ((AWPSHORT*)pImage->pPixels)[k])
408 (*max)[c] = ((AWPSHORT*)pImage->pPixels)[k];
409 break;
410 case AWP_FLOAT:
411 if ((*min)[c] > ((AWPFLOAT*)pImage->pPixels)[k])

413 if ((*max)[c] < ((AWPFLOAT*)pImage->pPixels)[k])
