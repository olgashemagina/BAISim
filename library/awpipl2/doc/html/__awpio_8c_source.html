27AWPRESULT _awpGetFileExt(const char* lpFileName, char* lpFileExt)
28{
29 AWPINT ilen;
30 AWPINT iextlen;
31 AWPINT i;
32 char test;
33 if (lpFileName == NULL || lpFileExt == NULL)
34 return AWP_BADARG;
35
36 ilen = strlen(lpFileName);
37 iextlen = 0;
38 i = ilen-1;
39 if (ilen == 0)
40 return AWP_BADARG;
41
42 while(iextlen < 5 || i > 0)
43 {
44
45 test = lpFileName[i];
46 if (test == '.')
47 break;
48 i--;
49 iextlen++;
50 }
51 strncpy(lpFileExt, &lpFileName[i+1], iextlen);
52 lpFileExt[iextlen] = '\0';
53 return AWP_OK;
54}

65AWPRESULT _awpLoadAWPImageV1(const char* lpFileName, awpImage** ppImage)
66{
67 /*common variables*/
68 FILE* f; /*the file*/
69 awpImage* p_tmp_image;
70 awpImage* Header;
71 AWPRESULT res;
72 AWPDWORD dwPixelSize = 0;
73 /*header variables*/
74 AWPDWORD dwMagic; /*magic number. must be 0x0A0D5C */
75 AWPBYTE bVersion; /*version number. must be 1*/
76 AWPWORD wSizeX; /*width of the image. Can be 1-65535*/
77 AWPWORD wSizeY; /*height of the image. Can be 1-65535*/
78 AWPDWORD dwType; /*Type of the image. Can be
79 0 - AWP_COLOR
80 1 - AWP_GRAYSCALE
81 2 - AWP_SHORT
82 3 - AWP_PALETTE
83 4 - AWP_FLOAT*/
84 AWPDWORD dwSize; /*size of pixels*/
85 AWPBYTE* pbBuffer;/*buffer for pixels*/
86 /*awp image creation variables*/
87 AWPBYTE bNumChannels; /*number of channels*/
88 AWPDWORD dwTypeImage; /*type of image*/
89 pbBuffer = NULL; /*init variable*/
90 f = fopen(lpFileName, "r+b");
91 if (f == NULL)
92 return AWP_OPEN_FILE_ERROR;
93 /*read file header varibles*/
94 Header = (awpImage*)malloc(sizeof(awpImage));
95 if (Header == NULL)
96 return AWP_BADMEMORY;
97 if (fread(Header, sizeof(awpImage), 1, f) != 1)
98 {
99 res = AWP_READ_FILE_ERROR;
100 _ERR_EXIT_
101 }
102 /*1. magic*/
103 dwMagic = Header->nMagic;
104
105 if (dwMagic != 0x0A0D5C)
106 {
107 res = AWP_NOT_AWP_FILE;
108 _ERR_EXIT_
109 }
110 /*2.version*/
111 bVersion = Header->bChannels;
112 if (bVersion != 1)
113 {
114 res = AWP_NOT_AWP_FILE;
115 _ERR_EXIT_
116 }
117 /*3.sizes*/
118 wSizeX = Header->sSizeX;
119 if (wSizeX == 0)
120 {
121 res = AWP_CORRUPT_FILE;
122 _ERR_EXIT_
123 }
124 wSizeY = Header->sSizeY;
125 if (wSizeY == 0)
126 {
127 res = AWP_CORRUPT_FILE;
128 _ERR_EXIT_
129 }
130 /*4. type*/
131 dwType = Header->dwType;
132 if (dwType > 4)
133 {
134 res = AWP_NOT_AWP_FILE;
135 _ERR_EXIT_
136 }
137 dwSize = (AWPDWORD)Header->pPixels;
138 /*find the size of one pixel and variables for creation awp image*/
139 switch(dwType)
140 {
141 case 0:
142 dwPixelSize = sizeof(OldAwpColor);
143 bNumChannels = 3;
144 dwTypeImage = AWP_BYTE;
145 break;
146 case 1:
147 dwPixelSize = sizeof(AWPBYTE);
148 bNumChannels = 1;
149 dwTypeImage = AWP_BYTE;
150 break;
151 case 2:
152 dwPixelSize = sizeof(AWPSHORT);
153 bNumChannels = 1;
154 dwTypeImage = AWP_SHORT;
155 break;
156 case 3:
157 dwPixelSize = sizeof(AWPBYTE);
158 bNumChannels = 1;
159 dwTypeImage = AWP_BYTE;
160 break;
161 case 4:
162 dwPixelSize = sizeof(AWPFLOAT);
163 bNumChannels = 1;
164 dwTypeImage = AWP_FLOAT;
165 break;
166 }
167 /*check the sizes*/
168 dwSize -= sizeof(awpImage);
169 if (dwPixelSize*wSizeX*wSizeY != dwSize )
170 {
171 res = AWP_CORRUPT_FILE;
172 _ERR_EXIT_
173 }
174 /*read the pixels*/
175 pbBuffer = (AWPBYTE*)malloc(dwPixelSize*wSizeX*wSizeY);
176 if (pbBuffer == NULL)
177 {
178 res = AWP_BADMEMORY;
179 _ERR_EXIT_
180 }
181 if (fread(pbBuffer, dwPixelSize*wSizeX*wSizeY, 1, f) != 1)
182 {
183 res = AWP_READ_FILE_ERROR;
184 _ERR_EXIT_
185 }
186 /*the file in the memory. now convert the memory to awpImage*/
187 _CHECK_RESULT_((res = awpCreateImage(&p_tmp_image, wSizeX, wSizeY, bNumChannels, dwTypeImage)))
188 /*copy the buffers*/
189 memcpy((AWPBYTE*)p_tmp_image->pPixels,pbBuffer, dwPixelSize*wSizeX*wSizeY);
190 *ppImage = p_tmp_image;
191CLEANUP:
192 _SAFE_FREE_(Header)
193 _SAFE_FREE_(pbBuffer)
194 if (f != NULL)
195 fclose(f);
196 return res;
197}

200AWPRESULT _awpLoadAWPImage(const char* lpFileName, awpImage** ppImage)
201{
202 FILE* f; /*file*/
203 awpImage* p_tmp_image; /*image*/
204 AWPRESULT res;
205 AWPINT isize,isize1;
206 AWPINT itmp;
207 *ppImage = NULL;
208 /*open file*/
209//#ifdef _USE_UNICODE_VS_

211// wchar_t* wstr = (wchar_t*)malloc(wchars_num*sizeof(wchar_t));
212// MultiByteToWideChar(CP_UTF8, 0, lpFileName, -1, wstr, wchars_num);
213//
214// f = _wfopen(wstr, L"r+b");
215// free(wstr);
216//#else
217 f = fopen(lpFileName, "r+b");
218//#endif
219 if (f == NULL)
220 return AWP_OPEN_FILE_ERROR;
221
222// *ppImage = NULL;
223// /*open file*/
224// wstr = (wchar_t*)malloc(wchars_num*sizeof(wchar_t));
225// wchars_num = MultiByteToWideChar(CP_UTF8, 0, lpFileName, -1, NULL, 0);
226// MultiByteToWideChar(CP_UTF8, 0, lpFileName, -1, wstr, wchars_num);
227//
228//
229//
230// f = _wfopen(wstr, L"r+b");
231// free(wstr);
232 if (f == NULL)
233 return AWP_OPEN_FILE_ERROR;
234
235 /*alloc memory for image */
236 p_tmp_image = NULL;
237 p_tmp_image = (awpImage*)malloc(sizeof(awpImage));
238
239 if (p_tmp_image == NULL)
240 {
241 fclose(f);
242 return AWP_NOTENOUGH_MEM;
243 }
244
245 /*read header from file*/
246 /*read magic number */
247 if( fread( &p_tmp_image->nMagic, sizeof(p_tmp_image->nMagic), 1, f) != 1 )
248 {
249 fclose( f );
250 free( p_tmp_image );
251 return AWP_READ_FILE_ERROR;
252 }
253
254 /*check header*/
255 if (p_tmp_image->nMagic != AWP_MAGIC2)
256 {
257 fclose( f );
258 free( p_tmp_image );
259 res = _awpLoadAWPImageV1(lpFileName, ppImage);
260 return res;
261 }
262
263 /*read number channels */
264 if( fread( &p_tmp_image->bChannels, sizeof(p_tmp_image->bChannels), 1, f) != 1 )
265 {
266 fclose( f );
267 free( p_tmp_image );
268 return AWP_READ_FILE_ERROR;
269 }
270
271 /*read X size */
272 if( fread( &p_tmp_image->sSizeX, sizeof(p_tmp_image->sSizeX), 1, f) != 1 )
273 {
274 fclose( f );
275 free( p_tmp_image );
276 return AWP_READ_FILE_ERROR;
277 }
278
279 /*read Y size */
280 if( fread( &p_tmp_image->sSizeY, sizeof(p_tmp_image->sSizeY), 1, f) != 1 )
281 {
282 fclose( f );
283 free( p_tmp_image );
284 return AWP_READ_FILE_ERROR;
285 }
286
287 /*read type image */
288 if( fread( &p_tmp_image->dwType, sizeof(p_tmp_image->dwType), 1, f) != 1 )
289 {
290 fclose( f );
291 free( p_tmp_image );
292 return AWP_READ_FILE_ERROR;
293 }
294
295 /*read size image */
296 if( fread( &isize, sizeof(isize), 1, f) != 1 )
297 {
298 fclose( f );
299 free( p_tmp_image );
300 return AWP_READ_FILE_ERROR;
301 }
302
303 itmp = ftell(f);
304
305 /*image sizes */
306 fseek( f, 0, SEEK_SET);
307 fseek( f, 0, SEEK_END);
308 isize1 = ftell(f);
309 fseek( f, itmp, SEEK_SET);
310
311 if (isize != (AWPINT)(isize1-itmp) )
312 {
313 fclose( f );
314 free( p_tmp_image );
315 return AWP_CORRUPT_FILE;
316 }
317
318 /*find size of pixels*/
319 isize = p_tmp_image->bChannels*p_tmp_image->sSizeX*p_tmp_image->sSizeY;
320 switch(p_tmp_image->dwType)
321 {
322 case AWP_BYTE: isize *= sizeof(AWPBYTE); break;
323 case AWP_SHORT: isize *= sizeof(AWPSHORT); break;
324 case AWP_FLOAT: isize *= sizeof(AWPFLOAT);break;
325 case AWP_DOUBLE: isize *= sizeof(AWPDOUBLE);break;
326 default:
327 {
328 fclose( f );
329 free( p_tmp_image );
330 return AWP_CORRUPT_FILE;
331 }
332 }
333 /*alloc memory for pixels*/
334 p_tmp_image->pPixels = NULL;
335 p_tmp_image->pPixels = malloc(isize);
336 if (p_tmp_image->pPixels == NULL)
337 {
338 fclose( f );
339 free( p_tmp_image );
340 return AWP_NOTENOUGH_MEM;
341 }
342
343 /*read data form file*/
344 if (fread( p_tmp_image->pPixels, isize, 1, f) != 1)
345 {
346 fclose( f );
347 free( p_tmp_image->pPixels );
348 free( p_tmp_image );
349 return AWP_READ_FILE_ERROR;
350 }
351 /*all went well*/
352 *ppImage = p_tmp_image;
353 fclose(f);
354 return AWP_OK;
355}

493AWPRESULT _awpLoadJPEGImage(const char* lpFileName, awpImage** ppImage)
494{
495 awpImage* pImage;
496 AWPRESULT res;
497
498 struct jpeg_decompress_struct cinfo;
499 struct _awp_jpg_error_mgr jerr;
500 AWPBYTE* pDst;
501 AWPBYTE* src;
502 AWPINT type_image;
503 AWPBYTE iChannels;
504 JDIMENSION i;
505 FILE* infile; /* source file */
506 JSAMPARRAY buffer; /* Output row buffer */
507 AWPINT row_stride; /* physical row width in output buffer */
508
509
510 // res = AWP_OK;/*initialize result */
511 infile=NULL;
512
513
514 if (lpFileName == NULL){
515 res = AWP_BADARG;
516 _ERR_EXIT_
517 }
518
519// wchars_num = MultiByteToWideChar(CP_UTF8, 0, lpFileName, -1, NULL, 0);
520// wstr = (wchar_t*)malloc(wchars_num * sizeof(wchar_t));
521// MultiByteToWideChar(CP_UTF8, 0, lpFileName, -1, wstr, wchars_num);
522//
523// infile = _wfopen(wstr, L"rb");
524// free(wstr);
525
526//#ifdef _USE_UNICODE_VS_
527// const int wchars_num = MultiByteToWideChar(CP_UTF8, 0, lpFileName, -1, NULL, 0);
528// wchar_t* wstr = (wchar_t*)malloc(wchars_num * sizeof(wchar_t));
529// MultiByteToWideChar(CP_UTF8, 0, lpFileName, -1, wstr, wchars_num);
530//
531// infile = _wfopen(wstr, L"rb");
532// free(wstr);
533//#else
534 infile = fopen(lpFileName, "rb");
535//#endif
536 if (infile == NULL){
537 res = AWP_OPEN_FILE_ERROR;
538 _ERR_EXIT_
539 }
540
541 cinfo.err = jpeg_std_error(&jerr.pub);
542 jerr.pub.error_exit = _awp_jpg_error_exit;
543
544 if (setjmp(jerr.setjmp_buffer))
545 {
546 jpeg_destroy_decompress(&cinfo);
547 fclose(infile);
548 res = AWP_BADIMAGE_FORMAT;
549 _ERR_EXIT_
550 }
551
552 jpeg_create_decompress(&cinfo);
553 jpeg_stdio_src(&cinfo, infile);
554 jpeg_read_header(&cinfo, TRUE);
555 jpeg_start_decompress(&cinfo);
556
557 row_stride = cinfo.output_width * cinfo.output_components;
558
559 type_image=AWP_BYTE;
560
561 switch(cinfo.output_components){
562 case 1:
563 iChannels=1;
564 break;
565 case 3:
566 iChannels=3;
567 break;
568 default:
569 res=AWP_BADIMAGE_FORMAT;
570 _ERR_EXIT_
571 }
572
573 _CHECK_RESULT_(( res=awpCreateImage(&pImage,(AWPWORD)cinfo.output_width,(AWPWORD)cinfo.output_height,iChannels,type_image)))
574
575 buffer = (*cinfo.mem->alloc_sarray)
576 ((j_common_ptr) &cinfo, JPOOL_IMAGE, row_stride, 1);
577
578 pDst = (AWPBYTE*)pImage->pPixels;
579 while (cinfo.output_scanline < cinfo.output_height)
580 {
581 jpeg_read_scanlines(&cinfo, buffer, 1);
582 src = (AWPBYTE*)buffer[0];
583 if (iChannels==3)
584 {
585 for (i = 0; i < cinfo.output_width; i++)
586 {
587 pDst[3*i + 2] = src[0];
588 pDst[3*i + 1] = src[1];
589 pDst[3*i] = src[2];
590 src += 3;
591 }
592 }
593 else
594 memcpy(pDst, buffer[0], row_stride);
595
596 pDst += row_stride;
597 }
598
599 jpeg_finish_decompress(&cinfo);
600 jpeg_destroy_decompress(&cinfo);
601
602 /* all OK */
603 *ppImage=pImage;
604
605CLEANUP:
606 if (infile != NULL)
607 fclose(infile);
608
609 return res;
610}

612AWPRESULT _awpSaveJPEGImage(const char* lpFileName, awpImage* pImage)
613{
614 struct jpeg_compress_struct cinfo;
615 struct _awp_jpg_error_mgr jerr;
616 FILE *fp;
617 AWPINT i;
618 AWPBYTE *line;
619 AWPINT iBytes;
620 AWPRESULT res;
621
622 if (lpFileName == NULL)
623 return AWP_BADARG;
624
625 // res = AWP_OK;/*initialize result */
626 fp=NULL;
627 line=NULL;
628
629 _CHECK_RESULT_((res = awpCheckImage(pImage)))
630
631 if( (pImage->dwType==AWP_BYTE) && ( (pImage->bChannels==1) || (pImage->bChannels==3))){
632 iBytes=pImage->bChannels;
633 }else{
634 res= AWP_NOTSUPPORT;
635 _ERR_EXIT_
636 }
637
638 if(pImage->dwType==AWP_BYTE && pImage->bChannels==3)
639 SwapBytes(pImage);
640
641 if (NULL == (fp = fopen(lpFileName,"wb"))){
642 res = AWP_CREATE_FILE_ERROR;
643 _ERR_EXIT_
644 }
645
646 cinfo.err = jpeg_std_error(&jerr.pub);
647 jerr.pub.error_exit = _awp_jpg_error_exit;
648
649 if (setjmp(jerr.setjmp_buffer))
650 {
651 jpeg_destroy_compress(&cinfo);
652 fclose(fp);
653 res = AWP_BADIMAGE_FORMAT;
654 _ERR_EXIT_
655 }
656
657 jpeg_create_compress(&cinfo);
658 jpeg_stdio_dest(&cinfo, fp);
659 cinfo.image_width = pImage->sSizeX;
660 cinfo.image_height = pImage->sSizeY;
661 cinfo.input_components = iBytes;
662 cinfo.in_color_space = iBytes==1 ? JCS_GRAYSCALE : JCS_RGB;
663 jpeg_set_defaults(&cinfo);
664 jpeg_set_quality(&cinfo, 100, TRUE);
665 jpeg_start_compress(&cinfo, TRUE);
666
667 for (i = 0, line = (AWPBYTE*)pImage->pPixels; i < pImage->sSizeY; i++, line += pImage->sSizeX*iBytes)
668 jpeg_write_scanlines(&cinfo, &line, 1);
669
670 jpeg_finish_compress(&(cinfo));
671 jpeg_destroy_compress(&(cinfo));
672
673CLEANUP:
674 if (fp != NULL)
675 fclose(fp);
676
677 return res;
678}

703GLOBAL(void)
704jpeg_buffer_src (j_decompress_ptr cinfo, const char* buffer, AWPINT len )
705{
706 struct jpeg_source_mgr* src;
707
708 /* The source object and input buffer are made permanent so that a series
709 * of JPEG images can be read from the same file by calling jpeg_stdio_src
710 * only before the first one. (If we discarded the buffer at the end of
711 * one image, we'd likely lose the start of the next one.)
712 * This makes it unsafe to use this manager and a different source
713 * manager serially with the same JPEG object. Caveat programmer.
714 */
715 if (cinfo->src == NULL) { /* first time for this JPEG object? */
716 cinfo->src = (struct jpeg_source_mgr *)
717 (*cinfo->mem->alloc_small) ((j_common_ptr) cinfo, JPOOL_PERMANENT,
718 sizeof(struct jpeg_source_mgr));
719 }
720
721 src = cinfo->src;
722 src->init_source = init_source;
723 src->fill_input_buffer = fill_input_buffer;
724 src->skip_input_data = skip_input_data;
725 src->resync_to_restart = jpeg_resync_to_restart; /* use default method */
726 src->term_source = term_source;
727 src->next_input_byte = (JOCTET *)buffer;
728 src->bytes_in_buffer = len;
729}

749GLOBAL(void)
750jpeg_buffer_dest (j_compress_ptr cinfo,char* buffer, AWPINT len )
751{
752 struct jpeg_destination_mgr* dest;
753
754 /* The destination object is made permanent so that multiple JPEG images
755 * can be written to the same file without re-executing jpeg_stdio_dest.
756 * This makes it dangerous to use this manager and a different destination
757 * manager serially with the same JPEG object, because their private object
