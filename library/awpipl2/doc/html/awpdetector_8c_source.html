11static AWPFLOAT _awpRectsOverlap(awpRect r1, awpRect r2)
12{
13 AWPFLOAT colInt, rowInt, intersection, area1, area2;
14 AWPINT width = r1.right - r1.left;
15 AWPINT height = r1.bottom - r1.top;
16 AWPINT _width = r2.right - r2.left;
17 AWPINT _height = r2.bottom - r2.top;
18 if (r2.left > r1.left + width)
19 return 0;
20 if (r2.top > r1.top + height)
21 return 0;
22 if (r2.left + _width < r1.left)
23 return 0;
24 if (r1.top + _height < r1.top)
25 return 0;
26
27 colInt = (AWPFLOAT)AWP_MIN(r2.left + _width, r1.left + width) - AWP_MAX(r1.left, r1.left);
28 rowInt = (AWPFLOAT)AWP_MIN(r2.top + _height, r1.top + height) - AWP_MAX(r2.top, r1.top);
29
30 intersection = colInt*rowInt;
31 area1 = (AWPFLOAT)(_width*_height);
32 area2 = (AWPFLOAT)(width*height);
33
34 return intersection / (area1+area2-intersection);
35}

37static void _awpClusterRects(int numSrc, awpRect* pSrc, int* numDst, awpRect** ppDst)
38{
39 AWPINT i,j, NumClusters, idx, eCount, cidx;
40 AWPBOOL found;
41 AWPFLOAT overlap;
42 awpRect r;
43 cawpRect* crects = (cawpRect*)malloc(numSrc*sizeof(cawpRect));
44 for ( i = 0; i < numSrc; i++)
45 {
46 crects[i].r = pSrc[i];
47 crects[i].c = -1;
48 }
49
50 NumClusters = 0;
51 found = FALSE;
52 idx = 0;
53 do
54 {
55 found = FALSE;
56 for ( i = 0; i < numSrc; i++)
57 {
58 if (crects[i].c == -1)
59 {
60 found = TRUE;
61 NumClusters++;
62 crects[i].c = NumClusters;
63 idx = i;
64 break;
65 }
66 }
67 if (found)
68 {
69 for ( i = 0; i < numSrc; i++)
70 {
71 if (i == idx)
72 continue;
73 if (crects[i].c != -1)
74 continue;
75 overlap = _awpRectsOverlap(crects[i].r , crects[idx].r);
76 if (overlap > 0.5)
77 {
78 crects[i].c = NumClusters;
79 }
80 }
81 }
82 }while(found);
83
84
85 eCount = 0;
86 cidx = 0;
87 (*numDst) = NumClusters;
88 *ppDst = (awpRect*)malloc(NumClusters*sizeof(awpRect));
89 for ( i = 0; i < NumClusters; i++)
90 {
91 cidx = i+1;
92 r.left = 0;
93 r.right = 0;
94 r.bottom = 0;
95 r.top = 0;
96 eCount = 0;
97 for ( j = 0; j < numSrc; j++)
98 {
99 if (crects[j].c == cidx)
100 {
101 r.left += crects[j].r.left;
102 r.bottom += crects[j].r.bottom;
103 r.right += crects[j].r.right;
104 r.top += crects[j].r.top;
105 eCount++;
106 }
107 }
108 r.left /= eCount;
109 r.right /= eCount;
110 r.bottom /= eCount;
111 r.top /= eCount;
112 (*ppDst)[i] = r;
113 }
114

127static AWPBOOL GMCTDetector(awpImage* pImage, awpImage* pImage1, awpRect* pRect, awpDetector* pDetector)
128{
129 AWPINT i,j,sx,sy, sw, sh, idx, x, y, width, height,ww;
130 AWPDOUBLE tarea, area, total, alfa, sum, invarea,s2,m,disp2;
131 AWPDOUBLE* pix;
132 AWPDOUBLE* pix1;
133 AWPINT iv[9];
134 awpWeak* weak;
135 alfa = (AWPDOUBLE)(pRect->right - pRect->left)/(AWPDOUBLE)pDetector->width;
136 pix = (AWPDOUBLE*)pImage->pPixels;
137 pix1 = NULL;
138 if (pImage1)
139 pix1 = (AWPDOUBLE*)pImage1->pPixels;
140
141 x = pRect->left;
142 y = pRect->top;
143 width = pRect->right - x;
144 height = pRect->bottom - y;
145 ww = pImage->sSizeX;
146 invarea = 1.0/(double)((width-1) * (height -1));
147 s2 = 1;
148 if (pix1 != NULL)
149 s2 = CalcSum(pix1, x+1, y+1, width-1, height-1 , ww) * invarea;
150
151 m = CalcSum(pix, x+1, y+1, width-1, height-1, ww )*invarea;
152 disp2 = s2 - m*m;
153
154 if (pix1 != NULL)
155 {
156 if ( (disp2 < 25*25) || ( m < 25) || (m > 195 ) )
157 return FALSE;
158 }
159
160
161 for ( i = 0; i < pDetector->nStages; i++)
162 {
163 sum = 0;
164 for (j = 0; j < pDetector->pStrongs[i].nWeaks; j++)
165 {
166 memset(iv, 0, sizeof(iv));
167
168 weak = &pDetector->pStrongs[i].pWeaks[j];
169 tarea = 1.0/(9.0*alfa*alfa*weak->w*weak->h);
170 area = 1.0/(alfa*alfa*weak->w*weak->h);
171 sx = pRect->left + (AWPINT)(alfa*weak->x + 0.5);
172 sy = pRect->top + (AWPINT)(alfa*weak->y + 0.5);
173 sw = (AWPINT)(alfa*weak->w + 0.5);
174 sh = (AWPINT)(alfa*weak->h + 0.5);
175
176 total = CalcSum(pix, sx, sy, 3*sw, 3*sh, ww)*tarea;
177
178 iv[0] = CalcSum(pix, sx, sy, sw, sh, ww) * area > total?1:0;
179 iv[1] = CalcSum(pix, sx + sw, sy, sw, sh, ww) * area> total?1:0;
180 iv[2] = CalcSum(pix, sx + 2*sw, sy, sw, sh, ww) * area> total?1:0;
181 iv[3] = CalcSum(pix, sx, sy + sh, sw, sh, ww) * area> total?1:0;
182 iv[4] = CalcSum(pix, sx + sw, sy + sh, sw, sh, ww) * area> total?1:0;
183 iv[5] = CalcSum(pix, sx + 2*sw, sy + sh, sw, sh, ww) * area> total?1:0;
184 iv[6] = CalcSum(pix, sx, sy + 2*sh, sw, sh, ww) * area> total?1:0;
185 iv[7] = CalcSum(pix, sx + sw, sy + 2*sh, sw, sh,ww) * area> total?1:0;
186 iv[8] = CalcSum(pix, sx + 2*sw, sy + 2*sh, sw, sh, ww) * area> total?1:0;
187
188
189 idx = 0;
190 idx |= iv[0];
191 idx =idx << 1;
192 idx |= iv[1];
193 idx =idx << 1;
194 idx |= iv[2];
195 idx =idx << 1;
196 idx |= iv[3];
197 idx =idx << 1;
198 idx |= iv[4];
199 idx =idx << 1;
200 idx |= iv[5];
201 idx =idx << 1;
202 idx |= iv[6];
203 idx =idx << 1;
204 idx |= iv[7];
205 idx =idx << 1;
206 idx |= iv[8];
207
208 sum += weak->bins[idx] > 0 ? weak->alfa : 0;
209 }
210
211 if ((sum - pDetector->pStrongs[i].alfa) < -0.00001)
212 return FALSE;
213 }

276 d->pStrongs[i].pWeaks[j].bins[k] = data[k+5];
