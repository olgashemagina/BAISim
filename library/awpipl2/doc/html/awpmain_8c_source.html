48int InputKey(int argc, char **argv, const char* key)
49{
50 int i = 0;
51 int result = -1;
52 for (i = 2; i < argc; i++)
53 {
54 if (strcmp(argv[i], key) == 0)
55 {
56 result = i + 1;
57 break;
58 }
59 }
60 if (result < 0)
61 {
62 printf("cannot find key %s.\n", key);
63 exit(-1);
64 }
65
66 return result;
67}
