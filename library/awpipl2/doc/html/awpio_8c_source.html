32AWPRESULT awpLoadImage(const char* lpFileName, awpImage** ppImage)
33{
34 /*variables*/
35 AWPRESULT res;
36 char lpExt[5];
37 /*arguments*/
38 if (lpFileName == NULL || ppImage == NULL)
39 return AWP_BADARG;
40
41 res = AWP_OK;
42
43 /*obtain file extention*/
44 res = _awpGetFileExt(lpFileName, lpExt);
45 if (res != AWP_OK)
46 return AWP_BADARG;
47
48 if (strcmp(lpExt, "awp") == 0 || strcmp(lpExt,"AWP") == 0)
49 res = _awpLoadAWPImage(lpFileName, ppImage);
50#ifdef HAVE_JPEG
51#if defined(WIN32) || defined(_WIN64)
52//#if 0
53 else if (strcmp(lpExt, "bmp") == 0 || strcmp(lpExt,"BMP") == 0)
54 res = _awpLoadBMPImage(lpFileName, ppImage);
55 else if (strcmp(lpExt, "ppm") == 0 || strcmp(lpExt,"PPM") == 0)
56 res = _awpLoadPPMImage(lpFileName, ppImage);
57 else if (strcmp(lpExt, "tga") == 0 || strcmp(lpExt,"TGA") == 0)
58 res = _awpLoadTGAImage(lpFileName, ppImage);
59//#endif
60#endif
61 else if (strcmp(lpExt, "jpg") == 0 || strcmp(lpExt,"JPG") == 0 || strcmp(lpExt, "jpeg") == 0 || strcmp(lpExt,"jpeg") == 0 || strcmp(lpExt,"JPEG") == 0)
62 res = _awpLoadJPEGImage(lpFileName, ppImage);
63#endif /* HAVE_JPEG */
64 else
65 return _awpLoadAWPImage(lpFileName, ppImage);
66
67 /*all went well*/
68 return res;
69}

112AWPRESULT awpJpegToImage( const AWPBYTE* pJpegData, AWPDWORD length, awpImage** ppImage )
113{
114 AWPRESULT res;
115
116 struct my_error_mgr jpeg_err;
117 struct jpeg_decompress_struct in_jpeg;
118 AWPBYTE* pDst;
119 JSAMPARRAY buffer; /* Output row buffer */
120 AWPINT row_stride;
121 AWPINT rows;
122 JDIMENSION i;
123 AWPBYTE* src = NULL;
124
125 //set error function
126 in_jpeg.err = jpeg_std_error( &jpeg_err.pub );
127 jpeg_err.pub.error_exit = my_error_exit;
128
129 if (setjmp(jpeg_err.setjmp_buffer))
130 {
131 jpeg_destroy_decompress(&in_jpeg);
132 return AWP_BADIMAGE_FORMAT;
133 }
134
135 res = AWP_OK;
136
137 jpeg_create_decompress( &in_jpeg );
138
139 /* read from buffer */
140 jpeg_buffer_src( &in_jpeg, (char*)pJpegData, length );
141 jpeg_read_header( &in_jpeg, TRUE );
142
143
144 if(!(in_jpeg.out_color_space != JCS_RGB || in_jpeg.out_color_space != JCS_GRAYSCALE)){
145 res = AWP_NOTSUPPORT;
146 _ERR_EXIT_
147 }
148
149 jpeg_start_decompress( &in_jpeg );
150
151 row_stride = in_jpeg.output_width*in_jpeg.output_components;
152 rows = in_jpeg.output_height;
153
154 /* create awp image */
155 _CHECK_RESULT_((res = awpCreateImage(ppImage,(AWPWORD)in_jpeg.output_width,(AWPWORD)in_jpeg.output_height,(AWPBYTE)(in_jpeg.out_color_space==JCS_RGB?3:1),AWP_BYTE)))
156
157 buffer = (*in_jpeg.mem->alloc_sarray)
158 ((j_common_ptr) &in_jpeg, JPOOL_PERMANENT, row_stride,rows );
159
160
161 pDst = (AWPBYTE*)((*ppImage)->pPixels);
162 while (in_jpeg.output_scanline < in_jpeg.output_height)
163 {
164 jpeg_read_scanlines(&in_jpeg, buffer, 1);
165 src = (AWPBYTE*)buffer[0];
166 for (i = 0; i < in_jpeg.output_width; i++)
167 {
168 pDst[3*i + 2] = src[0];
169 pDst[3*i + 1] = src[1];
170 pDst[3*i] = src[2];
171 src += 3;
172 }
173
174 pDst += row_stride;
175 }
176
177
178CLEANUP:
179
180 jpeg_finish_decompress( &in_jpeg );
181 jpeg_destroy_decompress( &in_jpeg );
182
183 return res;
184
185}

189AWPRESULT awpImageToJpeg( const awpImage* pImage, AWPBYTE** ppJpegData, AWPDWORD* length, AWPBYTE quality )
190{
191 AWPRESULT res;
192
193
194 struct jpeg_error_mgr jpeg_err;
195 struct jpeg_compress_struct out_jpeg;
196 AWPBYTE* pSrc;
197 AWPINT i;
198
199 JSAMPARRAY fp; /* Output row buffer */
200 AWPINT stride;
201 quality=100;
202 res = AWP_OK;
203 pSrc=NULL;
204
205 out_jpeg.err = jpeg_std_error( &jpeg_err );
206 jpeg_create_compress( &out_jpeg );
207
208 stride=pImage->sSizeX*pImage->bChannels;
209
210 fp = out_jpeg.mem->alloc_sarray( (j_common_ptr)&out_jpeg,
211 JPOOL_PERMANENT, stride, pImage->sSizeX );
212
213 pSrc=(AWPBYTE*)pImage->pPixels;
214
215 for(i=0;i<pImage->sSizeY;i++){
216 memcpy(&fp[i],&pSrc[i*pImage->sSizeX*pImage->bChannels],pImage->sSizeX*pImage->bChannels);
217 }
218
219 jpeg_buffer_dest( &out_jpeg, (char*) ppJpegData,*length );
220 out_jpeg.image_width = pImage->sSizeX;
221 out_jpeg.image_height = pImage->sSizeY;
222 out_jpeg.input_components = pImage->bChannels;
223
224 out_jpeg.in_color_space = JCS_RGB;
225
226 jpeg_set_defaults( &out_jpeg );
227 jpeg_start_compress( &out_jpeg, TRUE );
228
229 for( i=0; i<pImage->sSizeY; i++ )
230 jpeg_write_scanlines( &out_jpeg, &fp[i], 1 );
231
232 jpeg_finish_compress( &out_jpeg );
233
234
235 jpeg_destroy_compress( &out_jpeg );
236
237 return res;
238}

243AWPRESULT awpImagePack( const awpImage* pImage, AWPBYTE** ppData, AWPINT* length )
244{
245 AWPRESULT res;
246 long c;
247 AWPBYTE* pData;
248 AWPINT isize;
249
250 res = AWP_OK;
251 c=0;
252 pData=NULL;
253 *length=0;
254
255 /*check input image*/
256 _CHECK_RESULT_((res = awpCheckImage(pImage)))
257
258 /* calculate size image data */
259 _CHECK_RESULT_((res = awpGetImageSize(pImage, length)))
260
261 /* get size header */
262 _CHECK_RESULT_((res = awpGetImageHeaderSize(pImage,&isize)))
263
264 /* allocate data */
265
266 *length+=isize;
267 pData = (AWPBYTE*)malloc (*length);
268
269 if(pData==NULL){
270 res=AWP_BADMEMORY;
271 _ERR_EXIT_
272 }
273
274 /* copy all member in massive*/
275 memcpy(&pData[c],&pImage->nMagic,sizeof(pImage->nMagic));
276 c+=sizeof(pImage->nMagic);
277 memcpy(&pData[c],&pImage->bChannels,sizeof(pImage->bChannels));
278 c+=sizeof(pImage->bChannels);
279 memcpy(&pData[c],&pImage->sSizeX,sizeof(pImage->sSizeX));
280 c+=sizeof(pImage->sSizeX);
281 memcpy(&pData[c],&pImage->sSizeY,sizeof(pImage->sSizeY));
282 c+=sizeof(pImage->sSizeY);
283 memcpy(&pData[c],&pImage->dwType,sizeof(pImage->dwType));
284 c+=sizeof(pImage->dwType);
285 /* copy image data*/
286 memcpy(&pData[c],pImage->pPixels,*length-isize);
287
288 *ppData=pData;
289
290CLEANUP:
291
292 return res;
293}

296AWPRESULT awpImageUnpack( const AWPBYTE* pData, AWPINT length, awpImage** ppImage )
297{
298 AWPRESULT res;
299 awpImage *pImage;
300 long c;
301 AWPDWORD nMagic;
302 AWPBYTE bChannels;
303 AWPWORD sSizeX,sSizeY;
304 AWPDWORD dwType;
305 AWPINT iSize;
306
307 res = AWP_OK;
308 c=0;
309 pImage=NULL;
310
311 /* check input params */
312 if(pData==NULL || length==0){
313 res = AWP_BADARG;
314 _ERR_EXIT_
315 }
316
317 /* get image header in massive */
318 memcpy(&nMagic,&pData[c],sizeof(nMagic));
319 c+=sizeof(nMagic);
320
321 if(nMagic!=AWP_MAGIC2){
322 res=AWP_NOTAWPIMAGE;
323 _ERR_EXIT_
324 }
325
326 memcpy(&bChannels,&pData[c],sizeof(bChannels));
327 c+=sizeof(bChannels);
328 memcpy(&sSizeX,&pData[c],sizeof(sSizeX));
329 c+=sizeof(sSizeX);
330 memcpy(&sSizeY,&pData[c],sizeof(sSizeY));
331 c+=sizeof(sSizeY);
332 memcpy(&dwType,&pData[c],sizeof(dwType));
333 c+=sizeof(dwType);
334
335 /* create awp image */
336 _CHECK_RESULT_((res = awpCreateImage(&pImage,sSizeX,sSizeY,bChannels,dwType)))
337
338 /* get size header */
339 _CHECK_RESULT_((res = awpGetImageSize(pImage,&iSize)))
340
341 /* copy image data*/
342 memcpy(pImage->pPixels,&pData[c],iSize);
343
344 *ppImage=pImage;
345
346CLEANUP:
347
348 return res;
349}
