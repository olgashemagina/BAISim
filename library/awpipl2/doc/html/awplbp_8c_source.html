44static AWPBYTE _awpLBPCodeU1(awpImage* src, awpPoint p, AWPBOOL interpolation, awp2DPoint* pp)
45{
46 AWPBYTE lbp = 0;
47 AWPBYTE* s = (AWPBYTE*)src->pPixels;
48 AWPWORD ww = src->sSizeX*src->bChannels;
49 AWPBYTE cc = src->bChannels;
50 AWPBYTE pv = s[cc*p.X + src->bChannels*p.Y*src->sSizeX];
51 AWPINT shift = 0;
52 awpImage* intpl = NULL;
53 AWPDOUBLE* di = NULL;
54 lbp = 0;
55
56 if (interpolation)
57 {
58 awpCreateImage(&intpl, src->bChannels, 1, 1, AWP_DOUBLE);
59 di = _AWP_BPIX_(intpl, AWPDOUBLE);
60 }
61
62
63 if (!interpolation)
64 {
65 shift = (AWPINT)floor(cc*pp[0].X + pp[0].Y*ww + 0.5);
66 if (pv < s[shift])
67 lbp |= c_LBPBitShift[0];
68 }
69 else
70 {
71 awpBilinearInterpolation(src, &pp[0], intpl);
72 if (pv < di[0])
73 lbp |= c_LBPBitShift[0];
74 }
75
76 shift = (AWPINT)floor(cc*pp[1].X + pp[1].Y*ww);
77 if (pv < s[shift])
78 lbp |= c_LBPBitShift[1];
79
80 if (!interpolation)
81 {

83 if (pv < s[shift])
84 lbp |= c_LBPBitShift[2];

89 if (pv < di[0])
90 lbp |= c_LBPBitShift[2];
91 }

94 if (pv <s[shift])

238AWPRESULT awpLBPCode(awpImage* src, awpPoint p, AWPWORD shiftX, AWPWORD shiftY,
239 AWPFLOAT scale, AWPBYTE options, AWPBOOL interpolation, AWPBYTE* pcode)
240{
241 awp2DPoint pp[8];
242 awpPoint _p;
243
244 AWPDOUBLE radius = pow(c_ScaleBase, scale);
245
246
247 if (src == NULL || pcode == NULL)
248 return AWP_BADARG;
249 if (p.X < 1 || p.Y < 1)
250 return AWP_BADARG;
251 if (p.X + radius + shiftX > src->sSizeX - 1)
252 return AWP_BADARG;
253 if (p.Y + radius + shiftY > src->sSizeY - 1)
254 return AWP_BADARG;
255
256 if (options == AWP_LBP_U1R1 || options == AWP_LBP_U2R1 ||
257 options == AWP_LBP_U2R2 || options == AWP_LBP_U1R2)
258 {
259 _p = p;
260 awpLPBPoints(p, options, interpolation, pp);
261 }
262 else
263 {
264 _p.X = p.X + shiftX;
265 _p.Y = p.Y + shiftY;
266 awpLPBSPoints(p, shiftX, shiftY, scale, interpolation, pp);
267 }
268
269 switch (options)
270 {
271 case AWP_LBP_U1R1:
272 case AWP_LBP_U1R2:
273 *pcode = _awpLBPCodeU1(src, _p, interpolation, pp);
274
275 break;
276 case AWP_LBP_U2R1:
277 case AWP_LBP_U2R2:
278 *pcode = _awpLBPCodeU2(src, _p,interpolation, pp);
279 break;
280 default:
281 return AWP_BADARG;
282 }
283 return AWP_OK;
284}
285/*
286 compute LBP descriptor of grayscale image
287 the image type must be AWP_BYTE
288 number of channels must be - 1.
289 intput image size must be more then 8x8 pixels
290 output image dst must be AWP_DOUBLE type
291 sSizeY parameter must be = 1.
292 sSizeX = 256 if AWP_LBP_U1XX options is set
293 SSizeX = 59 if AWP_LBP_U2XX optinos is set
294*/
295AWPRESULT awpLBPDescriptor(awpImage* src, awpImage* dst, awpRect* roi, AWPBYTE options, AWPBOOL interpolation, AWPBOOL norm)
296{
297 AWPSHORT i;
298 AWPDOUBLE sum;
299 AWPWORD sx, sy, ex, ey;
300 AWPWORD x,y;
301 awpPoint p;
302 AWPDOUBLE* hst = NULL;
303 AWPRESULT res = AWP_OK;
304 AWPBYTE lbp = 0;
305
306 if (src == NULL || dst == NULL)
307 return AWP_BADARG;
308 if (src->bChannels != 1)
309 return AWP_BADARG;
310 if (src->dwType != AWP_BYTE)
311 return AWP_BADARG;
312 if (src->sSizeX < 8 || src->sSizeY < 8)
313 return AWP_BADARG;
314 if (dst->dwType != AWP_DOUBLE)
315 return AWP_BADARG;
316 if (dst->bChannels != 1)
317 return AWP_BADARG;
318 if (dst->sSizeY != 1)
319 return AWP_BADARG;
320 if (options == AWP_LBP_U1R1 || options == AWP_LBP_U1R2)
321 {
322 if (dst->sSizeX != 256)
323 return AWP_BADARG;
324 }
325 if (options == AWP_LBP_U2R1 || options == AWP_LBP_U2R2)
326 {
327 if (dst->sSizeX != 59)
328 return AWP_BADARG;
329 }
330
331 hst = (AWPDOUBLE*)dst->pPixels;
332
333 if (roi != NULL)
334 {
335 if (awpRectInImage(src, roi) != AWP_OK)
336 return AWP_BADARG;
337 if (roi->right - roi->left < 8 || roi->bottom - roi->top < 8)
338 return AWP_BADARG;
339 sx = roi->left;
340 ex = roi->right;
341 sy = roi->top;
342 ey = roi->bottom;
343 }
344 else
345 {
346 sx = 2;
347 ex = src->sSizeX-2;
348 sy = 2;
349 ey = src->sSizeY-2;
350 }
351
352 for (y = sy + 1; y < ey -1; y++)
353 {
354 for (x = sx + 1; x < ex -1; x++)
355 {
356 lbp = 0;
357 p.X = x;
358 p.Y = y;
359 awpLBPCode(src, p, 0,0,0,options, interpolation, &lbp);
360 hst[lbp]++;
361 }
362 }
363
364 if (res != AWP_OK)
365 return res;
366
367 if (norm)
368 {
369 //TODO: norm the descriptor
370 hst = (AWPDOUBLE*)dst->pPixels;
371 sum = 0;
372 for(i = 0; i < dst->sSizeX; i++)
373 {
374 sum += hst[i];
375 }
376
377 for ( i = 0; i < dst->sSizeX; i++)
378 {
379 hst[i] /= sum;
380 }
381 }
382 return AWP_OK;
383}
384/*
385 Performs splitting image or roi on n rows and m columns
386 And for each area it calculates its own LBP descriptor, which stores in its
387 In the image line dst.
388
389 src - grayscale image AWPBYTE, number of channles = 1
390 size of src must be more then n*8\F5m*8
391 output image must be AWPDOUBLE and has sSizeY = n*n
392 sSizeX = 256 if AWP_LBP_U1XX options is set
393 SSizeX = 59 if AWP_LBP_U2XX optinos is set
394
395*/
396AWPRESULT awpLBPGridDescriptor(awpImage* src, awpImage* dst, awpRect* roi, AWPBYTE n, AWPBYTE m, AWPBYTE options, AWPBOOL interpolation, AWPBOOL norm)
397{
398 AWPWORD w,h, sx, sy, i, j;
399 awpRect r;
400 AWPRESULT res = AWP_OK;
401 AWPDOUBLE ww,hh;
402 awpImage* hst = NULL;
403 AWPDOUBLE* b = NULL;
404 AWPDOUBLE* d = NULL;
405 if (src == NULL || dst == NULL)
406 return AWP_BADARG;
407 if (src->bChannels != 1)
408 return AWP_BADARG;
409 if (src->dwType != AWP_BYTE)
410 return AWP_BADARG;
411 if (dst->dwType != AWP_DOUBLE)
412 return AWP_BADARG;
413 if (dst->bChannels != 1)
414 return AWP_BADARG;
415
416 w = 0;
417 h = 0;
418 if (roi != NULL)
419 {
420 if (awpRectInImage(src, roi) != AWP_OK)
421 return AWP_BADARG;
422 w = roi->right - roi->left;
423 h = roi->bottom - roi->top;
424 sx = roi->left;
425 sy = roi->top;
426 }
427 else
428 {
429 w = src->sSizeX;
430 h = src->sSizeY;
431 sx = 0;
432 sy = 0;
433 }
434
435 if (w / m < 8 || h / n < 8)
436 return AWP_BADARG;
437
438 ww = (AWPDOUBLE)w / (AWPDOUBLE)n;
439 hh = (AWPDOUBLE)h / (AWPDOUBLE)m;
440
441 if ((res = awpCreateImage(&hst, dst->sSizeX, 1, 1, AWP_DOUBLE)) != AWP_OK)
442 return res;
443 b = (AWPDOUBLE*)hst->pPixels;
444 d = (AWPDOUBLE*)dst->pPixels;
445 for (i = 0; i < m; i++)
446 {
447 r.top = (AWPWORD)(sy + i*hh + 0.5);
448 r.bottom = (AWPWORD)(r.top + hh + 0.5);
449 for (j = 0; j < n; j++)
450 {
451 r.left = (AWPWORD)(sx + j*ww + 0.5);
452 r.right = (AWPWORD)(r.left + ww + 0.5);
453 awpZeroImage(hst);
454 if ((res = awpLBPDescriptor(src, hst, &r, options, interpolation, norm)) != AWP_OK)
455 {
456 awpReleaseImage(&hst);
457 return res;
458 }
459
460 memcpy(d, b, dst->sSizeX*sizeof(AWPDOUBLE));
461 d += dst->sSizeX;
462 }
463 }
464
465 awpReleaseImage(&hst);
466 return res;
467}
