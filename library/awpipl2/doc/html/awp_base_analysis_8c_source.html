12/*
13we assume that AWP_BYTE images has a min_value = 0 and max_value = 255.
14step between bins defined as 256 / hst->sSizeX
15hst->sSizeX cannot be more than 256
16*/
17static AWPRESULT _awpGetHstByte(awpImage* pImage, awpImage** ppHst)
18{
19 AWPRESULT res = AWP_OK;
20 awpImage* pHst = *ppHst;
21 AWPWORD i = 0;
22 AWPBYTE c = 0;
23 AWPWORD idx = 0;
24 AWPWORD x = 0;
25 AWPWORD y = 0;
26
27 AWPDOUBLE* pix = NULL;
28 AWPBYTE* spix = NULL;
29
30 AWPDOUBLE hst_step = 256. / (AWPDOUBLE)pHst->sSizeX;
31
32 pix = _AWP_BPIX_(pHst, AWPDOUBLE);
33 spix = _AWP_BPIX_(pImage, AWPBYTE);
34
35 /*setup histogramm value axis*/
36 for (i = 0; i < pHst->sSizeX; i++)
37 {
38 for (c = 0; c < pHst->bChannels; c++)
39 {
40 pix[i*pHst->bChannels + c] = floor(i*hst_step + 0.5);
41 }
42 }
43 /*calc histogramm*/
44 for (y = 0; y < pImage->sSizeY; y++)
45 {
46 for (x = 0; x < pImage->sSizeX; x++)
47 {
48 for (c = 0; c < pImage->bChannels; c++)
49 {
50 idx = (AWPBYTE)floor(spix[y*pImage->bChannels*pImage->sSizeX + x*pImage->bChannels + c] + 0.5);
51 pix[pHst->sSizeX*pHst->bChannels + idx*pHst->bChannels + c]++;
52 }
53 }
54 }
55
56 return res;
57}
58#define _CALC_IDX_(v)\
59 idx = (AWPBYTE)(255*(v[y*pImage->bChannels*pImage->sSizeX + x*pImage->bChannels + c] - image_min[c]) /(image_max[c] - image_min[c]) );

62/*
63for other datatypes we use histogramm between min value on the channel data
64and max value. num bins defined as histogramm width
65*/
66static AWPRESULT _awpGetHst(awpImage* pImage, awpImage** ppHst)
67{
68 AWPRESULT res = AWP_OK;
69 awpImage* pHst = *ppHst;
70 AWPDOUBLE* image_min = NULL;
71 AWPDOUBLE* image_max = NULL;
72 AWPDOUBLE* hst_step = NULL;
73 AWPWORD i = 0;
74 AWPBYTE c = 0;
75 AWPWORD idx = 0;
76 AWPWORD x = 0;
77 AWPWORD y = 0;
78 AWPSHORT* spix = NULL;
79 AWPFLOAT* fpix = NULL;
80 AWPDOUBLE* dpix = NULL;
81
82 AWPDOUBLE* pix = _AWP_BPIX_(pHst, AWPDOUBLE);
83
84 /*range of signal variation*/
85 if (res = awpMinMax(pImage, &image_min, &image_max))
86 {
87 if (res != AWP_OK)
88 _ERR_EXIT_
89 }
90
91 /*setup width step for each channel*/
92 hst_step = (AWPDOUBLE*)malloc(pImage->bChannels*sizeof(AWPDOUBLE));
93 if (hst_step == NULL)
94 _ERROR_EXIT_RES_(AWP_NOTENOUGH_MEM)
95
96 for (i = 0; i < pImage->bChannels; i++)
97 hst_step[i] = (image_max[i] - image_min[i]) / (AWPDOUBLE)pHst->sSizeX;
98
99 /*setup histogramm value axis*/
100 for (i = 0; i < pHst->sSizeX; i++)
101 {
102 for (c = 0; c < pHst->bChannels; c++)
103 {
104 pix[i*pHst->bChannels + c] = image_min[c] + i*hst_step[c];// + 0.5);
105 }
106 }
107 /*calc histogramm*/
108 switch (pImage->dwType)
109 {
110 case AWP_SHORT:
111 spix = _AWP_BPIX_(pImage, AWPSHORT)
112 break;
113 case AWP_FLOAT:
114 fpix = _AWP_BPIX_(pImage, AWPFLOAT)
115 break;
116 case AWP_DOUBLE:
117 dpix = _AWP_BPIX_(pImage, AWPDOUBLE)
118 break;
119 }
120 for (y = 0; y < pImage->sSizeY; y++)
121 {
122 for (x = 0; x < pImage->sSizeX; x++)
123 {
124 for (c = 0; c < pImage->bChannels; c++)
125 {
126 switch (pImage->dwType)
127 {
128 case AWP_SHORT:
129 _CALC_IDX_(spix)
130 break;
131 case AWP_FLOAT:
132 _CALC_IDX_(fpix)
133 break;
134 case AWP_DOUBLE:
135 _CALC_IDX_(dpix)
136 break;
137 }
138 pix[pHst->sSizeX*pHst->bChannels + idx*pHst->bChannels + c]++;
139 }
140 }
141 }
142
143CLEANUP:
144 if (image_min != NULL)
145 free(image_min);
146 if (image_max != NULL)
147 free(image_max);
148 if (hst_step != NULL)
149 free(hst_step);
150 return res;
151}

153obtain image histogramm function
154image histogramm is stored in the awpImage with height = 2
155and has num channels as in the source image pImage
156first row of the histogramm image represents value of brightess
157second row - number of pixels of this brightess
158for AWP_BYTE images num bins in the histogramm should be less than
159256 and more then 1.
160for other image types num bins should be more 1.

173 _ERROR_EXIT_RES_(AWP_BADMEMORY)

177 /*check histogramm image*/

216/*
217find the k-th moment of the distribution pHst ans store them to the pRes
218*/
219static AWPRESULT __awpHstMomemt(awpImage* pHst, awpImage* pMoment, AWPDOUBLE k)
220{
221 AWPRESULT res = AWP_OK;
222 AWPWORD x = 0;
223 AWPBYTE c = 0;
224 AWPDOUBLE* hst = NULL;
225 AWPDOUBLE* Moment = NULL;
226 AWPDOUBLE* sum = NULL;
227
228 if (pHst == NULL || pMoment == NULL || k < 1 || k>3)
229 _ERROR_EXIT_RES_(AWP_BADARG)
230
231 _AWP_BASE_ANALYSIS_CHECK_HST_(pHst);
232 _AWP_BASE_ANALYSIS_CHECK_RESULT_(pMoment);
233
234 Moment = (AWPDOUBLE*)pMoment->pPixels;
235 hst = (AWPDOUBLE*)pHst->pPixels;
236 sum = (AWPDOUBLE*)malloc(pHst->bChannels*sizeof(AWPDOUBLE));
237 memset(sum, 0, pHst->bChannels*sizeof(AWPDOUBLE));
238
239 for (x = 0; x < pHst->sSizeX; x++)
240 {
241 for (c = 0; c < pHst->bChannels; c++)
242 {
243 Moment[c] += pow(hst[pHst->bChannels*x + c], k)*hst[pHst->bChannels*pHst->sSizeX + x*pHst->bChannels + c];
244 sum[c] += hst[pHst->bChannels*pHst->sSizeX + x*pHst->bChannels + c];
245 }
246 }
247
248 for (c = 0; c < pHst->bChannels; c++)
249 {
250 Moment[c] /= sum[c];
251 }
252
253 free(sum);
254CLEANUP:
255 return res;
256}

280/*
281find the standard devation over histogramm
282*/
283AWPRESULT awpGetHstStdDev(awpImage* pHst, awpImage* pStdDev)
284{
285 AWPRESULT res = AWP_OK;
286 awpImage* mean = NULL;
287 AWPDOUBLE* m1 = NULL;
288 AWPDOUBLE* m2 = NULL;
289 AWPBYTE c = 0;
290
291 if (pHst == NULL || pStdDev == NULL)
292 _ERROR_EXIT_RES_(AWP_BADARG)
293
294 _AWP_BASE_ANALYSIS_CHECK_HST_(pHst);
295 _AWP_BASE_ANALYSIS_CHECK_RESULT_(pStdDev);
296
297 if (pHst->bChannels != pStdDev->bChannels)
298 _ERROR_EXIT_RES_(AWP_BADARG)
299
300 if (res = awpCreateImage(&mean, 1, 1, pStdDev->bChannels, AWP_DOUBLE) != AWP_OK)
301 _ERR_EXIT_
302
303 if (res = __awpHstMomemt(pHst, mean, 1) != AWP_OK)
304 _ERR_EXIT_
305 if (res = __awpHstMomemt(pHst, pStdDev, 2) != AWP_OK)
306 _ERR_EXIT_
307
308 m1 = _AWP_BPIX_(mean, AWPDOUBLE)
309 m2 = _AWP_BPIX_(pStdDev, AWPDOUBLE)
310
311 for (c = 0; c < pStdDev->bChannels; c++)
312 {
313 if (m2[c] - m1[c]*m1[c] < 1e-30)
314 {
315 m2[c] = 1e-30;
316 }
317 else
318 m2[c] = sqrt(m2[c] - m1[c]*m1[c]);
319 }
320
321CLEANUP:
322 _SAFE_RELEASE_(mean)
323 return res;
324}

376/*
377find the median values over histogramm
378*/
379AWPRESULT awpGetMedian(awpImage* pImage, awpImage* pMedian)
380{
381 AWPRESULT res = AWP_OK;
382 awpImage* channel = NULL;
383 AWPBYTE c = 0;
384 AWPDOUBLE* pix = NULL;
385 AWPFLOAT * fpix = NULL;
386 AWPBYTE * bpix = NULL;
387 AWPSHORT* spix = NULL;
388 AWPDOUBLE* median = NULL;
389 if (pImage == NULL || pMedian == NULL)
390 _ERROR_EXIT_RES_(AWP_BADARG)
391
392 _AWP_BASE_ANALYSIS_CHECK_RESULT_(pMedian);
393
394 median = _AWP_BPIX_(pMedian, AWPDOUBLE);
395
396 for (c = 0; c < pMedian->bChannels; c++)
397 {
398 if (res =awpGetChannel(pImage, &channel, c) != AWP_OK)
399 _ERR_EXIT_
400 if (pImage->dwType == AWP_DOUBLE)
401 {
402 pix = _AWP_BPIX_(channel, AWPDOUBLE);
403
404
405 qsort((void*)pix, pImage->sSizeX*pImage->sSizeY, sizeof(AWPDOUBLE), _awpDCompare);
406 median[c] = pix[channel->sSizeX*pImage->sSizeY/2];
407 }
408 else if (pImage->dwType == AWP_BYTE)
409 {
410 bpix = _AWP_BPIX_(channel, AWPBYTE);
411
412
413 qsort((void*)bpix, pImage->sSizeX*pImage->sSizeY, sizeof(AWPBYTE), _awpBCompare);
414 median[c] = bpix[channel->sSizeX*pImage->sSizeY / 2];
415 }
416 else if (pImage->dwType ==AWP_FLOAT)
417 {
418 fpix = _AWP_BPIX_(channel, AWPFLOAT);
419
420
421 qsort((void*)fpix, pImage->sSizeX*pImage->sSizeY, sizeof(AWPFLOAT), _awpFCompare);
422 median[c] = fpix[channel->sSizeX*pImage->sSizeY / 2];
423 }
424 else if (pImage->dwType == AWP_SHORT)
425 {
426 spix = _AWP_BPIX_(channel, AWPSHORT);
427
428
429 qsort((void*)spix, pImage->sSizeX*pImage->sSizeY, sizeof(AWPSHORT), _awpSCompare);
430 median[c] = spix[channel->sSizeX*pImage->sSizeY / 2];
431 }
432 _SAFE_RELEASE_(channel)
433 }
434CLEANUP:
435 _SAFE_RELEASE_(channel)
436 return res;
437}

446*/
447AWPRESULT awpGetHstEntropy(awpImage* pHst, awpImage* pEntropy)
448{
449 AWPRESULT res = AWP_OK;
450 awpImage* channel = NULL;
451 AWPBYTE c = 0;
452 AWPWORD x = 0;
453 AWPDOUBLE* pix = NULL;
454 AWPDOUBLE* entropy = NULL;
455 AWPDOUBLE* sum = NULL;
456 if (pHst == NULL || pEntropy == NULL)
457 _ERROR_EXIT_RES_(AWP_BADARG)
458
459 _AWP_BASE_ANALYSIS_CHECK_HST_(pHst);
460 _AWP_BASE_ANALYSIS_CHECK_RESULT_(pEntropy);
461
462 entropy = _AWP_BPIX_(pEntropy, AWPDOUBLE);
463 pix = _AWP_BPIX_(pHst, AWPDOUBLE);
464 sum = (AWPDOUBLE*)malloc(pHst->bChannels*sizeof(AWPDOUBLE));
465 memset(sum, 0, pHst->bChannels*sizeof(AWPDOUBLE));
466 pix += pHst->sSizeX*pHst->bChannels;
467 for (x = 0; x < pHst->sSizeX; x++)
468 {
469 for (c = 0; c < pHst->bChannels; c++)
470 {
471 sum[c] += pix[x*pHst->bChannels + c];
472 }
473 }
474
475 for (x = 0; x < pHst->sSizeX; x++)
476 {
477 for (c = 0; c < pHst->bChannels; c++)
478 {
479 entropy[c] += pix[x*pHst->bChannels + c] != 0 ? -pix[x*pHst->bChannels + c] / sum[c] * awpLog2(pix[x*pHst->bChannels + c] / sum[c]) : 0;
480 }
481 }
482CLEANUP:
483 return res;
484}

486AWPRESULT awpGet2DHistogramm(awpImage* pImage, awpImage* p2DHist, AWPBYTE low, AWPBYTE up, AWPINT needToConvert)
487{
488 AWPINT i, j;
489 AWPDOUBLE* dblPix;
490 awpColor* clPix;
491 AWPBYTE h,s, v;
492 AWPRESULT res = AWP_OK;
493 awpImage* pDblHst = NULL;
494 awpImage* p2DHisto = NULL;
495 // checking arguments
496 if (pImage == NULL || p2DHist == NULL)
497 return AWP_BADARG;
498 if (pImage->dwType != AWP_BYTE)
499 return AWP_BADARG;
500 if (pImage->bChannels != 3)
501 return AWP_BADARG;
502
503 if (p2DHist->dwType != AWP_BYTE && p2DHist->dwType != AWP_DOUBLE )
504 return AWP_BADARG;
505 if (p2DHist->bChannels != 1)
506 return AWP_BADARG;
507
508 // to build a two-dimensional histogram we use the 1st and 2nd channels of the image
509 // pImage and write the histogram into a temporary image like AWPDOUBLE
510 res = awpCreateImage(&pDblHst, 256, 256, 1, AWP_DOUBLE);
511 if (res != AWP_OK)
512 return res;
513 dblPix = (AWPDOUBLE*)pDblHst->pPixels;
514 clPix = (awpColor*)pImage->pPixels;
515 for (i = 0; i < pImage->sSizeY; i++)
516 {
517 for (j = 0; j < pImage->sSizeX; j++)
518 {
519 v = clPix[i*pImage->sSizeX + j].bBlue;
520 if (v>= low && v <= up)
521 {
522 h = clPix[i*pImage->sSizeX + j].bRed;
523 s = clPix[i*pImage->sSizeX + j].bGreen;
524 dblPix[s*256 + h]++;
525 }
526 }
527 }
528 awpCopyImage(p2DHist, &p2DHisto);
529 // convert a temporary image of type AWPDOUBLE to AWPBYTE
530 if ( needToConvert == TRUE )
531 {
532 if ((res = awpConvert(pDblHst, AWP_CONVERT_TO_BYTE_WITH_NORM)) != AWP_OK)
533 return res;
534 }
535 if((res = awpFastCopyImage(pDblHst, p2DHist)) != AWP_OK)
536 return res;
537 if ((res = awpReleaseImage(&pDblHst)) != AWP_OK)
538 return res;
539 if ((res = awpFilter(p2DHist, p2DHisto, AWP_BLUR)) != AWP_OK)
540 return res;
541 return AWP_OK;
542}
543/*Histogramm equalize function
544The function of the histogram equilibration. Brings an image histogram
545pImage, to the form as close as possible to the uniform.
546Source image should has AWP_BYTE datatype
547*/
548AWPRESULT awpHistogrammEqualize(awpImage* pImage)
549{
550 AWPRESULT res = AWP_OK;
551 awpImage* pHst = NULL;
552 awpImage* pFunc = NULL;
553 AWPDOUBLE* SumOfEntries = NULL;
554 AWPINT count = 0;
555 AWPINT i = 0;
556 AWPINT j = 0;
557 AWPBYTE c = 0;
558 AWPBYTE* p = NULL;
559
560 AWPDOUBLE* Hst;
561 AWPDOUBLE* PFunc;
562
563 /*check source image*/
564 if (res = awpCheckImage(pImage) != AWP_OK)
565 _ERR_EXIT_
566 if (pImage->dwType != AWP_BYTE)
567 _ERROR_EXIT_RES_(AWP_BADARG)
568 /*todo: use Options flag to setup output histogramm*/
569 if (res = awpGetHst(pImage, &pHst, 0) != AWP_OK)
570 _ERR_EXIT_
571 if (res = awpCreateImage(&pFunc, pHst->sSizeX, 1, pHst->bChannels, AWP_DOUBLE) != AWP_OK)
572 _ERR_EXIT_
573 count = pImage->sSizeX*pImage->sSizeY;
574 SumOfEntries = (AWPDOUBLE*)malloc(pImage->bChannels*sizeof(AWPDOUBLE));
575 Hst = _AWP_BPIX_(pHst, AWPDOUBLE);
576 Hst += pHst->sSizeX*pHst->bChannels;
577 PFunc = _AWP_BPIX_(pFunc, AWPDOUBLE);
578 for (i = 0; i < pHst->sSizeX; i++)
579 {
580 for (c = 0; c < pImage->bChannels; c++)
581 {
582 SumOfEntries[c] = 0;
583 for (j = 0; j <= i; j++)
584 SumOfEntries[c] += Hst[j*pHst->bChannels + c];
585 PFunc[i*pHst->bChannels + c] = SumOfEntries[c] / count;
586 }
587
588 }
589 p = _AWP_BPIX_(pImage, AWPBYTE);
590 for (i = 0; i < count; i++)
591 {
592 for (c = 0; c < pImage->bChannels; c++)
593 {
594 p[i*pImage->bChannels + c] = (AWPBYTE)(255 * PFunc[p[i*pImage->bChannels + c] * pImage->bChannels + c]);
595 }
596 }
597
598CLEANUP:
599 _SAFE_RELEASE_(pHst)
600 _SAFE_RELEASE_(pFunc)
601 _SAFE_FREE_(SumOfEntries)
602 return res;
603}

606AWPRESULT awpGetCentroid(const awpImage* pImg, awpPoint* p)
607{
608 AWPRESULT res;
609 AWPDOUBLE sum, x0, y0;
610 AWPBYTE* pix;
611 AWPINT x, y;
612 AWPDWORD i;
613
614 res = AWP_OK;
615 _CHECK_RESULT_((res = awpCheckImage(pImg)))
616
617 if (pImg->dwType != AWP_BYTE)
618 {
619 res = AWP_NOTSUPPORT;
620 _ERR_EXIT_
621 }
622
623 if (pImg->bChannels != 1)
624 {
625 res = AWP_NOTSUPPORT;
626 _ERR_EXIT_
627 }
628 x0 = 0;
629 y0 = 0;
630 sum = 0;
631 pix = (AWPBYTE*)pImg->pPixels;
632 for (i = 0; i < pImg->sSizeY; i++)
633 {
634 y = i;
635 for (x = 0; x <= pImg->sSizeX; x++)
636 {
637 sum += pix[y*pImg->sSizeX + x];
638 x0 += x*pix[y*pImg->sSizeX + x];
639 y0 += y*pix[y*pImg->sSizeX + x];
640 }
641 }
642
643 p->X = (AWPSHORT)floor(x0 / (double)sum + 0.5);
644 p->Y = (AWPSHORT)floor(y0 / (double)sum + 0.5);
645CLEANUP:
646 return res;
647}

649AWPRESULT awpGetOrientation(const awpImage* pImg, AWPDOUBLE* teta, AWPDOUBLE* mi, AWPDOUBLE* ma)
650{
651 /*local variables*/
652 AWPRESULT res;
653 AWPINT x, y, j;
654 AWPDOUBLE mxx, myy, mxy, intes, s; /*cental moments*/
655 AWPBYTE* pixels;
656 awpPoint center;
657 AWPDWORD i;
658 /*init out arguments */
659 *teta = 0;
660 *mi = 0;
661 *ma = 0;
662 res = AWP_OK;
663 /*check the arguments*/
664 _CHECK_RESULT_((res = awpCheckImage(pImg)))
665 if (teta == NULL || mi == NULL || ma == NULL)
666 {
667 res = AWP_BADARG;
668 _ERR_EXIT_
669 }
670 /*check the capabilities*/
671 if (pImg->bChannels > 1 || pImg->dwType != AWP_BYTE)
672 {
673 res = AWP_BADARG;
674 _ERR_EXIT_
675 }
676 /*init local variables*/
677 mxx = 0; myy = 0; mxy = 0;
678 if (awpGetCentroid(pImg, &center) != AWP_OK)
679 {
680 res = AWP_BADARG;
681 _ERR_EXIT_
682 }
683
684 /*find the moment*/
685 intes = 0;
686 s = 0;
687 pixels = (AWPBYTE*)pImg->pPixels;
688 for (i = 0; i < pImg->sSizeY; i++)
689 {
690 for (j = 0; j <pImg->sSizeX; j++)
691 {
692 x = j;
693 y = i;
694 intes += pixels[y*pImg->sSizeX + x];
695 if (pixels[y*pImg->sSizeX + x] > 0)
696 {
697 mxx += (x - center.X)*(x - center.X);
698 myy += (y - center.Y)*(y - center.Y);
699 mxy += (x - center.X)*(y - center.Y);
700 s++;
701 }
702 }
703 }
704 if (intes == 0)
705 {
706
707 res = AWP_BADARG;
708 _ERR_EXIT_
709 }
710 intes = s;
711 mxx /= intes;
712 myy /= intes;
713 mxy /= intes;
714
715 /*find the moment*/
716 if (mxy == 0 && myy <= mxx)
717 {
718 *teta = -AWP_PI / 2;
719 *ma = 4 * sqrt(mxx);
720 *mi = 4 * sqrt(myy);
721 }
722 else if (mxy == 0 && myy > mxx)
723 {
724 *teta = 0;
725 *ma = 4 * sqrt(myy);
726 *mi = 4 * sqrt(mxx);
727 }
728 else if (myy <= mxx)
729 {
730 *teta = atan(-2 * mxy*myy - mxx + sqrt((myy - mxx)*(myy - mxx) + 4 * mxy*mxy));
731 *ma = sqrt(8 * (myy + mxx + sqrt((myy - mxx)*(myy - mxx) + 4 * mxy*mxy)));
732 *mi = sqrt(8 * (myy + mxx - sqrt((myy - mxx)*(myy - mxx) + 4 * mxy*mxy)));
733
734 }
735 else if (myy > mxx)
736 {
737 *teta = atan(sqrt(mxx + myy + sqrt((mxx - myy)*(mxx - myy) + 4 * mxy*mxy)) / (-2 * mxy + 0.000001));
738 *ma = sqrt(8 * (myy + mxx + sqrt((myy - mxx)*(myy - mxx) + 4 * mxy*mxy)));
739 *mi = sqrt(8 * (myy + mxx - sqrt((myy - mxx)*(myy - mxx) + 4 * mxy*mxy)));
740 }
741
742 *teta = 180.0*(*teta) / AWP_PI;
743CLEANUP:
744 return res;
