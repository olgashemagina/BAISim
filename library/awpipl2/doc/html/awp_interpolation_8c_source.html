145AWPRESULT awpBilinearInterpolation (awpImage* src, awp2DPoint* p, awpImage* dst)
146{
147 // internal variables
148 awp2DPoint pp[6];
149 awpImage* I4 = NULL;
150 awpImage* I5 = NULL;
151
152 // check the validity of the arguments.
153 if (src == NULL || p == NULL || dst == NULL)
154 return AWP_BADARG;
155 // the intensity value in the interpolated point p is always of the type
156 // AWPDOUBLE, so the argument dst must be of type AWPDOUBLE
157 if (dst->dwType != AWP_DOUBLE)
158 return AWP_BADARG;
159 // the number of channels of the original image should be equal to the width
160 // image - result.
161 if (src->bChannels != dst->sSizeX)
162 return AWP_BADARG;
163 // the number of channels of the resulting image must be 1
164 if (dst->bChannels != 1)
165 return AWP_BADARG;
166 // точка должна лежать внутри изображения целиком.
167 if (p->X < 1 || p->X > src->sSizeX -1 || p->Y < 1 || p->Y > src->sSizeY)
168 return AWP_BADARG;
169
170 memset(pp, 0, sizeof(pp));
171
172 // find the coordinates of the surrounding points, including auxiliary
173 pp[0].X = floor(p->X);
174 pp[0].Y = floor(p->Y);
175
176 pp[1].X = ceil(p->X);
177 pp[1].Y = floor(p->Y);
178
179 pp[2].X = floor(p->X);
180 pp[2].Y = ceil(p->Y);
181
182 pp[3].X = ceil(p->X);
183 pp[3].Y = ceil(p->Y);
184
185 pp[4].X = p->X;
186 pp[4].Y = floor(p->Y);
187
188 pp[5].X = p->X;
189 pp[5].Y = ceil(p->Y);
190
191 if( floor(p->X) == ceil(p->X) && floor(p->Y) == ceil(p->Y) )
192 {
193 _awpWithoutInterpoalation(src, *p, dst);
194 return AWP_OK;
195 }
196
197 if( floor(p->X) == ceil(p->X) )
198 _awpSimpleYInterpoalation(src, pp[2], pp[0], *p, dst);
199 else if( floor(p->Y) == ceil(p->Y) )
200 _awpXInterpoalation(src, pp[3], pp[2], *p, dst);
201 else
202 {
203 // create buffers to store intermediate results
204 awpCreateImage(&I4, src->bChannels, 1, 1, AWP_DOUBLE);
205 awpCreateImage(&I5, src->bChannels, 1, 1, AWP_DOUBLE);
206 // perform linear interpolation for points pp [5] pp [4]
207 _awpXInterpoalation(src, pp[1], pp[0], pp[4], I4);
208 _awpXInterpoalation(src, pp[3], pp[2], pp[5], I5);
209 // execute linear interpolation for the point p
210 _awpYInterpoalation(I5, pp[5], I4, pp[4], *p, dst);
211 awpReleaseImage(&I4);
212 awpReleaseImage(&I5);
213 }
214 return AWP_OK;
215}
