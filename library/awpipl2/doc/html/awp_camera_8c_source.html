55AWPRESULT awpImageYToLenght(awpCamera* pCamera, awpImage* pImage, AWPWORD Y, double* Value)
56{
57 AWPDOUBLE delta, alfa;
58
59 if (pCamera == NULL)
60 return AWP_BADARG;
61 if (pImage == NULL)
62 return AWP_BADARG;
63 if (Value == NULL)
64 return AWP_BADARG;
65
66
67 if (Y <= 0 || Y > pImage->sSizeY)
68 return AWP_BADARG;
69
70
71 delta = pCamera->dChipHSize*((pImage->sSizeY -Y)/pImage->sSizeY - 0.5);
72 alfa = AWP_PI*pCamera->dAngle/180.;
73 *Value = pCamera->dHeight * tan(atan(delta/pCamera->dFocalLength) + alfa);
74
75 return AWP_OK;
76}

113 {
114 *Value = -1;

117
118
119 Y = 0;

254AWPRESULT awpImageObjectHeight(awpCamera* pCamera, awpImage* pImage, awpRect* pObject, double* Value)
255{
256 AWPDOUBLE Z, delta, gamma, tan_alfa;
257
258 if (pCamera == NULL)
259 return AWP_BADARG;
260 if (pImage == NULL)
261 return AWP_BADARG;
262 if (Value == NULL)
263 return AWP_BADARG;
264 if (pObject == NULL)
265 return AWP_BADARG;
266
267 if (awpRectInImage(pImage, pObject) != AWP_OK)
268 return AWP_BADARG;
269
270 Z = 0;
271 if (awpImageYToLenght(pCamera, pImage, pObject->bottom, &Z) != AWP_OK)
272 return AWP_BADARG;
273
274 delta = pCamera->dChipHSize*((pImage->sSizeY - pObject->top )/pImage->sSizeY - 0.5);
275 gamma = atan(delta / pCamera->dFocalLength);
276 tan_alfa = tan(gamma + AWP_PI*pCamera->dAngle/180.);
277 *Value = pCamera->dHeight - Z / tan_alfa;
278
279 return AWP_OK;
280}

282AWPRESULT awpSceneToImagePoint(awpCamera* pCamera, awpImage* pImage, awp3DPoint* pSrc, awp2DPoint* pDst)
283{
284 AWPDOUBLE gamma;
285
286 if (pCamera == NULL)
287 return AWP_BADARG;
288 if (pImage == NULL)
289 return AWP_BADARG;
290 if (pSrc == NULL)
291 return AWP_BADARG;
292 if (pDst == NULL)
293 return AWP_BADARG;
294
295 pDst->X = pImage->sSizeX*(pSrc->X*pCamera->dFocalLength / (pCamera->dChipWSize*pSrc->Z) + 0.5);
296 gamma = atan(pSrc->Z /(pCamera->dHeight - pSrc->Y));
297 pDst->Y = pImage->sSizeY*(pCamera->dFocalLength*tan(AWP_PI*pCamera->dAngle /180 - gamma)/pCamera->dChipHSize + 0.5);
298
299

303AWPRESULT awpImageToScenePoint(awpCamera* pCamera, awpImage* pImage, double Height, awp2DPoint* pSrc, awp3DPoint* pDst)
304{
305 AWPDOUBLE X, Y, Z;
306 awpPoint p;
307 awpRect r;
308
309 if (pCamera == NULL)
310 return AWP_BADARG;
311 if (pImage == NULL)
312 return AWP_BADARG;
313 if (pSrc == NULL)
314 return AWP_BADARG;
315 if (pDst == NULL)
316 return AWP_BADARG;
317
318 if (awpImageYToLenght(pCamera, pImage,(AWPWORD)pSrc->Y, &Z) != AWP_OK)
319 return AWP_BADARG;
320 p.X = (AWPSHORT)pSrc->X;
321 p.Y = (AWPSHORT)pSrc->Y;
322 if (awpImagePointToShift(pCamera, pImage, &p, &X) != AWP_OK)
323 return AWP_BADARG;
324
325 r.left = (AWPSHORT)pSrc->X;
326 r.right = r.left + 1;
327 r.top = (AWPSHORT)pSrc->Y;
328 r.bottom = (AWPSHORT)(r.top + Height);
329
330 if (awpImageObjectHeight(pCamera, pImage, &r, &Y) != AWP_OK)
331 return AWP_BADARG;
332
333 pDst->X = X;
334 pDst->Y = Y;
335 pDst->Z = Z;
336
337 return AWP_OK;
