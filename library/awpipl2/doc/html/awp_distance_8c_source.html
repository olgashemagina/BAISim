81#define __AWP_NCC__(u) \
82{\
83 AWPDOUBLE corr = 0;\
84 AWPDOUBLE norm1 = 0;\
85 AWPDOUBLE norm2 = 0;\
86 AWPDOUBLE avg1 = 0;\
87 AWPDOUBLE avg2 = 0; \
88 u* f3 = (u*)src1->pPixels;\

94 avg2 += f4[i]; \
95 } \
96 avg1 /= size; \
97 avg2 /= size; \
98 for( i = 0; i < size; i++)\
99 { \
100 corr += (f3[i] - avg1) * (f4[i]-avg2);\
101 norm1 += (f3[i]-avg1) * (f3[i] - avg1);\
102 norm2 += (f4[i]-avg2) * (f4[i] - avg2);\
103 } \
104 *res= corr / sqrt(norm1*norm2);\
105}

165/*
166Calculates the distance between two images. The result is written to the variable res.
167Images should have one data channel of the same size and
168be of the same type.
169*/
170AWPRESULT awpDistance(awpImage* src1, awpImage* src2, AWPBYTE options, AWPDOUBLE* res)
171{
172 AWPRESULT result = AWP_OK;
173
174 if (src1 == NULL || src2 == NULL || res == NULL)
175 return AWP_BADARG;
176 if (src1->sSizeX != src2->sSizeX)
177 return AWP_BADARG;
178 if (src1->sSizeY != src2->sSizeY)
179 return AWP_BADARG;
180 if (src1->dwType != src2->dwType)
181 return AWP_BADARG;
182 if (src1->bChannels != 1)
183 return AWP_BADARG;
184 if (src1->bChannels != src2->bChannels)
185 return AWP_BADARG;
186
187 *res = 0;
188
189 switch(options)
190 {
191 case AWP_DIST_L1:
192 result = awpL1Distance(src1, src2, res);
193 break;
194
195 case AWP_DIST_L2:
196 result = awpL2Distance(src1, src2, res);
197 break;
198
199 case AWP_DIST_NCC:
200 result = awpNCCDistance(src1, src2, res);
201 break;
202 case AWP_DIST_X2:
203 result = awpX2Distance(src1, src2, res);
204 break;
205 case AWP_DIST_MSE:
206 result = awpL2Distance(src1, src2, res);
207 _CHECK_RESULT_(result)
208 *res = *res / (AWPDOUBLE)(src1->sSizeX*src1->sSizeY);
209 break;
210
211 case AWP_DIST_RMSE:
212 result = awpL2Distance(src1, src2, res);
213 _CHECK_RESULT_(result)
214 *res = *res / (AWPDOUBLE)(src1->sSizeX*src1->sSizeY);
215 *res = sqrt(*res);
216 break;
217 default:
218 result = AWP_BADARG;
219 }
220
221CLEANUP:
222 return result;
223}

225/*
226calculates the distance between two images divided by
227n fragments. Each fragment of the image is represented by a row of pixels in
228variables src1 and src2
229weight is the weight of each fragment. If weight == NULL, it is not considered
230*/
231AWPRESULT awpGridDistance(awpImage* src1, awpImage* src2, awpImage* weight, AWPBYTE options, AWPDOUBLE* res)
232{
233 AWPRESULT result = AWP_OK;
234
235 awpImage* img1 = NULL;
236 awpImage* img2 = NULL;
237
238 AWPINT i = 0;
239 AWPINT linesize = 0;
240 AWPBYTE* b1 = NULL;
241 AWPBYTE* b2 = NULL;
242 AWPBYTE* b3 = NULL;
243 AWPBYTE* b4 = NULL;
244 AWPDOUBLE* w = NULL;
245
246 AWPDOUBLE value = 0;
247
248 if (src1 == NULL || src2 == NULL || res == NULL)
249 return AWP_BADARG;
250 if (src1->sSizeX != src2->sSizeX)
251 return AWP_BADARG;
252 if (src1->sSizeY != src2->sSizeY)
253 return AWP_BADARG;
254 if (src1->dwType != src2->dwType)
255 return AWP_BADARG;
256 if (src1->bChannels != 1)
257 return AWP_BADARG;
258 if (src1->bChannels != src2->bChannels)
259 return AWP_BADARG;
260 if (weight != NULL && weight->sSizeX != src1->sSizeY)
261 return AWP_BADARG;
262
263
264 if (awpCreateImage(&img1, src1->sSizeX, 1, 1, src1->dwType) != AWP_OK)
265 return AWP_BADMEMORY;
266 if (awpCreateImage(&img2, src1->sSizeX, 1, 1, src1->dwType) != AWP_OK)
267 {
268 awpReleaseImage(&img1);
269 return AWP_BADMEMORY;
270 }
271
272 switch(src1->dwType)
273 {
274 case AWP_BYTE:
275 linesize = src1->sSizeX*sizeof(AWPBYTE);
276 break;
277 case AWP_SHORT:
278 linesize = src1->sSizeX*sizeof(AWPSHORT);
279 break;
280 case AWP_FLOAT:
281 linesize = src1->sSizeX*sizeof(AWPFLOAT);
282 break;
283 case AWP_DOUBLE:
284 linesize = src1->sSizeX*sizeof(AWPDOUBLE);
285 break;
286 default:
287 {
288 awpReleaseImage(&img1);
289 awpReleaseImage(&img2);
290 return AWP_BADARG;
291 }
292 }
293
294 b1 = (AWPBYTE*)src1->pPixels;
295 b2 = (AWPBYTE*)src2->pPixels;
296 b3 = (AWPBYTE*)img1->pPixels;
297 b4 = (AWPBYTE*)img2->pPixels;
298 if (weight != NULL)
299 w = (AWPDOUBLE*)weight->pPixels;
300 *res = 0;
301
302 for (i = 0; i < src1->sSizeY; i++)
303 {
304 memcpy(b3,b1,linesize);
305 memcpy(b4,b2,linesize);
306 b1+= linesize;
307 b2+= linesize;
308
309 if ((result = awpDistance(img1, img2, options, &value)) != AWP_OK)
310 {
311 awpReleaseImage(&img1);
312 awpReleaseImage(&img2);
313 return result;
314 }
315
316 *res += w !=NULL ? w[i]*value : value;
317 }
318 awpReleaseImage(&img1);
319 awpReleaseImage(&img2);
320 return result;
321}
