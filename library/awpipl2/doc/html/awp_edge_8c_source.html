5AWPRESULT awpLocalMax(awpImage* pImage, awpImage** ppImage, AWPBYTE thr, AWPINT options)
6{
7 AWPRESULT res = AWP_OK;
8 awpImage* pTempImage = NULL;
9 AWPBYTE *pSrcPix, *pDstPix;
10 AWPWORD i, j;
11 AWPBYTE curcolor;
12
13 _CHECK_RESULT_((res = awpCheckImage(pImage)))
14 if ((pImage->bChannels != 1) || (pImage->dwType != AWP_BYTE))
15 _ERROR_EXIT_RES_(AWP_NOTSUPPORT)
16
17 _CHECK_RESULT_((res = awpCreateImage(&pTempImage, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, AWP_BYTE)))
18
19 pSrcPix = (AWPBYTE *) pImage->pPixels;
20 pDstPix = (AWPBYTE *) pTempImage->pPixels;
21
22 switch (options)
23 {
24 case _4_NEIGHBOURS:
25 for(j = 0; j < pImage->sSizeY; j++)
26 for(i = 0; i < pImage->sSizeX; i++)
27 {
28 curcolor = pSrcPix[(j * pImage->sSizeX) + i];
29
30 if ((curcolor < thr) ||
31 ((i > 0) && (pSrcPix[(j * pImage->sSizeX) + i - 1] > curcolor)) ||
32 ((i < pImage->sSizeX - 1) && (pSrcPix[(j * pImage->sSizeX) + i + 1] > curcolor)) ||
33 ((j > 0) && (pSrcPix[((j - 1) * pImage->sSizeX) + i] > curcolor)) ||
34 ((j < pImage->sSizeY - 1) && (pSrcPix[((j + 1) * pImage->sSizeX) + i] > curcolor)))
35 continue;
36
37 pDstPix[(j * pTempImage->sSizeX) + i] = curcolor;
38 }
39 break;
40 case _8_NEIGHBOURS:
41 for(j = 0; j < pImage->sSizeY; j++)
42 for(i = 0; i < pImage->sSizeX; i++)
43 {
44 curcolor = pSrcPix[(j * pImage->sSizeX) + i];
45
46 if ((curcolor < thr) ||
47 ((i > 0) && (j > 0) && (pSrcPix[((j - 1) * pImage->sSizeX) + i - 1] > curcolor)) ||
48 ((i > 0) && (pSrcPix[(j * pImage->sSizeX) + i - 1] > curcolor)) ||
49 ((i > 0) && (j < pImage->sSizeY - 1) && (pSrcPix[((j + 1) * pImage->sSizeX) + i - 1] > curcolor)) ||
50 ((j < pImage->sSizeY - 1) && (pSrcPix[((j + 1) * pImage->sSizeX) + i] > curcolor)) ||
51 ((i < pImage->sSizeX - 1) && (j < pImage->sSizeY - 1) && (pSrcPix[((j + 1) * pImage->sSizeX) + i + 1] > curcolor)) ||
52 ((i < pImage->sSizeX - 1) && (pSrcPix[(j * pImage->sSizeX) + i + 1] > curcolor)) ||
53 ((i < pImage->sSizeX - 1) && (j > 0) && (pSrcPix[((j - 1) * pImage->sSizeX) + i + 1] > curcolor)) ||
54 ((j > 0) && (pSrcPix[((j - 1) * pImage->sSizeX) + i] > curcolor)))
55 continue;
56
57 pDstPix[(j * pTempImage->sSizeX) + i] = curcolor;
58 }
59 break;
60 default:
61 res = AWP_BADARG;
62 _CHECK_RESULT_(awpReleaseImage(&pTempImage))
63 _ERR_EXIT_
64 }

142AWPRESULT awpFeatures(awpImage* pImage, awpImage** ppImage, AWPBYTE thr_max, AWPBYTE thr_min, AWPBYTE relative_thr, AWPINT radius, AWPINT MinMax_options, AWPINT options)
143{
144 AWPRESULT res = AWP_OK;
145 awpImage *pImageMax = NULL, *pImageMin = NULL, *pTempImage = NULL;
146 AWPBYTE *pSrcMaxPix, *pSrcMinPix, *pDstPix;
147 AWPWORD i, j;
148 AWPSHORT local_i, local_j;
149 AWPINT cur_x, cur_y;
150 AWPBYTE cur_max_color, cur_min_color, min_min_color;
151
152 if (radius == 0)
153 _ERROR_EXIT_RES_(AWP_BADARG)
154 _CHECK_RESULT_((res = awpCheckImage(pImage)))
155
156 _CHECK_RESULT_((res = awpLocalMax(pImage, &pImageMax, thr_max, MinMax_options)))
157 _CHECK_RESULT_((res = awpLocalMin(pImage, &pImageMin, thr_min, MinMax_options)))
158
159 _CHECK_RESULT_((res = awpCreateImage(&pTempImage, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, AWP_BYTE)))
160
161 pSrcMaxPix = (AWPBYTE *) pImageMax->pPixels;
162 pSrcMinPix = (AWPBYTE *) pImageMin->pPixels;
163 pDstPix = (AWPBYTE *) pTempImage->pPixels;
164
165 for(j = 0; j < pImage->sSizeY; j++)
166 for(i = 0; i < pImage->sSizeX; i++)
167 {
168 cur_max_color = pSrcMaxPix[(j * pImage->sSizeX) + i];
169 if (cur_max_color == 0)
170 continue;
171
172 min_min_color = 255;
173
174 for(local_j = - radius; local_j <= radius; local_j++)
175 for(local_i = - radius; local_i <= radius; local_i++)
176 {
177 if ((local_i == 0) && (local_j == 0)) continue;
178
179 cur_x = i + local_i;
180 cur_y = j + local_j;
181
182 if ((cur_x < 0) || (cur_y < 0) || (cur_x >= pImage->sSizeX) ||
183 (cur_y >= pImage->sSizeY)) continue;
184
185 cur_min_color = pSrcMinPix[(cur_y * pImage->sSizeX) + cur_x];
186
187 if (cur_min_color < min_min_color)
188 min_min_color = cur_min_color;
189 }
190
191 if ((cur_max_color - min_min_color >= relative_thr) || ((min_min_color == 255) && (options == LEAVE_UNKNOWN_MAX)))
192 pDstPix[(j * pImage->sSizeX) + i] = 255;
193 }
194
195 _CHECK_RESULT_((res = awpReleaseImage(&pImageMax)))
196 _CHECK_RESULT_((res = awpReleaseImage(&pImageMin)))
197
198 *ppImage = pTempImage;
199
200CLEANUP:
201 if (pImageMax != NULL)
202 if (res == AWP_OK)
203 res = awpReleaseImage(&pImageMax);
204 else
205 awpReleaseImage(&pImageMax);
206
207 if (pImageMin != NULL)
208 if (res == AWP_OK)
209 res = awpReleaseImage(&pImageMin);
210 else
211 awpReleaseImage(&pImageMin);
212
213 return res;
214}

324static const AWPDOUBLE sobel_v_mask[9] =
325 {-1,0,1,
326 -2,0,2,
327 -1,0,1};
328static const AWPDOUBLE sobel_h_mask[9] =
329 {1,2,1,

333/*
334 convolutions with Sobel masks on the double image pImage
335*/
336AWPRESULT awpEdgeSobel1(awpImage* pImage, awpImage* pVImage, awpImage* pHImage)
337{
338 AWPRESULT res = AWP_OK;
339 AWPDOUBLE* b = NULL;
340 AWPDOUBLE* v = NULL;
341 AWPDOUBLE* h = NULL;
342 AWPSHORT y = 0;
343 AWPSHORT x = 0;
344 AWPBYTE k = 0;
345 AWPSHORT xx = 0;
346 AWPSHORT yy = 0;
347 AWPDOUBLE sumv = 0;
348 AWPDOUBLE sumh = 0;
349 AWPWORD c = 0;
350 AWPBYTE _nc = 0;
351 AWPWORD _w = 0;
352 AWPWORD _h = 0;
353 _CHECK_RESULT_(res = awpCheckImage(pImage))
354 _CHECK_RESULT_(res = awpCheckImage(pVImage))
355 _CHECK_RESULT_(res = awpCheckImage(pHImage))
356
357 _CHECK_SAME_SIZES(pHImage, pVImage)
358 _CHECK_SAME_SIZES(pImage, pVImage)
359 _CHECK_NUM_CHANNELS(pHImage, pVImage)
360 _CHECK_SAME_TYPE(pHImage, pVImage)
361
362 //-------------------------
363 b = (AWPDOUBLE*)pImage->pPixels;
364 h = (AWPDOUBLE*)pHImage->pPixels;
365 v = (AWPDOUBLE*)pVImage->pPixels;
366
367 _nc = pImage->bChannels;
368 _w = pImage->sSizeX;
369 _h = pImage->sSizeY;
370
371 for (k = 0; k < _nc; k++)
372 {
373 for ( y = 1; y < _h - 1; y++)
374 {
375 for ( x = 1; x < _w - 1; x++)
376 {
377 sumv = 0;
378 sumh = 0;
379 c = 0;
380 for (yy = -1; yy <=1; yy++)
381 {
382 for (xx = -1; xx <= 1; xx++)
383 {
384 sumv += sobel_v_mask[c]*b[k+_nc*(x+xx) +(y+yy)*pImage->sSizeX*_nc];
385 sumh += sobel_h_mask[c]*b[k+_nc*(x+xx) +(y+yy)*pImage->sSizeX*_nc];
386 c++;
387 }
388 }
389 v[k+_nc*x+y*_nc*pImage->sSizeX] = sumv;
390 h[k+_nc*x+y*_nc*pImage->sSizeX] = sumh;
391 }
392 }
393 }
394
395CLEANUP:
396 return res;
397}

402__inline static double _awpAtan(double dx, double dy)
403{
404 AWPDOUBLE t = 0;
405 if (dx == 0)
406 return 0;
407 t = fabs(dy / dx);
408 if (dx > 0 && dy >= 0)
409 return atan(t);
410 else if (dx >=0 && dy < 0)
411 return 1.5*AWP_PI + atan(t);
412 else if (dx < 0 && dy < 0)
413 return AWP_PI + atan(t);
414 else
415 return 0.5*AWP_PI + atan(t);
416}

429AWPRESULT awpEdgeSobel(awpImage* pImage, awpImage* pGradAmpl, awpImage* pGradDir)
430{
431 AWPRESULT res = AWP_OK;
432 awpImage* img_dx = NULL;
433 awpImage* img_dy = NULL;
434 awpImage* img_grad = NULL;
435 awpImage* img_dir = NULL;
436 awpImage* img_src = NULL;
437
438 AWPINT i = 0;
439 AWPINT j = 0;
440 AWPBYTE k = 0;
441 AWPDOUBLE* dx = NULL;
442 AWPDOUBLE* dy = NULL;
443 AWPDOUBLE* grad = NULL;
444 AWPDOUBLE* dir = NULL;
445 AWPINT idx = 0;
446 AWPINT idx0 = 0;
447 _CHECK_RESULT_(res = awpCheckImage(pGradAmpl))
448 _CHECK_RESULT_(res = awpCheckImage(pImage))
449
450 _CHECK_RESULT_(res = awpCreateImage(&img_dx, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, AWP_DOUBLE))
451 _CHECK_RESULT_(res = awpCreateImage(&img_dy, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, AWP_DOUBLE))
452 _CHECK_RESULT_(res = awpCreateImage(&img_grad, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, AWP_DOUBLE))
453 _CHECK_RESULT_(res = awpCopyImage(pImage, &img_src))
454 if (img_src->dwType != AWP_DOUBLE)
455 _CHECK_RESULT_(res = awpConvert(img_src, AWP_CONVERT_TO_DOUBLE))
456
457 // call edge sobel dx-dy convolutoin
458 _CHECK_RESULT_(res = awpEdgeSobel1(img_src, img_dx, img_dy))
459
460 if (pGradDir != NULL)
461 {
462 _CHECK_RESULT_(res = awpCheckImage(pGradDir))
463 _CHECK_SAME_SIZES(pGradDir, pGradAmpl)
464 _CHECK_NUM_CHANNELS(pGradDir, pGradAmpl)
465 _CHECK_SAME_TYPE(pGradDir, pGradAmpl)
466 _CHECK_RESULT_(res = awpCreateImage(&img_dir, pImage->sSizeX, pImage->sSizeY, pImage->bChannels, AWP_DOUBLE))
467 }
468
469 dx = (AWPDOUBLE*)img_dx->pPixels;
470 dy = (AWPDOUBLE*)img_dy->pPixels;
471 grad = (AWPDOUBLE*)img_grad->pPixels;
472 if (pGradDir == NULL)
473 {
474 for (k = 0; k < pImage->bChannels; k++)
475 {
476 for (i = 0; i < pImage->sSizeY; i++)
477 {
478 idx0 = i*pImage->sSizeX*pImage->bChannels;
479 for (j = 0; j < pImage->sSizeX; j++)
480 {
481 grad[idx] = sqrt(dx[idx]*dx[idx] + dy[idx]*dy[idx]);
482 idx = idx0 + pImage->bChannels*j + k;
483 }
484 }
485
486 }
487 }
488 else
489 {
490 dir = (AWPDOUBLE*)img_dir->pPixels;
491 for (k = 0; k < pImage->bChannels; k++)
492 {
493 for (i = 0; i < pImage->sSizeY; i++)
494 {
495 idx0 = i*pImage->sSizeX*pImage->bChannels;
496 for (j = 0; j < pImage->sSizeX; j++)
497 {
498 dir[idx] = _awpAtan(dx[idx], dy[idx]);
499 grad[idx] = sqrt(dx[idx]*dx[idx] + dy[idx]*dy[idx]);
500 idx = idx0 + pImage->bChannels*j + k;
501 }
502 }
503 }
504 }
505 // convert result to desired type
506 switch(pImage->dwType)
507 {
508 case AWP_BYTE:
509 _CHECK_RESULT_(res = awpConvert(img_grad, AWP_CONVERT_TO_BYTE_WITH_NORM));
510 _CONVERT_SOBEL_ANGLE_(AWP_CONVERT_TO_BYTE_WITH_NORM)
511 break;
512 case AWP_SHORT:
513 _CHECK_RESULT_(res = awpConvert(img_grad, AWP_CONVERT_TO_SHORT));
514 _CONVERT_SOBEL_ANGLE_(AWP_CONVERT_TO_SHORT)
515 break;
516 case AWP_FLOAT:
517 _CHECK_RESULT_(res = awpConvert(img_grad, AWP_CONVERT_TO_FLOAT));
518 _CONVERT_SOBEL_ANGLE_(AWP_CONVERT_TO_FLOAT)
519 break;
520 }
521 _CHECK_RESULT_(res = awpCopyImage(img_grad, &pGradAmpl));
522
523 _COPY_SOBEL_RESULT_ANGLE_
524
525CLEANUP:
526 _SAFE_RELEASE_(img_dx);
527 _SAFE_RELEASE_(img_dy);
528 _SAFE_RELEASE_(img_grad);
529 _SAFE_RELEASE_(img_src);
530 _SAFE_RELEASE_(img_dir);
531 return res;
532}
