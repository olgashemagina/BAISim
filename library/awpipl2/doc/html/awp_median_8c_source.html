27AWPRESULT __awpMedian_double(awpImage* src, awpImage* dst, AWPBYTE radius)
28{
29 AWPRESULT res = AWP_OK;
30
31 AWPDOUBLE* _dst = NULL;
32 AWPDOUBLE* _src = NULL;
33 AWPDOUBLE* _buffer = NULL;
34 AWPINT buffer_size = 0;
35 AWPINT median_idx = 0;
36 AWPINT c = 0;
37 AWPSHORT y = 0;
38 AWPSHORT x = 0;
39 AWPBYTE k = 0;
40 AWPSHORT xx = 0;
41 AWPSHORT yy = 0;
42 AWPBYTE _nc = 0;
43 AWPWORD _w = 0;
44 AWPWORD _h = 0;
45 AWPINT sx = radius;
46 AWPINT sy = radius;
47 AWPINT _x = 0;
48 AWPINT _y = 0;
49
50 _dst = (AWPDOUBLE*)dst->pPixels;
51 _src = (AWPDOUBLE*)src->pPixels;
52 _nc = src->bChannels;
53 _w = src->sSizeX;
54 _h = src->sSizeY;
55 buffer_size = (2*radius + 1)*(2*radius + 1)*sizeof(AWPDOUBLE);
56 _buffer = (AWPDOUBLE*)malloc(buffer_size);
57 median_idx = buffer_size/sizeof(AWPDOUBLE) / 2;
58 for (k = 0; k < _nc; k++)
59 {
60 for ( y = 0; y < _h; y++)
61 {
62 for ( x = 0; x < _w; x++)
63 {
64 c = 0;
65 for (yy = y - sy; yy <= y + sy; yy++)
66 {
67 _y = yy < 0 ? abs(yy) : yy;
68 if (yy >= _h)
69 _y =_h - 1;
70 for (xx = x- sx; xx <= x + sx; xx++)
71 {
72 _x = xx < 0 ? abs(xx):xx >= _w ? _w -1 : xx;
73 _buffer[c] =_src[k+_nc*_x +_y*_w*_nc];
74 c++;
75 }
76 }
77 qsort(_buffer, buffer_size/sizeof(AWPDOUBLE), sizeof(AWPDOUBLE), compare);
78 _dst[k+_nc*x+y*_nc*_w] = _buffer[median_idx];
79 }
80 }
81 }
82 _SAFE_FREE_(_buffer)
83 return res;
84}

87 multichannel median filter
88*/
89AWPRESULT awpMedian(awpImage* src, awpImage* dst, AWPBYTE radius)
90{
91 AWPRESULT res = AWP_OK;
92 awpImage* median_src = NULL;
93 awpImage* median_dst = NULL;
94 _CHECK_RESULT_(res = awpCheckImage(src))
95 _CHECK_RESULT_(res = awpCheckImage(dst))
96 _CHECK_SAME_SIZES(src, dst)
97 _CHECK_SAME_TYPE(src, dst)
98 _CHECK_NUM_CHANNELS(src, dst)
99
100 _CHECK_RESULT_(res = awpCreateImage(&median_dst, src->sSizeX, src->sSizeY, src->bChannels, AWP_DOUBLE))
101 _CHECK_RESULT_(res = awpCopyImage(src, &median_src))
102 if (median_src->dwType != AWP_DOUBLE)
103 _CHECK_RESULT_(res = awpConvert(median_src, AWP_CONVERT_TO_DOUBLE))
104
105 _CHECK_RESULT_(res = __awpMedian_double(median_src, median_dst, radius))
106
107 switch (src->dwType)
108 {
109 case AWP_BYTE:
110 _CHECK_RESULT_(res = awpConvert(median_dst, AWP_CONVERT_TO_BYTE_WITH_NORM));
111 break;
112 case AWP_SHORT:
113 _CHECK_RESULT_(res = awpConvert(median_dst, AWP_CONVERT_TO_SHORT));
114 break;
115 case AWP_FLOAT:
116 _CHECK_RESULT_(res = awpConvert(median_dst, AWP_CONVERT_TO_FLOAT));
117 break;
118 }
119 _CHECK_RESULT_(res = awpCopyImage(median_dst, &dst));
120
121CLEANUP:
122 _SAFE_RELEASE_(median_src);
123 _SAFE_RELEASE_(median_dst);
124
125 return res;
126}
