27void __fastcall THsv2DForm::UpdatePicture()
28{
29 m_2Dhst.FreeImages();
30 if (viewSourceHSTAction->Checked)
31 {
32 this->RenederSource();
33 }
34 else if (viewSourceCondHSTAction->Checked)
35 {
36 this->RnderSourceCond();
37 }
38 else if (viewOutsideAction->Checked)
39 {
40 this->RenderOutsideSource();
41 }
42 else if (viewCondOutsideAction->Checked)
43 {
44 this->RenderOutsideCond();
45 }
46 else if (viewPredictorAction->Checked)
47 {
48 this->RenderPredictor();
49 }
50
51 if (m_2Dhst.GetImage() == NULL)
52 {
53 FImage1->Bitmap->Assign(NULL);
54 FImage1->Paint();
55 return;
56 }
57 awpImage* img = NULL;
58 awpCopyImage(m_2Dhst.GetImage(), &img);
59
60 awpConvert(img, AWP_CONVERT_TO_BYTE_WITH_NORM);
61 FImage1->Bitmap->SetAWPImage(img);
62 _AWP_SAFE_RELEASE_(img)
63 FImage1->BestFit();
64
65 mainForm->FImage1->Bitmap->GetAWPImage(&m_src);
66}

68void THsv2DForm::ProcessImage()
69{
70 awpImage* img = m_model.Predict(m_src, double(SpinEdit1->Value)); //MakeMask(m_src, m_2Dhst.GetImage(), double(SpinEdit1->Value));
71 if (img != NULL)
72 {
73 AWPBYTE* src_pix = _AWP_BPIX_(m_src, AWPBYTE)
74 AWPBYTE* img_pix = _AWP_BPIX_(img, AWPBYTE)
75
76 if (!CheckBox2->Checked)
77 mainForm->FImage2->Bitmap->SetAWPImage(img);
78 else
79 {
80 for (int i = 0; i < img->sSizeX*img->sSizeY; i++)
81 {
82 if (img_pix[i] > 0)
83 {
84 src_pix[3*i] = 255;
85 }
86 }
87 mainForm->FImage2->Bitmap->SetAWPImage(m_src);
88 }
89 mainForm->FImage2->BestFit();
90 _AWP_SAFE_RELEASE_(img)
91
92 }
93}
