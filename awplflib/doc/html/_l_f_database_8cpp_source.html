49 {

52

54  return true;
55  if (strExt == ".jpg")
56  return true;
57  if (strExt == ".jpeg")
58  return true;

60  return true;
61  if (strExt == ".bmp")

63  return false;

460  for (int i = 0; i < m_dataFiles.GetCount(); i++)
461  {

463  if (d == NULL)
464  continue;
465  string strImageName = d->GetImageFile();
466  printf("processing: %s\n", strImageName.c_str());
467  TLFImage img;
468  img.LoadFromFile(strImageName.c_str());
469 
470  for (int j = 0; j < d->GetCount(); j++)
471  {
472  TLFDetectedItem* item = d->GetDetectedItem(j);
473  TLFRect* lf_rect = item->GetBounds();
474  awpRect rect = lf_rect->GetRect();
475  detector->Init(img.GetImage());
476  int result = detector->ClassifyRect(rect, NULL, NULL);
477  awpImage* fragment = NULL;
478  awpCopyRect(img.GetImage(), &fragment, &rect);
479  if (fragment == NULL)
480  {
481  printf("failed fragment.\n");
482  continue;
483  }
484  rect.left = 0;
485  rect.top = 0;
486  rect.right = fragment->sSizeX;
487  rect.bottom = fragment->sSizeY;
488  detector->Init(fragment);
489  int result1 = detector->ClassifyRect( rect, NULL, NULL);
490  if (result1 != result || result == 0)
491  printf("failed with result %i.\n", result);
492  else
493  printf("success.\n");
494  }
495 
496  }
