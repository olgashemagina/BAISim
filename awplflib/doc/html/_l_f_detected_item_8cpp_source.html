275 bool TLFDetectedItem::LoadXML(TiXmlElement* parent)
276 {
277  if (parent == NULL)
278  return false;
279  if (strcmp(parent->Value(), GetName()) != 0)
280  return false;
281 
282  this->m_strDetectorName = parent->Attribute("DetectorName");
283  double v;
284  parent->QueryDoubleAttribute("Raiting", &v);
285  this->m_Raiting = v;
286  m_type = parent->Attribute("Type");
287  parent->Attribute("Angle", &this->m_angle);
288  parent->Attribute("Racurs", &this->m_racurs);
289  parent->Attribute("Bw", &this->m_bw);
290  parent->Attribute("Bh", &this->m_bh);
291  if (parent->Attribute("Comment") != NULL)
292  {
293  m_strComment = parent->Attribute("Comment");
294  }
295 
296  awpRect r;
297  int iv;
298  parent->Attribute("left", &iv);
299  r.left = iv;
300  parent->Attribute("top", &iv);
301  r.top = iv;
302  parent->Attribute("right", &iv);
303  r.right = iv;
304  parent->Attribute("bottom", &iv);
305  r.bottom = iv;
306 
307  if (this->m_pRect == NULL)
308  this->m_pRect = new TLFRect();
309 
310  this->m_pRect->SetRect(r);
311 
312  TiXmlElement* child = parent->FirstChildElement();
313  if (child)
314  {
315  if (m_zone == NULL)
316  m_zone = new TLFZone();
317  m_zone->LoadXML(child);
318  }
319 
320  return true;
321 }

351 void TLFDetectedItem::Rotate(awpPoint c, int angle)
352 {
353  TLFRect* rect1 = this->GetBounds();
354  awpRect awpr = rect1->GetRect();
355 
356  double fi = angle*AWP_PI / 180.;
357 
358  // rotate rect
359  awpPoint p;
360  int w, h;
361  p.X = awpr.left;
362  p.Y = awpr.top;
363  w = awpr.right - awpr.left;
364  h = awpr.bottom - awpr.top;
365  p.X = int(c.X + (p.X - c.X)*cos(fi) + (p.Y - c.Y)*sin(fi));
366  p.Y = int(c.Y - (p.X - c.X)*sin(fi) + (p.Y - c.Y)*cos(fi));
367  awpr.left = p.X;
368  awpr.right = p.X + w;
369  awpr.top = p.Y;
370  awpr.bottom = p.Y + h;
371  rect1->SetRect(awpr);
372 }

411 void TLFDetectedItem::Update(ILFDetectEngine* engine, TLFRect* rect)
412 {
413  if (engine == NULL || m_predictor == NULL)
414  return;
415 
416  ILFScanner* s = engine->GetScanner(0);
417 
418  if (rect == NULL && m_predictor->GetPredicted() == NULL)
419  this->m_state = 3;
420  else if (rect == NULL)
421  {
422  TLFRect* predicted = m_predictor->GetPredicted();
423  if (predicted != NULL)
424  {
425  this->m_pRect->SetRect(predicted->GetRect());
426  this->m_state = 2;
427  }
428  else
429  this->m_state = 3;
430  }
431  else
432  {
433  TLFRect* predicted = m_predictor->GetPredicted();
434  if (predicted != NULL)
435  {
436  this->m_pRect->SetRect(predicted->GetRect());
437  m_state = 2;
438  }
439  else
440  {
441  this->m_predictor->Learn(engine, rect);
442  this->m_pRect->SetRect(rect->GetRect());
443  m_state = 1;
444  }
445  }
446 }
