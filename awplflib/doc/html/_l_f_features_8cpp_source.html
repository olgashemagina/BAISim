107 double TLFSFeature::fCalcValue(TLFImage* pImage)
108 {
109  if (pImage == NULL)
110  {
111  m_fValue = 0;
112  return m_fValue;
113  }
114  else
115  {
116  double v1, v2, s, v;
117  s = m_w*m_h;
118  if (s == 0)
119  {
120  m_fValue = 0;
121  return m_fValue;
122  }
123 
124  v1 = pImage->CalcLnSum(this->m_sx, this->m_sy, this->m_w, this->m_h);
125  v2 = pImage->CalcSqSum(this->m_sx, this->m_sy, this->m_w, this->m_h);
126  v1 /= s;
127  v2 /= s;
128  v = v2 - v1*v1;
129  if (v >= 0)
130  {
131  m_fValue = sqrt(v);
132  return m_fValue;
133  }
134  else
135  {
136  m_fValue = 0;
137  return m_fValue;
138  }

499  double s, v1, v2, v3, v4, v5, v6, v7, v8, v9, value, sigma;

599 int TLFLBPFeature::CalcValue(awpImage* pImage, double avg)
600 {
601  if (m_w*m_h == 0 || pImage == NULL)
602  return 0;
603  double tarea = 9.0*m_w*m_h;
604  double area = 1.0 / (double)(m_w*m_h);

609  memset(iv, 0, sizeof(iv));
610 
611  double* pix = (double*)pImage->pPixels;
612  int i = 0;
613 
614  v[0] = CalcSum(pix, m_sx, m_sy, m_w, m_h, pImage->sSizeX) * area;
