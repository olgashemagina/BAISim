128 {
129  if (bin < 0 || bin > this->m_num_bins)

236 bool TLFHistogramm::LoadXML(TiXmlElement* parent)
237 {
238  if (parent == NULL)
239  return false;
240  if (strcmp(parent->Value(), GetName()) != 0)
241  return false;
242  parent->QueryDoubleAttribute("min", &this->m_min);
243  parent->QueryDoubleAttribute("max", &this->m_max);
244  parent->QueryIntAttribute("num", &this->m_num_bins);
245  if (m_num_bins < 0)
246  return false;
247  if (m_data != NULL)
248  free(m_data);
249  if (m_num_bins == 0)
250  return true;
251  m_data = (double*)malloc(m_num_bins*sizeof(double));
252  int c = 0;
253  TLFDouble d(0.);
254  for(TiXmlNode* child = parent->FirstChild(); child; child = child->NextSibling() )
255  {
256  TiXmlElement* e = child->ToElement();
257  if(!d.LoadXML(e))
258  return false;
259  m_data[c] = d.Get();
260  c++;
261  }
262 
263  return true;
264 }
