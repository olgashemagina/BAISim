133 double TLFClusterMaker::Distance(TLFClusterItem* di1, TLFClusterItem* di2)
134 {
135  if (di1 == NULL || di2 == NULL || di1 == di2)
136  return FLT_MAX;
137  awpPoint p1 = di1->GetDi()->GetBounds()->Center();
138  awpPoint p2 = di2->GetDi()->GetBounds()->Center();
139 
140  return sqrt((double)(p1.X - p2.X)*(p1.X-p2.X) + (double)(p1.Y - p2.Y)*(p1.Y - p2.Y));

230 void TLFClusterMaker::Clustering()
231 {
232  this->MakeClusterList();
233  if (this->m_clusterItems.GetCount() == 0)
234  return;
235  awpImage* img = NULL;
236  awpCreateImage(&img, m_clusterItems.GetCount(), m_clusterItems.GetCount(), 1, AWP_DOUBLE);
237  if (img != NULL)
238  {
239  awpFill(img, FLT_MAX);
240  AWPDOUBLE* data = (AWPDOUBLE*)img->pPixels;
241  for (int i = 0; i < img->sSizeY; i++)
242  {
243  TLFClusterItem* d1 = (TLFClusterItem*)m_clusterItems.Get(i);
244  for (int j = 0; j < img->sSizeX; j++)
245  {
246  TLFClusterItem* d2 = (TLFClusterItem*)m_clusterItems.Get(j);
247  data[i*img->sSizeX + j] = Distance(d1,d2);
248  }
249  }
250  m_matrix.SetImage(img);
251  awpReleaseImage(&img);
252 
253  int item_idx = GetUnclustered();
254  m_cluster_count = 0;
255 
256  while (item_idx >= 0)
257  {
258  TLFClusterItem* item = (TLFClusterItem*)m_clusterItems.Get(item_idx);
259  item->SetClusterID(m_cluster_count);
260  while (GetNearleastItemToCluster(m_cluster_count, item_idx) < 12)
261  {
262  item = (TLFClusterItem*)m_clusterItems.Get(item_idx);
263  item->SetClusterID(m_cluster_count);
264  }
265  m_cluster_count++;
266  item_idx = GetUnclustered();
267  }
268  }

285  awpCreateImage(&_2d_histogramm, LF_NUM_CLUSTERS, m_cluster_count, 1, AWP_DOUBLE);
286  AWPDOUBLE* data = (AWPDOUBLE*)_2d_histogramm->pPixels;
287  if (_2d_histogramm != NULL)
288  {
289  for (int i = 0; i < m_clusterItems.GetCount(); i++)
290  {
291  TLFClusterItem* ci = (TLFClusterItem*)m_clusterItems.Get(i);
292  int x, y;
293  x = ci->GetDi()->GetClusterIdx();
294  if (x > 0)
295  {
296  y = ci->GetClusterID();
297  data[y*LF_NUM_CLUSTERS + x]++;
298  }
299  }
301  for (int i = 0; i < m_clusterItems.GetCount(); i++)
302  {
303  TLFClusterItem* ci = (TLFClusterItem*)m_clusterItems.Get(i);
304  if (ci->GetDi() != NULL && ci->GetDi()->GetClusterIdx() == -1)
305  {
306  AWPDOUBLE* data = (AWPDOUBLE*)_2d_histogramm->pPixels;
307  data += (LF_NUM_CLUSTERS*ci->GetClusterID());
308  int n = 0;
309  for (int k = 0; k < LF_NUM_CLUSTERS; k++)
310  {
311  if (data[k] > 0)
312  n++;
313  }
314  if (n == 0)
315  {
316  //assign new cluster_index
317  ci->GetDi()->SetClusterIdx(GetNewClusterIndex());
318  }
319  else
320  {
321  // find nearlest with cluster_index
322  double dist = FLT_MAX;
323  ci->GetDi()->SetClusterIdx(GetnearlestItemWithClusterIndex(m_matrix.GetImage(), ci->GetClusterID(), dist));
324  }
325  }
326  }
327  awpReleaseImage(&_2d_histogramm);
328  }

331 void TLFClusterMaker::UpdateBlobs()
332 {
333  // .
334  for (int i = 0; i < LF_NUM_CLUSTERS; i++)
335  {
336  m_blobs[i].angle = 0;
337  m_blobs[i].cx = 0;
338  m_blobs[i].cy = 0;
339  m_blobs[i].major = 0;
340  m_blobs[i].minor = 0;
341  m_blobs[i].mxx = 0;
342  m_blobs[i].mxy = 0;
343  m_blobs[i].myy = 0;
344  m_blobs[i].area = 0;
345  }
346  //
347  for (int i = 0; i < m_clusterItems.GetCount(); i++)
348  {
349  TLFClusterItem* ci = (TLFClusterItem*)m_clusterItems.Get(i);
350  TLFDetectedItem* di = ci->GetDi();
351  if (di != NULL)
352  {
353  int idx = di->GetClusterIdx();
354  awpPoint p = di->GetBounds()->Center();
355  m_blobs[idx].area += 1;
356  m_blobs[idx].cx += p.X;
357  m_blobs[idx].cy += p.Y;
358  }
359  }
360 
361  for (int i = 0; i < LF_NUM_CLUSTERS; i++)
362  {
363  if (m_blobs[i].area > 0)
364  {
365  m_blobs[i].cx /= m_blobs[i].area;
366  m_blobs[i].cy /= m_blobs[i].area;
367  }
368  }
369 
370  for (int i = 0; i < m_clusterItems.GetCount(); i++)
371  {
372  TLFClusterItem* ci = (TLFClusterItem*)m_clusterItems.Get(i);
373  TLFDetectedItem* di = ci->GetDi();
374  if (di != NULL)
375  {
376  int idx = di->GetClusterIdx();
377  awpPoint p = di->GetBounds()->Center();
378  if (m_blobs[idx].area > 0)
379  {
380  m_blobs[idx].mxx += (p.X - m_blobs[idx].cx)*(p.X - m_blobs[idx].cx);
381  m_blobs[idx].mxy += (p.Y - m_blobs[idx].cy)*(p.X - m_blobs[idx].cx);
382  m_blobs[idx].myy += (p.Y - m_blobs[idx].cy)*(p.Y- m_blobs[idx].cy);
383  }
384  }
385  }
386 
387  for (int i = 0; i < LF_NUM_CLUSTERS; i++)
388  {
389  if (m_blobs[i].area > 0)
390  {
391  m_blobs[i].mxx /= m_blobs[i].area;
392  m_blobs[i].myy /= m_blobs[i].area;
393  // moments.mxy /= moments.area;
394  m_blobs[i].mxy = 0;
395  if (m_blobs[i].mxy == 0 && m_blobs[i].myy <= m_blobs[i].mxx)
396  {
397  m_blobs[i].angle = -AWP_PI / 2;
398  m_blobs[i].major = 4 * sqrt(m_blobs[i].mxx);
399  m_blobs[i].minor = 4 * sqrt(m_blobs[i].myy);
400  }
401  else if (m_blobs[i].mxy == 0 && m_blobs[i].myy > m_blobs[i].mxx)
402  {
403  m_blobs[i].angle = 0;
404  m_blobs[i].major = 4 * sqrt(m_blobs[i].myy);
405  m_blobs[i].minor = 4 * sqrt(m_blobs[i].mxx);
406  }
407  else if (m_blobs[i].myy <= m_blobs[i].mxx)
408  {
409  m_blobs[i].angle = atan(-2 * m_blobs[i].mxy*m_blobs[i].myy - m_blobs[i].mxx + sqrt((m_blobs[i].myy - m_blobs[i].mxx)*(m_blobs[i].myy - m_blobs[i].mxx) + 4 * m_blobs[i].mxy*m_blobs[i].mxy));
410  m_blobs[i].major = sqrt(8 * (m_blobs[i].myy + m_blobs[i].mxx + sqrt((m_blobs[i].myy - m_blobs[i].mxx)*(m_blobs[i].myy - m_blobs[i].mxx) + 4 * m_blobs[i].mxy*m_blobs[i].mxy)));
411  m_blobs[i].minor = sqrt(8 * (m_blobs[i].myy + m_blobs[i].mxx - sqrt((m_blobs[i].myy - m_blobs[i].mxx)*(m_blobs[i].myy - m_blobs[i].mxx) + 4 * m_blobs[i].mxy*m_blobs[i].mxy)));
412 
413  }
414  else if (m_blobs[i].myy > m_blobs[i].mxx)
415  {
416  m_blobs[i].angle = atan(sqrt(m_blobs[i].mxx + m_blobs[i].myy + sqrt((m_blobs[i].mxx - m_blobs[i].myy)*(m_blobs[i].mxx - m_blobs[i].myy) + 4 * m_blobs[i].mxy*m_blobs[i].mxy)) / (-2 * m_blobs[i].mxy + 0.000001));
417  m_blobs[i].major = sqrt(8 * (m_blobs[i].myy + m_blobs[i].mxx + sqrt((m_blobs[i].myy - m_blobs[i].mxx)*(m_blobs[i].myy - m_blobs[i].mxx) + 4 * m_blobs[i].mxy*m_blobs[i].mxy)));
418  m_blobs[i].minor = sqrt(8 * (m_blobs[i].myy + m_blobs[i].mxx - sqrt((m_blobs[i].myy - m_blobs[i].mxx)*(m_blobs[i].myy - m_blobs[i].mxx) + 4 * m_blobs[i].mxy*m_blobs[i].mxy)));
419  }
420 
421  }
422  }
423 
424  for (int i = 0; i < LF_NUM_CLUSTERS; i++)
425  {
426  if (m_blobs[i].area > 0 && m_blobs[i].state == 0)
427  {
428  m_blobs[i].state = 1;
429 
430  // todo: GUID
431  LF_UUID_CREATE(m_blobs[i].id)
432  }
433  else if (m_blobs[i].area > 0 && m_blobs[i].state == 1)
434  {
435  m_blobs[i].state = 2;
436  }
437  else if (m_blobs[i].area == 0 && (m_blobs[i].state == 2 || m_blobs[i].state == 1))
438  {
439  m_blobs[i].state = 3;
440  }
441  else if (m_blobs[i].area == 0 && m_blobs[i].state == 3)
442  {
443  m_blobs[i].state = 0;
444  LF_NULL_UUID_CREATE(m_blobs[i].id)
445  }
446  }
447 }

478 void TLFClusterRecursive::MakeClusters()
479 {
480  if (this->m_detector == NULL)
481  return;
482  TLFTileScanner* scanner = dynamic_cast<TLFTileScanner*>(m_detector->GetScanner());
483  if (scanner == NULL)
484  return;
485  int w = scanner->GetNumX() -2;
486  int h = scanner->GetNumY() -2;
487 
488  for (int i = 0; i < this->m_clusters.GetCount(); i++)
489  {
490  TLFClusterDescr* cd = (TLFClusterDescr*)m_clusters.Get(i);
491  cd->Clear();
492  }
493 
494  for (int i = 0; i < h; i++)
495  {
496  for (int j = 0; j < w; j++)
497  {
498  TLFDetectedItem* di = m_detector->GetItem(i*w + j);
499  if (di != NULL && di->HasObject() && di->GetClusterIdx() > 0)
500  {
501  int idx = di->GetClusterIdx();
502  GetDescr(idx);
503  if (m_descr != NULL)
504  m_descr->IncPower();
505  Search(j, i, idx);
506  }
507  }
508 
509  }
510  for (int i = 0; i < h; i++)
511  {
512  for (int j = 0; j < w; j++)
513  {
514  TLFDetectedItem* di = m_detector->GetItem(i*w + j);
515  if (di != NULL && di->HasObject() && di->GetClusterIdx() == -1)
516  {
517  m_cluster_idx++;
518  int blob_idx = this->GetBlobIndex();
519  //printf("cluster id = %d \t color id = %d\n", m_cluster_idx, blob_idx);
520  m_descr = new TLFClusterDescr(m_cluster_idx, blob_idx);
521  Search(j, i, m_cluster_idx);
522  m_clusters.Add(m_descr);
523  }
524  }
525  }
526 
527  for (int i = 0; i < m_detector->GetNumItems(); i++)
528  {
529  TLFDetectedItem* di = m_detector->GetItem(i);
530  if (di->HasObject() && di->GetClusterIdx() > 0)
531  {
532  int idx = di->GetClusterIdx();
533  GetDescr(idx);
534  if (m_descr && m_descr->GetPower() < 20)
535  di->SetHasObject(false);
536  }
537  }
538 
539  for (int i = m_clusters.GetCount() - 1; i >= 0; i--)
540  {
541  TLFClusterDescr* cd = (TLFClusterDescr*)m_clusters.Get(i);
542  //printf("id = %d \t power = %d\n", cd->GetId(), cd->GetPower());
543  if (cd->GetPower() < 20)
544  {
545 
546  if (cd->GetBlobId() >= 0)
547  m_blobs[cd->GetBlobId()].color = -1;
548  //printf("delete cluster id = %d \t power = %d\n", cd->GetId(), cd->GetPower());
549  m_clusters.Delete(i);
550  }
551  }
552  m_cluster_count = m_clusters.GetCount();
553  MakeClusterList();
554  UpdateBlobs();
555 
556 }
